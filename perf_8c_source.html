104 void gplot_init(const XCSF *xcsf)
105 {
106  // generate file name for writing performance based on the current date-time
107  time_t t = time(NULL);
108  struct tm tm = *localtime(&t);
109  sprintf(fname, "out/%04d-%02d-%02d-%02d%02d%02d.dat", tm.tm_year + 1900,
110  tm.tm_mon + 1, tm.tm_mday, tm.tm_hour, tm.tm_min, tm.tm_sec);
111  // generate Gnuplot title
112  char title[100];
113  gplot_title(xcsf, title);
114  // open file for writing performance
115  fout = fopen(fname, "wt");
116  if(fout == 0) {
117  printf("Error opening file: %s. %s.\n", fname, strerror(errno));
118  exit(EXIT_FAILURE);
119  }
120  // execute Gnuplot
121  gp = popen("gnuplot -persistent", "w");
122  if(gp != NULL) {
123  fprintf(gp, "set terminal wxt noraise enhanced font 'Arial,12'\n");
124  fprintf(gp, "set grid\n");
125  fprintf(gp, "set border linewidth 1\n");
126  fprintf(gp, "set title \"%s\"\n", title);
127  fprintf(gp, "set xlabel 'Trials'\n");
128  if(xcsf->num_actions < 2) {
129  fprintf(gp, "set ylabel 'System Error'\n");
130  }
131  fprintf(gp, "set style line 1 lt -1 lw 1 ps 1 lc rgb 'red'\n");
132  fprintf(gp, "set style line 2 lt -1 lw 1 ps 1 lc rgb 'blue'\n");
133  }
134  else {
135  printf("Error starting gnuplot\n");
136  }
137 }

144 static void gplot_title(const XCSF *xcsf, char *title)
145 {
146  char buffer[20];
147  title[0] = '\0';
148  switch(xcsf->COND_TYPE) {
149  case COND_TYPE_DUMMY: strcat(title, " dummy cond"); break;
150  case COND_TYPE_HYPERRECTANGLE: strcat(title, " hyperrectangle cond"); break;
151  case COND_TYPE_HYPERELLIPSOID: strcat(title, " hyperellipsoid cond"); break;
152  case COND_TYPE_NEURAL: strcat(title, " neural cond"); break;
153  case COND_TYPE_GP: strcat(title, " tree-GP cond"); break;
154  case COND_TYPE_DGP: strcat(title, " graph-DGP cond"); break;
155  case COND_TYPE_TERNARY: strcat(title, " ternary cond"); break;
156  case RULE_TYPE_DGP: strcat(title, " graph-DGP rules"); break;
157  case RULE_TYPE_NEURAL: strcat(title, " neural rules"); break;
158  }
159  if(xcsf->COND_TYPE < RULE_TYPE_DGP) {
160  strcat(title, ", action integer");
161  }
162  switch(xcsf->PRED_TYPE) {
163  case PRED_TYPE_CONSTANT: strcat(title, ", constant pred"); break;
164  case PRED_TYPE_NLMS_LINEAR: strcat(title, ", linear nlms"); break;
165  case PRED_TYPE_NLMS_QUADRATIC: strcat(title, ", quadratic nlms"); break;
166  case PRED_TYPE_RLS_LINEAR: strcat(title, ", linear rls"); break;
167  case PRED_TYPE_RLS_QUADRATIC: strcat(title, ", quadratic rls"); break;
168  case PRED_TYPE_NEURAL: strcat(title, ", neural pred"); break;
169  }
170  if(xcsf->SAM_NUM > 0) {
171  strcat(title, ", SAM");
172  }
173  sprintf(buffer, ", P=%d", xcsf->POP_SIZE);
174  strcat(title, buffer);
175 }

198 static void gplot_draw(const XCSF *xcsf, _Bool test_error)
199 {
200  (void)xcsf;
201  if(gp != NULL) {
202  if(xcsf->num_actions < 2) {
203  // regression
204  fprintf(gp, "plot '%s' using 1:2 title 'Train Error' w lp ls 1 pt 4 pi 50, ", fname);
205  if(test_error) {
206  fprintf(gp, "'%s' using 1:3 title 'Test Error' w lp ls 2 pt 8 pi 50", fname);
207  }
208  }
209  else {
210  // reinforcement learning
211  fprintf(gp, "plot '%s' using 1:2 title 'Performance' w lp ls 1 pt 4 pi 50, ", fname);
212  if(test_error) {
213  fprintf(gp, "'%s' using 1:3 title 'Error' w lp ls 2 pt 8 pi 50", fname);
214  }
215  }
216  fprintf(gp,"\nreplot\n");
217  fflush(gp);
218  }
219  else {
220  printf("Error writing to gnuplot\n");
221  }
