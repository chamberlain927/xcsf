76 void gplot_init(XCSF *xcsf)
77 {
78  // file name for writing performance uses the current date-time
79  time_t t = time(NULL);
80  struct tm tm = *localtime(&t);
81  sprintf(fname, "out/%04d-%02d-%02d-%02d%02d%02d.dat", tm.tm_year + 1900,
82  tm.tm_mon + 1, tm.tm_mday, tm.tm_hour, tm.tm_min, tm.tm_sec);
83 
84  // create file for writing performance
85  fout = fopen(fname, "wt");
86  if(fout == 0) {
87  printf("Error opening file: %s. %s.\n", fname, strerror(errno));
88  exit(EXIT_FAILURE);
89  }
90 
91  // set gnuplot title
92  char buffer[20];
93  char title[100];
94  title[0] = '\0';
95 
96  switch(xcsf->COND_TYPE) {
97  case -1: strcat(title, " dummy cond"); break;
98  case 0: strcat(title, " hyperrectangle cond"); break;
99  case 1: strcat(title, " hyperellipsoid cond"); break;
100  case 2: strcat(title, " neural cond"); break;
101  case 3: strcat(title, " tree-GP cond"); break;
102  case 4: strcat(title, " graph-DGP cond"); break;
103  case 11: strcat(title, " graph-DGP rules"); break;
104  case 12: strcat(title, " neural rules"); break;
105  }
106 
107  if(xcsf->COND_TYPE < 10) {
108  switch(xcsf->PRED_TYPE) {
109  case 0: strcat(title, ", linear nlms"); break;
110  case 1: strcat(title, ", quadratic nlms"); break;
111  case 2: strcat(title, ", linear rls"); break;
112  case 3: strcat(title, ", quadratic rls"); break;
113  case 4: strcat(title, ", neural pred"); break;
114  }
115  }
116 
117  if(xcsf->SAM_NUM > 0) {
118  strcat(title, ", SAM");
119  }
120 
121  sprintf(buffer, ", P=%d", xcsf->POP_SIZE);
122  strcat(title, buffer);
123 
124  // execute gnuplot
125 #ifdef _WIN32

127 #else

129 #endif
130  if(gp != NULL) {
131  fprintf(gp, "set terminal wxt noraise enhanced font 'Arial,12'\n");
132  fprintf(gp, "set grid\n");

135  //fprintf(gp, "set nokey\n");
136  fprintf(gp, "set xlabel 'Trials'\n");
137  if(xcsf->num_actions < 2) {
138  fprintf(gp, "set ylabel 'System Error'\n");
139  }
140  fprintf(gp, "set style line 1 lt -1 lw 1 ps 1 lc rgb 'red'\n");
141  fprintf(gp, "set style line 2 lt -1 lw 1 ps 1 lc rgb 'blue'\n");
142  }
143  else {
144  printf("Error starting gnuplot\n");
145  }
146 }
