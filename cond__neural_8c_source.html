50 void cond_neural_init(XCSF *xcsf, CL *c)
51 {
52  COND_NEURAL *new = malloc(sizeof(COND_NEURAL));
53  neural_init(xcsf, &new->net);
54 
55  // weights
56  uint32_t lopt = 0;
57  if(xcsf->COND_EVOLVE_WEIGHTS) {
58  lopt |= LAYER_EVOLVE_WEIGHTS;
59  }
60  // neurons
61  int hmax = fmax(xcsf->COND_MAX_HIDDEN_NEURONS, 1);
62  int hinit = xcsf->COND_NUM_HIDDEN_NEURONS;
63  if(hinit < 1) {
64  hinit = irand_uniform(1, hmax);
65  }
66  if(hmax < hinit) {
67  hmax = hinit;
68  }
69  if(xcsf->COND_EVOLVE_NEURONS) {
70  lopt |= LAYER_EVOLVE_NEURONS;
71  }
72  else {
73  hmax = hinit;
74  }
75  // functions
76  int f = xcsf->COND_HIDDEN_NEURON_ACTIVATION;
77  if(xcsf->COND_EVOLVE_FUNCTIONS) {
78  lopt |= LAYER_EVOLVE_FUNCTIONS;
79  }
80 
81  // hidden layer
82  LAYER *l = neural_layer_connected_init(xcsf, xcsf->num_x_vars, hinit, hmax, f, lopt);
83  neural_layer_insert(xcsf, &new->net, l, 0);
84 
85  // output layer
86  lopt &= ~LAYER_EVOLVE_NEURONS; // never evolve the number of output neurons
87  l = neural_layer_connected_init(xcsf, hmax, 1, 1, LOGISTIC, lopt);
88  neural_layer_insert(xcsf, &new->net, l, 1);
89 
90  c->cond = new;
91 }

151 _Bool cond_neural_general(XCSF *xcsf, CL *c1, CL *c2)
152 {
153  if(c1->exp < xcsf->THETA_SUB || c2->exp < xcsf->THETA_SUB) {
154  return false;
155  }
156  for(int i = 0; i < xcsf->THETA_SUB; i++) {
157  int i1 = (c1->exp + i) % xcsf->THETA_SUB;
158  int i2 = (c2->exp + i) % xcsf->THETA_SUB;
159  if(c1->mhist[i1] == false && c2->mhist[i2] == true) {
160  return false;
161  }
162  }
163  return true;
164 }
