51 void cond_neural_init(const XCSF *xcsf, CL *c)
52 {
53  COND_NEURAL *new = malloc(sizeof(COND_NEURAL));
54  neural_init(xcsf, &new->net);
55 
56  uint32_t lopt = cond_neural_lopt(xcsf);
57  int hmax = fmax(xcsf->COND_MAX_HIDDEN_NEURONS, 1);
58  int hinit = xcsf->COND_NUM_HIDDEN_NEURONS;
59  if(hinit < 1) {
60  hinit = irand_uniform(1, hmax);
61  }
62  if(hmax < hinit) {
63  hmax = hinit;
64  }
65  if(!xcsf->COND_EVOLVE_NEURONS) {
66  hmax = hinit;
67  }
68 
69  // hidden layer
70  int f = xcsf->COND_HIDDEN_ACTIVATION;
71  LAYER *l = neural_layer_connected_init(xcsf, xcsf->num_x_vars, hinit, hmax, f, lopt);
72  neural_layer_insert(xcsf, &new->net, l, 0);
73 
74  // output layer
75  f = xcsf->COND_OUTPUT_ACTIVATION;
76  lopt &= ~LAYER_EVOLVE_NEURONS; // never evolve the number of output neurons
77  l = neural_layer_connected_init(xcsf, hinit, 1, 1, f, lopt);
78  neural_layer_insert(xcsf, &new->net, l, 1);
79 
80  c->cond = new;
81 }

83 static uint32_t cond_neural_lopt(const XCSF *xcsf)
84 {
85  uint32_t lopt = 0;
86  if(xcsf->COND_EVOLVE_WEIGHTS) {

93  lopt |= LAYER_EVOLVE_FUNCTIONS;
94  }
95  return lopt;
96 }
