86 void neural_layer_noise_forward(const XCSF *xcsf, LAYER *l, const double *input)
87 {
88  if(!xcsf->train) {
89  for(int i = 0; i < l->num_inputs; i++) {
90  l->output[i] = input[i];
91  }
92  }
93  else {
94  for(int i = 0; i < l->num_inputs; i++) {
95  l->rand[i] = rand_uniform(0,1);
96  if(l->rand[i] < l->probability) {
97  l->output[i] = input[i] + rand_normal(0, l->scale);
98  }
99  else {
100  l->output[i] = input[i];
101  }
102  }
103  }
104 }
