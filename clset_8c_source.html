88 static void clset_pop_del(XCSF *xcsf, SET *kset)
89 {
90  double avg_fit = clset_total_fit(xcsf, &xcsf->pset) / xcsf->pset.num;
91  double total = 0;
92  for(const CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
93  total += cl_del_vote(xcsf, iter->cl, avg_fit);
94  }
95  CLIST *del = NULL;
96  CLIST *delprev = NULL;
97  int delsize = 0;
98  for(int i = 0; i < 2; i++) {
99  double p = rand_uniform(0,total);
100  double sum = 0;
101  CLIST *prev = NULL;
102  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
103  sum += cl_del_vote(xcsf, iter->cl, avg_fit);
104  if(sum > p) {
105  int size = cl_cond_size(xcsf, iter->cl) + cl_pred_size(xcsf, iter->cl);
106  if(del == NULL || size > delsize) {
107  del = iter;
108  delprev = prev;
109  delsize = size;
110  }
111  break;
112  }
113  prev = iter;
114  }
115  }
116  // decrement numerosity
117  (del->cl->num)--;
118  (xcsf->pset.num)--;
119  // macro classifier must be deleted
120  if(del->cl->num == 0) {
121  clset_add(xcsf, kset, del->cl);
122  (xcsf->pset.size)--;
123  if(delprev == NULL) {
124  xcsf->pset.list = del->next;
125  }
126  else {
127  delprev->next = del->next;
128  }
129  free(del);
130  }
131 }

156 void clset_match(XCSF *xcsf, SET *mset, SET *kset, const double *x)
157 {
158  _Bool *act_covered = calloc(xcsf->num_actions, sizeof(_Bool));
159 #ifdef PARALLEL_MATCH
160  // prepare for parallel processing of matching conditions
161  CLIST *blist[xcsf->pset.size];
162  int j = 0;
163  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
164  blist[j] = iter;
165  j++;
166  }
167  // update current matching conditions setting m flags in parallel
168 #pragma omp parallel for
169  for(int i = 0; i < xcsf->pset.size; i++) {
170  cl_match(xcsf, blist[i]->cl, x);
171  }
172  // build match set list in series
173  for(int i = 0; i < xcsf->pset.size; i++) {
174  if(cl_m(xcsf, blist[i]->cl)) {
175  clset_add(xcsf, mset, blist[i]->cl);
176  int action = cl_action(xcsf, blist[i]->cl, x);
177  act_covered[action] = true;
178  }
179  }
180 #else
181  // update matching conditions and build match set list in series
182  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
183  if(cl_match(xcsf, iter->cl, x)) {
184  clset_add(xcsf, mset, iter->cl);
185  int action = cl_action(xcsf, iter->cl, x);
186  act_covered[action] = true;
187  }
188  }
189 #endif
190  // perform covering if all actions are not represented
191  clset_cover(xcsf, mset, kset, x, act_covered);
192  free(act_covered);
193 }

203 static void clset_cover(XCSF *xcsf, SET *mset, SET *kset, const double *x, _Bool *act_covered)
204 {
205  int attempts = 0;
206  _Bool again;
207  do {
208  again = false;
209  for(int i = 0; i < xcsf->num_actions; i++) {
210  if(!act_covered[i]) {
211  // new classifier with matching condition and action
212  CL *new = malloc(sizeof(CL));
213  cl_init(xcsf, new, (mset->num)+1, xcsf->time);
214  cl_cover(xcsf, new, x, i);
215  clset_add(xcsf, &xcsf->pset, new);
216  clset_add(xcsf, mset, new);
217  act_covered[i] = true;
218  }
219  }
220  // enforce population size
221  int prev_psize = xcsf->pset.size;
222  clset_pop_enforce_limit(xcsf, kset);
223  // if a macro classifier was deleted, remove any deleted rules from the match set
224  if(prev_psize > xcsf->pset.size) {
225  int prev_msize = mset->size;
226  clset_validate(xcsf, mset);
227  // if the deleted classifier was in the match set,
228  // check if an action is now not covered
229  if(prev_msize > mset->size) {
230  for(int i = 0; i < xcsf->num_actions; i++) {
231  if(!clset_action_covered(xcsf, mset, i)) {
232  act_covered[i] = false;
233  again = true;
234  }
235  }
236  }
237  }
238  attempts++;
239  if(attempts > MAX_COVER) {
240  printf("Error: maximum covering attempts (%d) exceeded\n", MAX_COVER);
241  exit(EXIT_FAILURE);
242  }
243 
244  } while(again);
245 }

254 void clset_pred(const XCSF *xcsf, const SET *set, const double *x, double *p)
255 {
256  double *presum = calloc(xcsf->num_y_vars, sizeof(double));
257  double fitsum = 0;
258 #ifdef PARALLEL_PRED
259  CLIST *blist[set->size];
260  int j = 0;
261  for(CLIST *iter = set->list; iter != NULL; iter = iter->next) {
262  blist[j] = iter;
263  j++;
264  }
265 #pragma omp parallel for reduction(+:presum[:xcsf->num_y_vars],fitsum)
266  for(int i = 0; i < set->size; i++) {
267  const double *predictions = cl_predict(xcsf, blist[i]->cl, x);
268  for(int var = 0; var < xcsf->num_y_vars; var++) {
269  presum[var] += predictions[var] * blist[i]->cl->fit;
270  }
271  fitsum += blist[i]->cl->fit;
272  }
273 #pragma omp parallel for
274  for(int var = 0; var < xcsf->num_y_vars; var++) {
275  p[var] = presum[var]/fitsum;
276  }
277 #else
278  for(CLIST *iter = set->list; iter != NULL; iter = iter->next) {
279  const double *predictions = cl_predict(xcsf, iter->cl, x);
280  for(int var = 0; var < xcsf->num_y_vars; var++) {
281  presum[var] += predictions[var] * iter->cl->fit;
282  }
283  fitsum += iter->cl->fit;
284  }
285  for(int var = 0; var < xcsf->num_y_vars; var++) {
286  p[var] = presum[var]/fitsum;
287  }
288 #endif
289  free(presum);
290 }

389 {
390  double acc_sum = 0;

393  int i = 0;
394  for(const CLIST *iter = set->list; iter != NULL; iter = iter->next) {
395  accs[i] = cl_acc(xcsf, iter->cl);
396  acc_sum += accs[i] * iter->cl->num;
397  i++;
398  }
399  // update fitnesses
400  i = 0;
401  for(const CLIST *iter = set->list; iter != NULL; iter = iter->next) {
402  cl_update_fit(xcsf, iter->cl, acc_sum, accs[i]);
403  i++;
404  }

413 static void clset_subsumption(XCSF *xcsf, SET *set, SET *kset)
414 {
415  CL *s = NULL;
416  const CLIST *iter;
417  // find the most general subsumer in the set
418  for(iter = set->list; iter != NULL; iter = iter->next) {
419  CL *c = iter->cl;
420  if(cl_subsumer(xcsf, c) && (s == NULL || cl_general(xcsf, c, s))) {
421  s = c;
422  }
423  }
424  // subsume the more specific classifiers in the set
425  if(s != NULL) {
426  iter = set->list;
427  while(iter != NULL) {
428  CL *c = iter->cl;
429  iter = iter->next;
430  if(s != c && cl_general(xcsf, s, c)) {
431  s->num += c->num;
432  c->num = 0;
433  clset_add(xcsf, kset, c);
434  clset_validate(xcsf, set);
435  clset_validate(xcsf, &xcsf->pset);
436  }
437  }
438  }
439 }

446 void clset_validate(const XCSF *xcsf, SET *set)
447 {
448  (void)xcsf;
449  set->size = 0;
450  set->num = 0;
451  CLIST *prev = NULL;
452  CLIST *iter = set->list;
453  while(iter != NULL) {
454  if(iter->cl == NULL || iter->cl->num == 0) {
455  if(prev == NULL) {
456  set->list = iter->next;
457  free(iter);
458  iter = set->list;
459  }
460  else {
461  prev->next = iter->next;
462  free(iter);
463  iter = prev->next;
464  }
465  }
466  else {
467  set->size++;
468  set->num += iter->cl->num;
469  prev = iter;
470  iter = iter->next;
471  }
472  }
473 }
