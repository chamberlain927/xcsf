88 static void clset_pop_del(XCSF *xcsf, SET *kset)
89 {
90  CLIST *del = NULL;
91  CLIST *delprev = NULL;
92  // select any rules that never match
93  clset_pop_never_match(xcsf, &del, &delprev);
94  // if none found, select a rule using roulette wheel
95  if(del == NULL) {
96  clset_pop_roulette(xcsf, &del, &delprev);
97  }
98  // decrement numerosity
99  (del->cl->num)--;
100  (xcsf->pset.num)--;
101  // macro classifier must be deleted
102  if(del->cl->num == 0) {
103  clset_add(xcsf, kset, del->cl);
104  (xcsf->pset.size)--;
105  if(delprev == NULL) {
106  xcsf->pset.list = del->next;
107  }
108  else {
109  delprev->next = del->next;
110  }
111  free(del);
112  }
113 }

121 static void clset_pop_never_match(const XCSF *xcsf, CLIST **del, CLIST **delprev)
122 {
123  CLIST *prev = NULL;
124  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
125  if(iter->cl->mtotal == 0 && iter->cl->age > xcsf->M_PROBATION) {
126  *del = iter;
127  *delprev = prev;
128  break;
129  }
130  prev = iter;
131  }

134 /*
135  * @brief Selects a classifier from the population for deletion via roulette wheel.
136  * @param xcsf The XCSF data structure.
137  * @param del A pointer to the rule to be deleted (set by this function).
138  * @param delprev A pointer to the rule previous to the one being deleted (set by this function).
139  *
140  * @details Two classifiers are selected using roulete wheel selection with the
141  * deletion vote and the one with the largest condition + prediction size is
142  * chosen. For fixed-length representations the effect is the same as one
143  * roulete spin.
144  */
145 static void clset_pop_roulette(const XCSF *xcsf, CLIST **del, CLIST **delprev)
146 {
147  double avg_fit = clset_total_fit(xcsf, &xcsf->pset) / xcsf->pset.num;
148  double total = 0;
149  for(const CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
150  total += cl_del_vote(xcsf, iter->cl, avg_fit);
151  }
152  int delsize = 0;
153  for(int i = 0; i < 2; i++) {
154  double p = rand_uniform(0,total);
155  double sum = 0;
156  CLIST *prev = NULL;
157  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
158  sum += cl_del_vote(xcsf, iter->cl, avg_fit);
159  if(sum > p) {
160  int size = cl_cond_size(xcsf, iter->cl) + cl_pred_size(xcsf, iter->cl);
161  if(*del == NULL || size > delsize) {
162  *del = iter;
163  *delprev = prev;
164  delsize = size;
165  }
166  break;
167  }
168  prev = iter;
169  }
170  }
171 }

196 void clset_match(XCSF *xcsf, SET *mset, SET *kset, const double *x)
197 {
198  _Bool *act_covered = calloc(xcsf->num_actions, sizeof(_Bool));
199 #ifdef PARALLEL_MATCH
200  // prepare for parallel processing of matching conditions
201  CLIST *blist[xcsf->pset.size];
202  int j = 0;
203  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
204  blist[j] = iter;
205  j++;
206  }
207  // update current matching conditions setting m flags in parallel
208 #pragma omp parallel for
209  for(int i = 0; i < xcsf->pset.size; i++) {
210  cl_match(xcsf, blist[i]->cl, x);
211  }
212  // build match set list in series
213  for(int i = 0; i < xcsf->pset.size; i++) {
214  if(cl_m(xcsf, blist[i]->cl)) {
215  clset_add(xcsf, mset, blist[i]->cl);
216  int action = cl_action(xcsf, blist[i]->cl, x);
217  act_covered[action] = true;
218  }
219  }
220 #else
221  // update matching conditions and build match set list in series
222  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
223  if(cl_match(xcsf, iter->cl, x)) {
224  clset_add(xcsf, mset, iter->cl);
225  int action = cl_action(xcsf, iter->cl, x);
226  act_covered[action] = true;
227  }
228  }
229 #endif
230  // perform covering if all actions are not represented
231  clset_cover(xcsf, mset, kset, x, act_covered);
232  free(act_covered);
233  // update statistics
234  xcsf->msetsize += (mset->size - xcsf->msetsize) * (10 / (double) xcsf->PERF_AVG_TRIALS);
235  xcsf->mfrac += (clset_mfrac(xcsf) - xcsf->mfrac) * (10 / (double) xcsf->PERF_AVG_TRIALS);
236 }

246 static void clset_cover(XCSF *xcsf, SET *mset, SET *kset, const double *x, _Bool *act_covered)
247 {
248  int attempts = 0;
249  _Bool again;
250  do {
251  again = false;
252  for(int i = 0; i < xcsf->num_actions; i++) {
253  if(!act_covered[i]) {
254  // new classifier with matching condition and action
255  CL *new = malloc(sizeof(CL));
256  cl_init(xcsf, new, (mset->num)+1, xcsf->time);
257  cl_cover(xcsf, new, x, i);
258  clset_add(xcsf, &xcsf->pset, new);
259  clset_add(xcsf, mset, new);
260  act_covered[i] = true;
261  }
262  }
263  // enforce population size
264  int prev_psize = xcsf->pset.size;
265  clset_pop_enforce_limit(xcsf, kset);
266  // if a macro classifier was deleted, remove any deleted rules from the match set
267  if(prev_psize > xcsf->pset.size) {
268  int prev_msize = mset->size;
269  clset_validate(xcsf, mset);
270  // if the deleted classifier was in the match set,
271  // check if an action is now not covered
272  if(prev_msize > mset->size) {
273  for(int i = 0; i < xcsf->num_actions; i++) {
274  if(!clset_action_covered(xcsf, mset, i)) {
275  act_covered[i] = false;
276  again = true;
277  }
278  }
279  }
280  }
281  attempts++;
282  if(attempts > MAX_COVER) {
283  printf("Error: maximum covering attempts (%d) exceeded\n", MAX_COVER);
284  exit(EXIT_FAILURE);
285  }
286 
287  } while(again);
288 }

297 void clset_pred(const XCSF *xcsf, const SET *set, const double *x, double *p)
298 {
299  double *presum = calloc(xcsf->num_y_vars, sizeof(double));
300  double fitsum = 0;
301 #ifdef PARALLEL_PRED
302  CLIST *blist[set->size];
303  int j = 0;
304  for(CLIST *iter = set->list; iter != NULL; iter = iter->next) {
305  blist[j] = iter;
306  j++;
307  }
308 #pragma omp parallel for reduction(+:presum[:xcsf->num_y_vars],fitsum)
309  for(int i = 0; i < set->size; i++) {
310  const double *predictions = cl_predict(xcsf, blist[i]->cl, x);
311  for(int var = 0; var < xcsf->num_y_vars; var++) {
312  presum[var] += predictions[var] * blist[i]->cl->fit;
313  }
314  fitsum += blist[i]->cl->fit;
315  }
316 #pragma omp parallel for
317  for(int var = 0; var < xcsf->num_y_vars; var++) {
318  p[var] = presum[var]/fitsum;
319  }
320 #else
321  for(CLIST *iter = set->list; iter != NULL; iter = iter->next) {
322  const double *predictions = cl_predict(xcsf, iter->cl, x);
323  for(int var = 0; var < xcsf->num_y_vars; var++) {
324  presum[var] += predictions[var] * iter->cl->fit;
325  }
326  fitsum += iter->cl->fit;
327  }
328  for(int var = 0; var < xcsf->num_y_vars; var++) {
329  p[var] = presum[var]/fitsum;
330  }
331 #endif
332  free(presum);
333 }

456 static void clset_subsumption(XCSF *xcsf, SET *set, SET *kset)
457 {
458  // find the most general subsumer in the set
459  CL *s = NULL;
460  for(const CLIST *iter = set->list; iter != NULL; iter = iter->next) {
461  CL *c = iter->cl;
462  if(cl_subsumer(xcsf, c) && (s == NULL || cl_general(xcsf, c, s))) {
463  s = c;
464  }
465  }
466  // subsume the more specific classifiers in the set
467  if(s != NULL) {
468  const CLIST *iter = set->list;
469  while(iter != NULL) {
470  CL *c = iter->cl;
471  iter = iter->next;
472  if(s != c && cl_general(xcsf, s, c)) {
473  s->num += c->num;
474  c->num = 0;
475  clset_add(xcsf, kset, c);
476  clset_validate(xcsf, set);
477  clset_validate(xcsf, &xcsf->pset);
478  }
479  }
480  }
481 }

488 void clset_validate(const XCSF *xcsf, SET *set)
489 {
490  (void)xcsf;
491  set->size = 0;
492  set->num = 0;
493  CLIST *prev = NULL;
494  CLIST *iter = set->list;
495  while(iter != NULL) {
496  if(iter->cl == NULL || iter->cl->num == 0) {
497  if(prev == NULL) {
498  set->list = iter->next;
499  free(iter);
500  iter = set->list;
501  }
502  else {
503  prev->next = iter->next;
504  free(iter);
505  iter = prev->next;
506  }
507  }
508  else {
509  set->size++;
510  set->num += iter->cl->num;
511  prev = iter;
512  iter = iter->next;
513  }
514  }
515 }

727 double clset_mfrac(const XCSF *xcsf)
728 {
729  double mfrac = 0;
730  // most general rule below EPS_0
731  for(const CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
732  double e = iter->cl->err;
733  if(e < xcsf->EPS_0 && iter->cl->exp > 1 / xcsf->BETA) {
734  double m = cl_mfrac(xcsf, iter->cl);
735  if(m > mfrac) {
736  mfrac = m;
737  }
738  }
739  }
740  // lowest error rule
741  if(mfrac == 0) {
742  double error = DBL_MAX;
743  for(const CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
744  double e = iter->cl->err;
745  if(e < error && iter->cl->exp > 1 / xcsf->BETA) {
746  mfrac = cl_mfrac(xcsf, iter->cl);
747  error = e;
748  }
749  }
750  }
751  return mfrac;
752 }
