80 static void clset_pop_del(XCSF *xcsf)
81 {
82  CLIST *del = NULL;
83  CLIST *delprev = NULL;
84  // select any rules that never match
85  clset_pop_never_match(xcsf, &del, &delprev);
86  // if none found, select a rule using roulette wheel
87  if(del == NULL) {
88  clset_pop_roulette(xcsf, &del, &delprev);
89  }
90  // decrement numerosity
91  (del->cl->num)--;
92  (xcsf->pset.num)--;
93  // macro classifier must be deleted
94  if(del->cl->num == 0) {
95  clset_add(&xcsf->kset, del->cl);
96  (xcsf->pset.size)--;
97  if(delprev == NULL) {
98  xcsf->pset.list = del->next;
99  }
100  else {
101  delprev->next = del->next;
102  }
103  free(del);
104  }
105 }

113 static void clset_pop_never_match(const XCSF *xcsf, CLIST **del, CLIST **delprev)
114 {
115  CLIST *prev = NULL;
116  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
117  if(iter->cl->mtotal == 0 && iter->cl->age > xcsf->M_PROBATION) {

126 /*
127  * @brief Selects a classifier from the population for deletion via roulette wheel.
128  * @param xcsf The XCSF data structure.
129  * @param del A pointer to the rule to be deleted (set by this function).
130  * @param delprev A pointer to the rule previous to the one being deleted (set by this function).
131  *
132  * @details Two classifiers are selected using roulete wheel selection with the
133  * deletion vote and the one with the largest condition + prediction size is
134  * chosen. For fixed-length representations the effect is the same as one
135  * roulete spin.
136  */
137 static void clset_pop_roulette(const XCSF *xcsf, CLIST **del, CLIST **delprev)
138 {
139  double avg_fit = clset_total_fit(&xcsf->pset) / xcsf->pset.num;
140  double total = 0;
141  for(const CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
142  total += cl_del_vote(xcsf, iter->cl, avg_fit);
143  }
144  int delsize = 0;
145  for(int i = 0; i < 2; i++) {
146  double p = rand_uniform(0,total);
147  double sum = 0;
148  CLIST *prev = NULL;
149  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
150  sum += cl_del_vote(xcsf, iter->cl, avg_fit);
151  if(sum > p) {
152  int size = cl_cond_size(xcsf, iter->cl) + cl_pred_size(xcsf, iter->cl);
153  if(*del == NULL || size > delsize) {
154  *del = iter;
155  *delprev = prev;
156  delsize = size;
157  }
158  break;
159  }
160  prev = iter;
161  }
162  }
163 }

184 void clset_match(XCSF *xcsf, const double *x)
185 {
186  _Bool *act_covered = calloc(xcsf->num_actions, sizeof(_Bool));
187 #ifdef PARALLEL_MATCH
188  // prepare for parallel processing of matching conditions
189  CLIST *blist[xcsf->pset.size];
190  int j = 0;
191  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
192  blist[j] = iter;
193  j++;
194  }
195  // update current matching conditions setting m flags in parallel
196 #pragma omp parallel for
197  for(int i = 0; i < xcsf->pset.size; i++) {
198  cl_match(xcsf, blist[i]->cl, x);
199  }
200  // build match set list in series
201  for(int i = 0; i < xcsf->pset.size; i++) {
202  if(cl_m(xcsf, blist[i]->cl)) {
203  clset_add(&xcsf->mset, blist[i]->cl);
204  int action = cl_action(xcsf, blist[i]->cl, x);
205  act_covered[action] = true;
206  }
207  }
208 #else
209  // update matching conditions and build match set list in series
210  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
211  if(cl_match(xcsf, iter->cl, x)) {
212  clset_add(&xcsf->mset, iter->cl);
213  int action = cl_action(xcsf, iter->cl, x);
214  act_covered[action] = true;
215  }
216  }
217 #endif
218  // perform covering if all actions are not represented
219  clset_cover(xcsf, x, act_covered);
220  free(act_covered);
221  // update statistics
222  xcsf->msetsize += (xcsf->mset.size - xcsf->msetsize) * (10 / (double) xcsf->PERF_TRIALS);
223  xcsf->mfrac += (clset_mfrac(xcsf) - xcsf->mfrac) * (10 / (double) xcsf->PERF_TRIALS);
224 }

232 static void clset_cover(XCSF *xcsf, const double *x, _Bool *act_covered)
233 {
234  int attempts = 0;
235  _Bool again;
236  do {
237  again = false;
238  for(int i = 0; i < xcsf->num_actions; i++) {
239  if(!act_covered[i]) {
240  // new classifier with matching condition and action
241  CL *new = malloc(sizeof(CL));
242  cl_init(xcsf, new, (xcsf->mset.num)+1, xcsf->time);
243  cl_cover(xcsf, new, x, i);
244  clset_add(&xcsf->pset, new);
245  clset_add(&xcsf->mset, new);
246  act_covered[i] = true;
247  }
248  }
249  // enforce population size
250  int prev_psize = xcsf->pset.size;
251  clset_pop_enforce_limit(xcsf);
252  // if a macro classifier was deleted, remove any deleted rules from the match set
253  if(prev_psize > xcsf->pset.size) {
254  int prev_msize = xcsf->mset.size;
255  clset_validate(&xcsf->mset);
256  // if the deleted classifier was in the match set,
257  // check if an action is now not covered
258  if(prev_msize > xcsf->mset.size) {
259  for(int i = 0; i < xcsf->num_actions; i++) {
260  if(!clset_action_covered(xcsf, i)) {
261  act_covered[i] = false;
262  again = true;
263  }
264  }
265  }
266  }
267  attempts++;
268  if(attempts > MAX_COVER) {
269  printf("Error: maximum covering attempts (%d) exceeded\n", MAX_COVER);
270  exit(EXIT_FAILURE);
271  }
272 
273  } while(again);
274 }

283 void clset_pred(const XCSF *xcsf, const SET *set, const double *x, double *p)
284 {
285  double *presum = calloc(xcsf->num_y_vars, sizeof(double));
286  double fitsum = 0;
287 #ifdef PARALLEL_PRED
288  CLIST *blist[set->size];
289  int j = 0;
290  for(CLIST *iter = set->list; iter != NULL; iter = iter->next) {
291  blist[j] = iter;
292  j++;
293  }
294 #pragma omp parallel for reduction(+:presum[:xcsf->num_y_vars],fitsum)
295  for(int i = 0; i < set->size; i++) {
296  const double *predictions = cl_predict(xcsf, blist[i]->cl, x);
297  for(int var = 0; var < xcsf->num_y_vars; var++) {
298  presum[var] += predictions[var] * blist[i]->cl->fit;
299  }
300  fitsum += blist[i]->cl->fit;
301  }
302 #pragma omp parallel for
303  for(int var = 0; var < xcsf->num_y_vars; var++) {
304  p[var] = presum[var]/fitsum;
305  }
306 #else
307  for(CLIST *iter = set->list; iter != NULL; iter = iter->next) {
308  const double *predictions = cl_predict(xcsf, iter->cl, x);
309  for(int var = 0; var < xcsf->num_y_vars; var++) {
310  presum[var] += predictions[var] * iter->cl->fit;
311  }
312  fitsum += iter->cl->fit;
313  }
314  for(int var = 0; var < xcsf->num_y_vars; var++) {
315  p[var] = presum[var]/fitsum;
316  }
317 #endif
318  free(presum);
319 }

434 static void clset_subsumption(XCSF *xcsf, SET *set)
435 {
436  // find the most general subsumer in the set
437  CL *s = NULL;
438  for(const CLIST *iter = set->list; iter != NULL; iter = iter->next) {
439  CL *c = iter->cl;
440  if(cl_subsumer(xcsf, c) && (s == NULL || cl_general(xcsf, c, s))) {
441  s = c;
442  }
443  }
444  // subsume the more specific classifiers in the set
445  if(s != NULL) {
446  const CLIST *iter = set->list;
447  while(iter != NULL) {
448  CL *c = iter->cl;
449  iter = iter->next;
450  if(s != c && cl_general(xcsf, s, c)) {
451  s->num += c->num;
452  c->num = 0;
453  clset_add(&xcsf->kset, c);
454  clset_validate(set);
455  clset_validate(&xcsf->pset);
456  }
457  }
458  }
459 }

465 void clset_validate(SET *set)
466 {
467  set->size = 0;
468  set->num = 0;
469  CLIST *prev = NULL;
470  CLIST *iter = set->list;
471  while(iter != NULL) {
472  if(iter->cl == NULL || iter->cl->num == 0) {
473  if(prev == NULL) {
474  set->list = iter->next;
475  free(iter);
476  iter = set->list;
477  }
478  else {
479  prev->next = iter->next;
480  free(iter);
481  iter = prev->next;
482  }
483  }
484  else {
485  set->size++;
486  set->num += iter->cl->num;
487  prev = iter;
488  iter = iter->next;
489  }
490  }
491 }

700 double clset_mfrac(const XCSF *xcsf)
701 {
702  double mfrac = 0;
703  // most general rule below EPS_0
704  for(const CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
705  double e = iter->cl->err;
706  if(e < xcsf->EPS_0 && iter->cl->exp > 1 / xcsf->BETA) {
707  double m = cl_mfrac(xcsf, iter->cl);
708  if(m > mfrac) {
709  mfrac = m;
710  }
711  }
712  }
713  // lowest error rule
714  if(mfrac == 0) {
715  double error = DBL_MAX;
716  for(const CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
717  double e = iter->cl->err;
718  if(e < error && iter->cl->exp > 1 / xcsf->BETA) {
719  mfrac = cl_mfrac(xcsf, iter->cl);
720  error = e;
721  }
722  }
723  }
724  return mfrac;
725 }
