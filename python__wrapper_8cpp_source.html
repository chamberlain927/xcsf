125  void single_reset() {
126  // initialise population if first execution
127  if(xcs.pset.list == NULL) {
128  pop_init(&xcs);
129  }
130  // clear any previous sets
131  if(mset.list != NULL) {
132  set_free(&xcs, &mset);
133  }
134  if(aset.list != NULL) {
135  set_free(&xcs, &aset);
136  }
137  if(kset.list != NULL) {
138  set_kill(&xcs, &kset);
139  }
140  // initialise current sets
141  set_init(&xcs, &mset);
142  set_init(&xcs, &aset);
143  set_init(&xcs, &kset);
144  }

163  double fit(np::ndarray &train_X, np::ndarray &train_Y, _Bool shuffle) {
164  // check inputs are correctly sized
165  if(train_X.shape(0) != train_Y.shape(0)) {
166  printf("error: training X and Y rows are not equal\n");
167  return 0;
168  }
169  // load training data
170  train_data->rows = train_X.shape(0);
171  train_data->x_cols = train_X.shape(1);
172  train_data->y_cols = train_Y.shape(1);
173  train_data->x = reinterpret_cast<double*>(train_X.get_data());
174  train_data->y = reinterpret_cast<double*>(train_Y.get_data());
175  // first execution
176  if(xcs.time == 0) {
177  pop_init(&xcs);
178  }
179  // execute
180  return xcsf_fit1(&xcs, train_data, shuffle);
181  }

183  double fit(np::ndarray &train_X, np::ndarray &train_Y,
184  np::ndarray &test_X, np::ndarray &test_Y, _Bool shuffle) {
185  // check inputs are correctly sized
186  if(train_X.shape(0) != train_Y.shape(0)) {
187  printf("error: training X and Y rows are not equal\n");
188  return 0;
189  }
190  if(test_X.shape(0) != test_Y.shape(0)) {
191  printf("error: testing X and Y rows are not equal\n");
192  return 0;
193  }
194  if(train_X.shape(1) != test_X.shape(1)) {
195  printf("error: number of training and testing X cols are not equal\n");
196  return 0;
197  }
198  if(train_Y.shape(1) != test_Y.shape(1)) {
199  printf("error: number of training and testing Y cols are not equal\n");
200  return 0;
201  }
202  // load training data
203  train_data->rows = train_X.shape(0);
204  train_data->x_cols = train_X.shape(1);
205  train_data->y_cols = train_Y.shape(1);
206  train_data->x = reinterpret_cast<double*>(train_X.get_data());
207  train_data->y = reinterpret_cast<double*>(train_Y.get_data());
208  // load testing data
209  test_data->rows = test_X.shape(0);
210  test_data->x_cols = test_X.shape(1);
211  test_data->y_cols = test_Y.shape(1);
212  test_data->x = reinterpret_cast<double*>(test_X.get_data());
213  test_data->y = reinterpret_cast<double*>(test_Y.get_data());
214  // first execution
215  if(xcs.time == 0) {
216  pop_init(&xcs);
217  }
218  // execute
219  return xcsf_fit2(&xcs, train_data, test_data, shuffle);
220  }

222  np::ndarray predict(np::ndarray &T) {
223  // inputs to predict
224  double *input = reinterpret_cast<double*>(T.get_data());
225  int rows = T.shape(0);
226  // predicted outputs
227  double *output = (double *) malloc(sizeof(double) * rows * xcs.num_y_vars);
228  xcsf_predict(&xcs, input, output, rows);
229  // return numpy array
230  np::ndarray result = np::from_data(output, np::dtype::get_builtin<double>(),
231  p::make_tuple(rows, xcs.num_y_vars),
232  p::make_tuple(sizeof(double)*xcs.num_y_vars, sizeof(double)), p::object());
233  return result;
234  }
