84  double fit(np::ndarray &train_X, np::ndarray &train_Y, _Bool shuffle) {
85  // check inputs are correctly sized
86  if(train_X.shape(0) != train_Y.shape(0)) {
87  printf("error: training X and Y rows are not equal\n");
88  return 0;
89  }
90  // load training data
91  train_data.rows = train_X.shape(0);
92  train_data.x_cols = train_X.shape(1);
93  train_data.y_cols = train_Y.shape(1);
94  train_data.x = reinterpret_cast<double*>(train_X.get_data());
95  train_data.y = reinterpret_cast<double*>(train_Y.get_data());
96  // first execution
97  if(xcs.time == 0) {
98  pop_init(&xcs);
99  }
100  // execute
101  return xcsf_fit1(&xcs, &train_data, shuffle);
102  }

104  double fit(np::ndarray &train_X, np::ndarray &train_Y,
105  np::ndarray &test_X, np::ndarray &test_Y, _Bool shuffle) {
106  // check inputs are correctly sized
107  if(train_X.shape(0) != train_Y.shape(0)) {
108  printf("error: training X and Y rows are not equal\n");
109  return 0;
110  }
111  if(test_X.shape(0) != test_Y.shape(0)) {
112  printf("error: testing X and Y rows are not equal\n");
113  return 0;
114  }
115  if(train_X.shape(1) != test_X.shape(1)) {
116  printf("error: number of training and testing X cols are not equal\n");
117  return 0;
118  }
119  if(train_Y.shape(1) != test_Y.shape(1)) {
120  printf("error: number of training and testing Y cols are not equal\n");
121  return 0;
122  }
123  // load training data
124  train_data.rows = train_X.shape(0);
125  train_data.x_cols = train_X.shape(1);
126  train_data.y_cols = train_Y.shape(1);
127  train_data.x = reinterpret_cast<double*>(train_X.get_data());
128  train_data.y = reinterpret_cast<double*>(train_Y.get_data());
129  // load testing data
130  test_data.rows = test_X.shape(0);
131  test_data.x_cols = test_X.shape(1);
132  test_data.y_cols = test_Y.shape(1);
133  test_data.x = reinterpret_cast<double*>(test_X.get_data());
134  test_data.y = reinterpret_cast<double*>(test_Y.get_data());
135  // first execution
136  if(xcs.time == 0) {
137  pop_init(&xcs);
138  }
139  // execute
140  return xcsf_fit2(&xcs, &train_data, &test_data, shuffle);
141  }

143  np::ndarray predict(np::ndarray &T) {
144  // inputs to predict
145  double *input = reinterpret_cast<double*>(T.get_data());
146  int rows = T.shape(0);
147  // predicted outputs
148  double *output = (double *) malloc(sizeof(double) * rows * xcs.num_y_vars);
149  xcsf_predict(&xcs, input, output, rows);
150  // return numpy array
151  np::ndarray result = np::from_data(output, np::dtype::get_builtin<double>(),
152  p::make_tuple(rows, xcs.num_y_vars),
153  p::make_tuple(sizeof(double)*xcs.num_y_vars, sizeof(double)), p::object());
154  return result;
155  }
