142  double fit(np::ndarray &train_X, np::ndarray &train_Y, _Bool shuffle) {
143  // check inputs are correctly sized
144  if(train_X.shape(0) != train_Y.shape(0)) {
145  printf("error: training X and Y n_samples are not equal\n");
146  exit(EXIT_FAILURE);
147  }
148  // load training data
149  train_data->n_samples = train_X.shape(0);
150  train_data->x_dim = train_X.shape(1);
151  train_data->y_dim = train_Y.shape(1);
152  train_data->x = reinterpret_cast<double*>(train_X.get_data());
153  train_data->y = reinterpret_cast<double*>(train_Y.get_data());
154  // first execution
155  if(xcs.time == 0) {
156  clset_pop_init(&xcs);
157  }
158  // execute
159  return xcsf_fit(&xcs, train_data, NULL, shuffle);
160  }

162  double fit(np::ndarray &train_X, np::ndarray &train_Y,
163  np::ndarray &test_X, np::ndarray &test_Y, _Bool shuffle) {
164  // check inputs are correctly sized
165  if(train_X.shape(0) != train_Y.shape(0)) {
166  printf("error: training X and Y n_samples are not equal\n");
167  exit(EXIT_FAILURE);
168  }
169  if(test_X.shape(0) != test_Y.shape(0)) {
170  printf("error: testing X and Y n_samples are not equal\n");
171  exit(EXIT_FAILURE);
172  }
173  if(train_X.shape(1) != test_X.shape(1)) {
174  printf("error: number of training and testing X cols are not equal\n");
175  exit(EXIT_FAILURE);
176  }
177  if(train_Y.shape(1) != test_Y.shape(1)) {
178  printf("error: number of training and testing Y cols are not equal\n");
179  exit(EXIT_FAILURE);
180  }
181  // load training data
182  train_data->n_samples = train_X.shape(0);
183  train_data->x_dim = train_X.shape(1);
184  train_data->y_dim = train_Y.shape(1);
185  train_data->x = reinterpret_cast<double*>(train_X.get_data());
186  train_data->y = reinterpret_cast<double*>(train_Y.get_data());
187  // load testing data
188  test_data->n_samples = test_X.shape(0);
189  test_data->x_dim = test_X.shape(1);
190  test_data->y_dim = test_Y.shape(1);
191  test_data->x = reinterpret_cast<double*>(test_X.get_data());
192  test_data->y = reinterpret_cast<double*>(test_Y.get_data());
193  // first execution
194  if(xcs.time == 0) {
195  clset_pop_init(&xcs);
196  }
197  // execute
198  return xcsf_fit(&xcs, train_data, test_data, shuffle);
199  }

201  np::ndarray predict(np::ndarray &T) {
202  // inputs to predict
203  double *input = reinterpret_cast<double*>(T.get_data());
204  int n_samples = T.shape(0);
205  // predicted outputs
206  double *output = (double *) malloc(sizeof(double) * n_samples * xcs.y_dim);
207  xcsf_predict(&xcs, input, output, n_samples);
208  // return numpy array
209  np::ndarray result = np::from_data(output, np::dtype::get_builtin<double>(),
210  p::make_tuple(n_samples, xcs.y_dim),
211  p::make_tuple(sizeof(double)*xcs.y_dim, sizeof(double)), p::object());
212  return result;
213  }
