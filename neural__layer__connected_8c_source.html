41 LAYER *neural_layer_connected_init(XCSF *xcsf, int in, int n_init, int n_max, int f, uint32_t o)
42 {
43  LAYER *l = malloc(sizeof(LAYER));
44  l->layer_type = CONNECTED;
45  l->layer_vptr = &layer_connected_vtbl;
46  l->function = f;
47  l->num_inputs = in;
48  l->num_outputs = n_max;
49  l->num_weights = in * n_max;
50  l->num_active = n_init;
51  l->state = calloc(l->num_outputs, sizeof(double));
52  l->output = calloc(l->num_outputs, sizeof(double));
53  l->biases = calloc(l->num_outputs, sizeof(double));
54  l->bias_updates = calloc(l->num_outputs, sizeof(double));
55  l->weight_updates = calloc(l->num_weights, sizeof(double));
56  l->delta = calloc(l->num_outputs, sizeof(double));
57  l->weights = malloc(l->num_weights * sizeof(double));
58  for(int i = 0; i < l->num_weights; i++) {
59  l->weights[i] = rand_normal(0,0.1);
60  }
61  l->active = calloc(l->num_outputs, sizeof(_Bool));
62  for(int i = 0; i < l->num_active; i++) {
63  l->active[i] = true;
64  }
65  l->options = o;
66  if(l->options & LAYER_EVOLVE_ETA) {
67  l->eta = rand_uniform(ETA_MIN,ETA_MAX);
68  }
69  else {
70  l->eta = xcsf->PRED_ETA;
71  }
72  return l;
73 }

126 void neural_layer_connected_forward(XCSF *xcsf, LAYER *l, double *input)
127 {
128  (void)xcsf;
129  for(int i = 0; i < l->num_outputs; i++) {
130  if(l->active[i]) {
131  l->state[i] = 0;
132  for(int j = 0; j < l->num_inputs; j++) {
133  l->state[i] += input[j] * l->weights[i * l->num_inputs + j];
134  }
135  l->state[i] += l->biases[i];
136  l->state[i] = constrain(-100, 100, l->state[i]);
137  l->output[i] = neural_activate(l->function, l->state[i]);
138  }
139  else {
140  l->state[i] = 0;
141  l->output[i] = 0;
142  }
143  }
144 }

146 void neural_layer_connected_backward(XCSF *xcsf, LAYER *l, NET *net)
147 {
148  // net->input[] = this layer's input
149  // net->delta[] = previous layer's delta
150  (void)xcsf;
151  for(int i = 0; i < l->num_active; i++) {
152  l->delta[i] *= neural_gradient(l->function, l->state[i]);
153  if(l->options & LAYER_SGD_WEIGHTS) {
154  l->bias_updates[i] += l->delta[i];
155  for(int j = 0; j < l->num_inputs; j++) {
156  l->weight_updates[i * l->num_inputs + j] += l->delta[i] * net->input[j];
157  }
158  }
159  }
160  if(net->delta) { // input layer has no delta or weights
161  for(int i = 0; i < l->num_active; i++) {
162  for(int j = 0; j < l->num_inputs; j++) {
163  net->delta[j] += l->delta[i] * l->weights[i * l->num_inputs + j];
164  }
165  }
166  }
167 }

169 void neural_layer_connected_update(XCSF *xcsf, LAYER *l)
170 {
171  if(l->options & LAYER_SGD_WEIGHTS) {
172  for(int i = 0; i < l->num_active; i++) {
173  l->biases[i] += l->eta * l->bias_updates[i];
174  l->bias_updates[i] *= xcsf->PRED_MOMENTUM;
175  }
176  int w = l->num_inputs * l->num_active;
177  for(int i = 0; i < w; i++) {
178  l->weights[i] += l->eta * l->weight_updates[i];
179  l->weight_updates[i] *= xcsf->PRED_MOMENTUM;
180  }
181  }

184 _Bool neural_layer_connected_mutate(XCSF *xcsf, LAYER *l)
185 {
186  _Bool mod = false;
187 
188  if(l->options & LAYER_EVOLVE_NEURONS) {
189  if(rand_uniform(0,1) < xcsf->P_MUTATION) {
190  int idx = l->num_active - 1;
191  // remove
192  if(l->num_active > 1 && rand_uniform(0,1) < 0.5) {
193  l->active[idx] = false;
194  l->num_active--;
195  mod = true;
196  }
197  // add
198  else if(l->num_active < l->num_outputs) {
199  l->active[idx] = true;
200  l->num_active++;
201  // randomise weights
202  l->biases[idx] = 0;
203  for(int i = 0; i < l->num_inputs; i++) {
204  l->weights[idx * l->num_inputs + i] = rand_normal(0,0.1);
205  }
206  mod = true;
207  }
208  }
209  }
210 
211  if(l->options & LAYER_EVOLVE_WEIGHTS) {
212  int w = l->num_inputs * l->num_active;
213  for(int i = 0; i < w; i++) {
214  double orig = l->weights[i];
215  l->weights[i] += rand_normal(0, xcsf->S_MUTATION);
216  if(l->weights[i] != orig) {
217  mod = true;
218  }
219  }
220  for(int i = 0; i < l->num_active; i++) {
221  double orig = l->biases[i];
222  l->biases[i] += rand_normal(0, xcsf->S_MUTATION);
223  if(l->biases[i] != orig) {
224  mod = true;
225  }
226  }
227  }
228 
229  if(l->options & LAYER_EVOLVE_ETA) {
230  double orig = l->eta;
231  l->eta += rand_normal(0, xcsf->E_MUTATION);
232  if(l->eta > ETA_MAX) {
233  l->eta = ETA_MAX;
234  }
235  else if(l->eta < ETA_MIN) {
236  l->eta = ETA_MIN;
237  }
238  if(l->eta != orig) {
239  mod = true;
240  }
241  }
242 
243  if(l->options & LAYER_EVOLVE_FUNCTIONS) {
244  if(rand_uniform(0,1) < xcsf->F_MUTATION) {
245  int orig = l->function;
246  l->function = irand_uniform(0, NUM_ACTIVATIONS);
247  if(l->function != orig) {
248  mod = true;
249  }
250  }
251  }
252 
253  return mod;
254 }
