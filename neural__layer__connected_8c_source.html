48 LAYER *neural_layer_connected_init(const XCSF *xcsf, int in, int n_init, int n_max, int f, uint32_t o)
49 {
50  LAYER *l = malloc(sizeof(LAYER));
51  l->layer_type = CONNECTED;
52  l->layer_vptr = &layer_connected_vtbl;
53  l->function = f;
54  l->num_inputs = in;
55  l->num_outputs = n_init;
56  l->max_outputs = n_max;
57  l->num_weights = in * n_init;
58  l->state = calloc(l->num_outputs, sizeof(double));
59  l->output = calloc(l->num_outputs, sizeof(double));
60  l->biases = calloc(l->num_outputs, sizeof(double));
61  l->bias_updates = calloc(l->num_outputs, sizeof(double));
62  l->weight_updates = calloc(l->num_weights, sizeof(double));
63  l->delta = calloc(l->num_outputs, sizeof(double));
64  l->weights = malloc(l->num_weights * sizeof(double));
65  for(int i = 0; i < l->num_weights; i++) {
66  l->weights[i] = rand_normal(0,0.1);
67  }
68  l->options = o;
69  if(l->options & LAYER_EVOLVE_ETA) {
70  l->eta = rand_uniform(ETA_MIN,ETA_MAX);
71  }
72  else {
73  l->eta = xcsf->PRED_ETA;
74  }
75  return l;
76 }

126 void neural_layer_connected_forward(const XCSF *xcsf, const LAYER *l, const double *input)
127 {
128  (void)xcsf;
129  for(int i = 0; i < l->num_outputs; i++) {
130  l->state[i] = l->biases[i];
131  int offset = i * l->num_inputs;
132  for(int j = 0; j < l->num_inputs; j++) {
133  l->state[i] += input[j] * l->weights[offset + j];
134  }
135  l->state[i] = constrain(-100, 100, l->state[i]);
136  l->output[i] = neural_activate(l->function, l->state[i]);
137  }
138 }

140 void neural_layer_connected_backward(const XCSF *xcsf, const LAYER *l, const NET *net)
141 {
142  // net->input[] = this layer's input
143  // net->delta[] = previous layer's delta
144  (void)xcsf;
145  for(int i = 0; i < l->num_outputs; i++) {
146  l->delta[i] *= neural_gradient(l->function, l->state[i]);
147  if(l->options & LAYER_SGD_WEIGHTS) {
148  l->bias_updates[i] += l->delta[i];
149  int offset = i * l->num_inputs;
150  for(int j = 0; j < l->num_inputs; j++) {
151  l->weight_updates[offset + j] += l->delta[i] * net->input[j];
152  }
153  }
154  }
155  if(net->delta) { // input layer has no delta or weights
156  for(int i = 0; i < l->num_outputs; i++) {
157  int offset = i * l->num_inputs;
158  for(int j = 0; j < l->num_inputs; j++) {
159  net->delta[j] += l->delta[i] * l->weights[offset + j];
160  }
161  }
162  }
163 }

165 void neural_layer_connected_update(const XCSF *xcsf, const LAYER *l)
166 {
167  if(l->options & LAYER_SGD_WEIGHTS) {
168  for(int i = 0; i < l->num_outputs; i++) {
169  l->biases[i] += l->eta * l->bias_updates[i];
170  l->bias_updates[i] *= xcsf->PRED_MOMENTUM;
171  }
172  for(int i = 0; i < l->num_weights; i++) {
173  l->weights[i] += l->eta * l->weight_updates[i];
174  l->weight_updates[i] *= xcsf->PRED_MOMENTUM;
175  }
176  }

179 void neural_layer_connected_resize(const XCSF *xcsf, LAYER *l, const LAYER *prev)
180 {
181  (void)xcsf;
182  int num_weights = prev->num_outputs * l->num_outputs;
183  double *weights = malloc(num_weights * sizeof(double));
184  double *weight_updates = malloc(num_weights * sizeof(double));
185  for(int i = 0; i < l->num_outputs; i++) {
186  int orig_offset = i * l->num_inputs;
187  int offset = i * prev->num_outputs;
188  for(int j = 0; j < prev->num_outputs; j++) {
189  if(j < l->num_inputs) {
190  weights[offset + j] = l->weights[orig_offset + j];
191  weight_updates[offset + j] = l->weight_updates[orig_offset + j];
192  }
193  else {
194  weights[offset + j] = rand_normal(0,0.1);
195  weight_updates[offset + j] = 0;
196  }
197  }
198  }
199  free(l->weights);
200  free(l->weight_updates);
201  l->weights = weights;
202  l->weight_updates = weight_updates;
203  l->num_weights = num_weights;
204  l->num_inputs = prev->num_outputs;
205 }

225 static _Bool mutate_neurons(const XCSF *xcsf, LAYER *l)
226 {
227  _Bool mod = false;
228  if(rand_uniform(0,1) < xcsf->P_MUTATION) {
229  // number of neurons to add or remove
230  int n = irand_uniform(1, xcsf->MAX_NEURON_MOD);
231  // remove neurons
232  if(rand_uniform(0,1) < 0.5) {
233  if(l->num_outputs - n < 1) {
234  n = l->num_outputs - 1;
235  }
236  if(n > 0) {
237  neuron_remove(l, n);
238  mod = true;
239  }
240  }
241  // add neurons
242  else {
243  if(l->num_outputs + n > l->max_outputs) {
244  n = l->max_outputs - l->num_outputs;
245  }
246  if(n > 0) {
247  neuron_add(l, n);
248  mod = true;
249  }
250  }
251  }
252  return mod;
253 }

255 static void neuron_add(LAYER *l, int n)
256 {
257  l->num_outputs += n;
258  int num_weights = l->num_outputs * l->num_inputs;
259  double *weights = malloc(num_weights * sizeof(double));
260  double *weight_updates = malloc(num_weights * sizeof(double));
261  double *state = calloc(l->num_outputs, sizeof(double));
262  double *output = calloc(l->num_outputs, sizeof(double));
263  double *biases = malloc(l->num_outputs * sizeof(double));
264  double *bias_updates = malloc(l->num_outputs * sizeof(double));
265  double *delta = calloc(l->num_outputs, sizeof(double));
266  memcpy(weights, l->weights, l->num_weights * sizeof(double));
267  memcpy(weight_updates, l->weight_updates, l->num_weights * sizeof(double));
268  for(int i = l->num_weights; i < num_weights; i++) {
269  weights[i] = rand_normal(0,0.1);
270  weight_updates[i] = 0;
271  }
272  memcpy(biases, l->biases, (l->num_outputs - n) * sizeof(double));
273  memcpy(bias_updates, l->bias_updates, (l->num_outputs - n) * sizeof(double));
274  for(int i = l->num_outputs - n; i < l->num_outputs; i++) {
275  biases[i] = 0;
276  bias_updates[i] = 0;
277  }
278  free(l->weights);
279  free(l->weight_updates);
280  free(l->state);
281  free(l->output);
282  free(l->biases);
283  free(l->bias_updates);
284  free(l->delta);
285  l->weights = weights;
286  l->weight_updates = weight_updates;
287  l->state = state;
288  l->output = output;
289  l->biases = biases;
290  l->bias_updates = bias_updates;
291  l->delta = delta;
292  l->num_weights = num_weights;
293 }

327 static _Bool mutate_weights(const XCSF *xcsf, const LAYER *l)
328 {
329  _Bool mod = false;
330  for(int i = 0; i < l->num_weights; i++) {
331  double orig = l->weights[i];
332  l->weights[i] += rand_normal(0, xcsf->S_MUTATION);

338  double orig = l->biases[i];
339  l->biases[i] += rand_normal(0, xcsf->S_MUTATION);

341  mod = true;
342  }
343  }
