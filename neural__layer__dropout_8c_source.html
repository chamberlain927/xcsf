86 void neural_layer_dropout_forward(XCSF *xcsf, LAYER *l, double *input)
87 {
88  if(!xcsf->train) {
89  for(int i = 0; i < l->num_inputs; i++) {
90  l->output[i] = input[i];
91  }
92  }
93  else {
94  for(int i = 0; i < l->num_inputs; i++) {
95  l->rand[i] = rand_uniform(0,1);
96  if(l->rand[i] < l->probability) {
97  l->output[i] = 0;
98  }
99  else {
100  l->output[i] = input[i] * l->scale;
101  }
102  }
103  }
104 }

106 void neural_layer_dropout_backward(XCSF *xcsf, LAYER *l, NET *net)
107 {
108  (void)xcsf;
109  if(!net->delta) {
110  return;
111  }
112  for(int i = 0; i < l->num_inputs; i++) {
113  if(l->rand[i] < l->probability) {
114  net->delta[i] = 0;
115  }
116  else {
117  net->delta[i] += l->delta[i] * l->scale;
118  }
119  }
120 }
