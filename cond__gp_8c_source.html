125 _Bool cond_gp_general(XCSF *xcsf, CL *c1, CL *c2)
126 {
127  if(c1->exp < xcsf->THETA_SUB || c2->exp < xcsf->THETA_SUB) {
128  return false;
129  }
130  for(int i = 0; i < xcsf->THETA_SUB; i++) {
131  int i1 = (c1->exp + i) % xcsf->THETA_SUB;
132  int i2 = (c2->exp + i) % xcsf->THETA_SUB;
133  if(c1->mhist[i1] == false && c2->mhist[i2] == true) {
134  return false;
135  }
136  }
137  return true;
138 }
