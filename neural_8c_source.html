77 void neural_layer_insert(const XCSF *xcsf, NET *net, LAYER *l, int p)
78 {
79  (void)xcsf;
80  // empty list
81  if(net->head == NULL) {
82  net->head = malloc(sizeof(LLIST));
83  net->head->layer = l;
84  net->head->prev = NULL;
85  net->head->next = NULL;
86  net->tail = net->head;
87  net->n_inputs = l->n_inputs;
88  net->n_outputs = l->n_outputs;
89  }
90  // insert at head
91  else if(p >= net->n_layers) {
92  LLIST *new = malloc(sizeof(LLIST));
93  new->layer = l;
94  new->next = net->head;
95  new->prev = NULL;
96  net->head->prev = new;
97  net->head = new;
98  net->n_outputs = l->n_outputs;
99  }
100  // insert before head
101  else {
102  LLIST *iter = net->tail;
103  for(int i = 0; i < p && iter != NULL; i++) {
104  iter = iter->prev;
105  }
106  LLIST *new = malloc(sizeof(LLIST));
107  new->layer = l;
108  new->prev = iter;
109  new->next = iter->next;
110  iter->next = new;
111  // new tail
112  if(new->next == NULL) {
113  net->tail = new;
114  net->n_inputs = l->n_inputs;
115  }
116  else {
117  new->next->prev = new;
118  }
119  }
120  net->n_layers++;
121 }

129 void neural_layer_remove(const XCSF *xcsf, NET *net, int p)
130 {
131  LLIST *iter = net->tail;
132  for(int i = 0; i < p && iter != NULL; i++) {
133  iter = iter->prev;
134  }
135  // head
136  if(iter->prev == NULL) {
137  net->head = iter->next;
138  if(iter->next != NULL) {
139  iter->next->prev = NULL;
140  }
141  }
142  // tail
143  if(iter->next == NULL) {
144  net->tail = iter->prev;
145  if(iter->prev != NULL) {
146  iter->prev->next = NULL;
147  }
148  }
149  // middle
150  if(iter->prev != NULL && iter->next != NULL) {
151  iter->next->prev = iter->prev;
152  iter->prev->next = iter->next;
153  }
154  net->n_layers--;
155  layer_free(xcsf, iter->layer);
156  free(iter->layer);
157  free(iter);
158 }

216 _Bool neural_mutate(const XCSF *xcsf, NET *net)
217 {
218  sam_init(xcsf, net->mu, N_MU);
219  _Bool mod = false;
220  const LAYER *prev = NULL;
221  for(const LLIST *iter = net->tail; iter != NULL; iter = iter->prev) {
222  // previous layer has grown or shrunk: weight vector must be resized
223  if(prev != NULL && iter->layer->n_inputs != prev->n_outputs) {
224  layer_resize(xcsf, iter->layer, prev);
225  }
226  // mutate this layer
227  if(layer_mutate(xcsf, iter->layer, net->mu)) {
228  mod = true;
229  }
230  prev = iter->layer;
231  }
232  if((net->nopt & NETWORK_EVOLVE_ETA) && mutate_eta(net)) {
233  mod = true;
234  }
235  return mod;
236 }

238 static _Bool mutate_eta(NET *net)
239 {
240  double orig = net->eta;
241  net->eta += rand_normal(0, net->mu[0]);
242  net->eta = constrain(ETA_MIN, ETA_MAX, net->eta);
243  if(net->eta != orig) {
244  return true;
245  }
246  return false;
247 }

270 void neural_learn(const XCSF *xcsf, NET *net, const double *truth, const double *input)
271 {
272  /* reset deltas */
273  for(const LLIST *iter = net->tail; iter != NULL; iter = iter->prev) {
274  memset(iter->layer->delta, 0, iter->layer->n_outputs * sizeof(double));
275  }
276 
277  // calculate output layer error
278  const LAYER *p = net->head->layer;
279  for(int i = 0; i < p->n_outputs; i++) {
280  p->delta[i] = (truth[i] - p->output[i]);
281  }
282 
283  /* backward phase */
284  for(const LLIST *iter = net->head; iter != NULL; iter = iter->next) {
285  const LAYER *l = iter->layer;
286  if(iter->next == NULL) {
287  net->input = input;
288  net->delta = 0;
289  }
290  else {
291  const LAYER *prev = iter->next->layer;
292  net->input = prev->output;
293  net->delta = prev->delta;
294  }
295  layer_backward(xcsf, l, net);
296  }
297 
298  /* update phase */
299  for(const LLIST *iter = net->tail; iter != NULL; iter = iter->prev) {
300  layer_update(xcsf, iter->layer, net->eta);
301  }
302 }

399  neural_layer_set_vptr(l);

401  neural_layer_insert(xcsf, net, l, i);
402  }
403  s += fread(net->mu, sizeof(double), N_MU, fp);
