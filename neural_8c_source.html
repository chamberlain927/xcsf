62 void neural_layer_insert(XCSF *xcsf, NET *net, LAYER *l, int p)
63 {
64  (void)xcsf;
65  // empty list
66  if(net->head == NULL) {
67  net->head = malloc(sizeof(LLIST));
68  net->head->layer = l;
69  net->head->prev = NULL;
70  net->head->next = NULL;
71  net->tail = net->head;
72  net->num_inputs = l->num_inputs;
73  net->num_outputs = l->num_outputs;
74  }
75  // insert at head
76  else if(p >= net->num_layers) {
77  LLIST *new = malloc(sizeof(LLIST));
78  new->layer = l;
79  new->next = net->head;
80  new->prev = NULL;
81  net->head->prev = new;
82  net->head = new;
83  net->num_outputs = l->num_outputs;
84  }
85  // insert before head
86  else {
87  LLIST *iter = net->tail;
88  for(int i = 0; i < p && iter != NULL; i++) {
89  iter = iter->prev;
90  }
91  LLIST *new = malloc(sizeof(LLIST));
92  new->layer = l;
93  new->prev = iter;
94  new->next = iter->next;
95  iter->next = new;
96  // new tail
97  if(new->next == NULL) {
98  net->tail = new;
99  net->num_inputs = l->num_inputs;
100  }
101  else {
102  new->next->prev = new;
103  }
104  }
105  net->num_layers++;
106 }

114 void neural_layer_remove(XCSF *xcsf, NET *net, int p)
115 {
116  LLIST *iter = net->tail;
117  for(int i = 0; i < p && iter != NULL; i++) {
118  iter = iter->prev;
119  }
120  // head
121  if(iter->prev == NULL) {
122  net->head = iter->next;
123  if(iter->next != NULL) {
124  iter->next->prev = NULL;
125  }
126  }
127  // tail
128  if(iter->next == NULL) {
129  net->tail = iter->prev;
130  if(iter->prev != NULL) {
131  iter->prev->next = NULL;
132  }
133  }
134  // middle
135  if(iter->prev != NULL && iter->next != NULL) {
136  iter->next->prev = iter->prev;
137  iter->prev->next = iter->next;
138  }
139  net->num_layers--;
140  layer_free(xcsf, iter->layer);
141  free(iter->layer);
142  free(iter);
143 }

231 void neural_learn(XCSF *xcsf, NET *net, double *truth, double *input)
232 {
233  /* reset deltas */
234  for(LLIST *iter = net->tail; iter != NULL; iter = iter->prev) {
235  LAYER *l = iter->layer;
236  for(int j = 0; j < l->num_outputs; j++) {
237  l->delta[j] = 0;
238  }
239  }
240 
241  // calculate output layer error
242  LAYER *p = net->head->layer;
243  for(int i = 0; i < p->num_outputs; i++) {
244  p->delta[i] = (truth[i] - p->output[i]);
245  }
246 
247  /* backward phase */
248  for(LLIST *iter = net->head; iter != NULL; iter = iter->next) {
249  LAYER *l = iter->layer;
250  if(iter->next == NULL) {
251  net->input = input;
252  net->delta = 0;
253  }
254  else {
255  LAYER *prev = iter->next->layer;
256  net->input = prev->output;
257  net->delta = prev->delta;
258  }
259  layer_backward(xcsf, l, net);
260  }
261 
262  /* update phase */
263  for(LLIST *iter = net->tail; iter != NULL; iter = iter->prev) {
264  layer_update(xcsf, iter->layer);
265  }
266 }
