93 static void gemm_cpu(int TA, int TB, int M, int N, int K, double ALPHA,
94  const double *A, int lda,
95  const double *B, int ldb,
96  double BETA,
97  double *C, int ldc)
98 {
99  for(int i = 0; i < M; i++) {
100  for(int j = 0; j < N; j++) {
101  C[i*ldc+j] *= BETA;
102  }
103  }
104  if(!TA && !TB) {
105  gemm_nn(M, N, K, ALPHA, A, lda, B, ldb, C, ldc);
106  }
107  else if(TA && !TB) {
108  gemm_tn(M, N, K, ALPHA, A, lda, B, ldb, C, ldc);
109  }
110  else if(!TA && TB) {
111  gemm_nt(M, N, K, ALPHA, A, lda, B, ldb, C, ldc);
112  }
113  else {
114  gemm_tt(M, N, K, ALPHA, A, lda, B, ldb, C, ldc);
