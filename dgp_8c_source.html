106 void graph_rand(const XCSF *xcsf, GRAPH *dgp)
107 {
108  dgp->t = irand_uniform(1,xcsf->MAX_T);
109  for(int i = 0; i < dgp->n; i++) {
110  dgp->function[i] = irand_uniform(0, NUM_FUNC);
111  dgp->initial_state[i] = rand_uniform(0,1);
112  dgp->state[i] = rand_uniform(0,1);
113  }
114  for(int i = 0; i < dgp->klen; i++) {
115  // other nodes within the graph
116  if(rand_uniform(0,1) < 0.5) {
117  dgp->connectivity[i] = irand_uniform(0,dgp->n);
118  }
119  // external inputs
120  else {
121  dgp->connectivity[i] = -(irand_uniform(1,xcsf->num_x_vars+1));
122  }
123  }

132 void graph_update(const XCSF *xcsf, const GRAPH *dgp, const double *inputs)
133 {
134  if(xcsf->RESET_STATES) {
135  graph_reset(xcsf, dgp);
136  }
137  double in[xcsf->MAX_K];
138  for(int t = 0; t < dgp->t; t++) {
139  // synchronously update each node in parallel
140  for(int i = 0; i < dgp->n; i++) {
141  int shift = i * xcsf->MAX_K;
142  // each connection
143  for(int k = 0; k < xcsf->MAX_K; k++) {
144  int c = dgp->connectivity[shift + k];
145  if(c < 0) { // external input
146  in[k] = inputs[abs(c)-1];
147  }
148  else { // another node within the graph
149  in[k] = dgp->state[c];
150  }
151  }
152  dgp->tmp[i] = node_activate(dgp->function[i], in, xcsf->MAX_K);
153  }
154  memcpy(dgp->state, dgp->tmp, sizeof(double) * dgp->n);
155  }
156 }

199 _Bool graph_mutate(const XCSF *xcsf, GRAPH *dgp)
200 {
201  _Bool mod = false;
202  int orig;
203  for(int i = 0; i < dgp->n; i++) {
204  // mutate function
205  if(rand_uniform(0,1) < xcsf->F_MUTATION) {
206  orig = dgp->function[i];
207  dgp->function[i] = irand_uniform(0, NUM_FUNC);
208  if(orig != dgp->function[i]) {
209  mod = true;
210  }
211  }
212  // mutate connectivity map
213  int shift = i * xcsf->MAX_K;
214  for(int j = 0; j < xcsf->MAX_K; j++) {
215  int idx = shift + j;
216  if(rand_uniform(0,1) < xcsf->P_MUTATION) {
217  orig = dgp->connectivity[idx];
218  // external connection
219  if(rand_uniform(0,1) < 0.5) {
220  dgp->connectivity[idx] = -(irand_uniform(1,xcsf->num_x_vars+1));
221  }
222  // another node
223  else {
224  dgp->connectivity[idx] = irand_uniform(0,dgp->n);
225  }
226  if(orig != dgp->connectivity[idx]) {
227  mod = true;
228  }
229  }
230  }
231  }
232  // mutate number of update cycles
233  if(rand_uniform(0,1) < xcsf->S_MUTATION) {
234  orig = dgp->t;
235  dgp->t = irand_uniform(1,xcsf->MAX_T);
236  if(orig != dgp->t) {
237  mod = true;
238  }
239  }
240  return mod;
241 }

253 _Bool graph_crossover(const XCSF *xcsf, GRAPH *dgp1, GRAPH *dgp2)
254 {
255  if(rand_uniform(0,1) > xcsf->P_CROSSOVER) {
256  return false;
257  }
258  // number of update cycles
259  if(rand_uniform(0,1) < 0.5) {
260  int tmp = dgp1->t;
261  dgp1->t = dgp2->t;
262  dgp2->t = tmp;
263  }
264  for(int i = 0; i < dgp1->n; i++) {
265  // functions
266  if(rand_uniform(0,1) < 0.5) {
267  int tmp = dgp1->function[i];
268  dgp1->function[i] = dgp2->function[i];
269  dgp2->function[i] = tmp;
270  }
271  // initial states
272  if(rand_uniform(0,1) < 0.5) {
273  double tmp = dgp1->initial_state[i];
274  dgp1->initial_state[i] = dgp2->initial_state[i];
275  dgp2->initial_state[i] = tmp;
276  }
277  // states
278  if(rand_uniform(0,1) < 0.5) {
279  double tmp = dgp1->state[i];
280  dgp1->state[i] = dgp2->state[i];
281  dgp2->state[i] = tmp;
282  }
283  }
284  // connectivity map
285  for(int i = 0; i < dgp1->klen; i++) {
286  if(rand_uniform(0,1) < 0.5) {
287  double tmp = dgp1->connectivity[i];
288  dgp1->connectivity[i] = dgp2->connectivity[i];
289  dgp2->connectivity[i] = tmp;
290  }
291  }
292  return true;
293 }

302 static double node_activate(int function, const double *inputs, int k)
303 {
304  double state = 0;
305  switch(function)
306  {
307  case 0: // Fuzzy NOT
308  state = 1 - inputs[0];
309  break;
310  case 1: // Fuzzy AND (CFMQVS)
311  state = inputs[0];
312  for(int i = 1; i < k; i++) {
313  state *= inputs[i];
314  }
315  break;
316  case 2: // Fuzzy OR (CFMQVS)
317  state = inputs[0];
318  for(int i = 1; i < k; i++) {
319  state += inputs[i];
320  }
321  break;
322  default: // Invalid function
323  printf("Error updating node: Invalid function: %d\n", function);
324  exit(EXIT_FAILURE);
325  }
326  state = constrain(0, 1, state);
327  return state;
328 }
