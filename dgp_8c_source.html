128 static int random_connection(int n_nodes, int n_inputs)
129 {
130  // another node within the graph

132  return irand_uniform(0, n_nodes) + n_inputs;

160 static void synchronous_update(const XCSF *xcsf, const GRAPH *dgp, const double *inputs)
161 {
162  for(int i = 0; i < dgp->n; i++) {
163  for(int k = 0; k < xcsf->MAX_K; k++) {
164  int c = dgp->connectivity[i * xcsf->MAX_K + k];
165  if(c < xcsf->x_dim) { // external input
166  dgp->tmp_input[k] = inputs[c];
167  }
168  else { // another node within the graph
169  dgp->tmp_input[k] = dgp->state[c - xcsf->x_dim];
170  }
171  }
172  dgp->tmp_state[i] = node_activate(dgp->function[i], dgp->tmp_input, xcsf->MAX_K);
173  }
174  memcpy(dgp->state, dgp->tmp_state, sizeof(double) * dgp->n);
175 }

219 _Bool graph_mutate(const XCSF *xcsf, GRAPH *dgp)
220 {
221  sam_adapt(xcsf, dgp->mu, DGP_N_MU);
222  _Bool mod = false;
223  // mutate functions
224  for(int i = 0; i < dgp->n; i++) {
225  if(rand_uniform(0,1) < dgp->mu[0]) {
226  int orig = dgp->function[i];
227  dgp->function[i] = irand_uniform(0, NUM_FUNC);
228  if(orig != dgp->function[i]) {
229  mod = true;
230  }
231  }
232  }
233  // mutate connectivity map
234  for(int i = 0; i < dgp->klen; i++) {
235  if(rand_uniform(0,1) < dgp->mu[1]) {
236  int orig = dgp->connectivity[i];
237  dgp->connectivity[i] = random_connection(dgp->n, xcsf->x_dim);
238  if(orig != dgp->connectivity[i]) {
239  mod = true;
240  }
241  }
242  }
243  // mutate number of update cycles
244  if(rand_uniform(0,1) < dgp->mu[2]) {
245  int orig = dgp->t;
246  dgp->t = irand_uniform(1,xcsf->MAX_T);
247  if(orig != dgp->t) {
248  mod = true;
249  }
250  }
251  return mod;
252 }

274 static double node_activate(int function, const double *inputs, int k)
275 {
276  double state = 0;
277  switch(function)
278  {
279  case 0: // Fuzzy NOT
280  state = 1 - inputs[0];
281  break;
282  case 1: // Fuzzy AND (CFMQVS)
283  state = inputs[0];
284  for(int i = 1; i < k; i++) {
285  state *= inputs[i];
286  }
287  break;
288  case 2: // Fuzzy OR (CFMQVS)
289  state = inputs[0];
290  for(int i = 1; i < k; i++) {
291  state += inputs[i];
292  }
293  break;
294  default: // Invalid function
295  printf("Error updating node: Invalid function: %d\n", function);
296  exit(EXIT_FAILURE);
297  }
298  state = clamp(0, 1, state);
299  return state;
300 }
