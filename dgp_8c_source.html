109 void graph_rand(const XCSF *xcsf, GRAPH *dgp)
110 {
111  dgp->t = irand_uniform(1,xcsf->MAX_T);
112  for(int i = 0; i < dgp->n; i++) {
113  dgp->function[i] = irand_uniform(0, NUM_FUNC);
114  dgp->initial_state[i] = rand_uniform(0,1);
115  dgp->state[i] = rand_uniform(0,1);
116  }
117  for(int i = 0; i < dgp->klen; i++) {
118  // other nodes within the graph
119  if(rand_uniform(0,1) < 0.5) {
120  dgp->connectivity[i] = irand_uniform(0,dgp->n);
121  }
122  // external inputs
123  else {
124  dgp->connectivity[i] = -(irand_uniform(1,xcsf->x_dim+1));
125  }
126  }
127 }

135 void graph_update(const XCSF *xcsf, const GRAPH *dgp, const double *inputs)
136 {
137  if(xcsf->RESET_STATES) {
138  graph_reset(xcsf, dgp);
139  }
140  double in[xcsf->MAX_K];
141  for(int t = 0; t < dgp->t; t++) {
142  // synchronously update each node in parallel
143  for(int i = 0; i < dgp->n; i++) {
144  int shift = i * xcsf->MAX_K;
145  // each connection
146  for(int k = 0; k < xcsf->MAX_K; k++) {
147  int c = dgp->connectivity[shift + k];
148  if(c < 0) { // external input
149  in[k] = inputs[abs(c)-1];
150  }
151  else { // another node within the graph
152  in[k] = dgp->state[c];
153  }
154  }
155  dgp->tmp[i] = node_activate(dgp->function[i], in, xcsf->MAX_K);
156  }
157  memcpy(dgp->state, dgp->tmp, sizeof(double) * dgp->n);
158  }
159 }

202 _Bool graph_mutate(const XCSF *xcsf, GRAPH *dgp)
203 {
204  sam_adapt(xcsf, dgp->mu, DGP_N_MU);
205  _Bool mod = false;
206  int orig;
207  for(int i = 0; i < dgp->n; i++) {
208  // mutate functions
209  if(rand_uniform(0,1) < dgp->mu[0]) {
210  orig = dgp->function[i];
211  dgp->function[i] = irand_uniform(0, NUM_FUNC);
212  if(orig != dgp->function[i]) {
213  mod = true;
214  }
215  }
216  // mutate connectivity map
217  int shift = i * xcsf->MAX_K;
218  for(int j = 0; j < xcsf->MAX_K; j++) {
219  int idx = shift + j;
220  if(rand_uniform(0,1) < dgp->mu[1]) {
221  orig = dgp->connectivity[idx];
222  // external connection
223  if(rand_uniform(0,1) < 0.5) {
224  dgp->connectivity[idx] = -(irand_uniform(1,xcsf->x_dim+1));
225  }
226  // another node
227  else {
228  dgp->connectivity[idx] = irand_uniform(0,dgp->n);
229  }
230  if(orig != dgp->connectivity[idx]) {
231  mod = true;
232  }
233  }
234  }
235  }
236  // mutate number of update cycles
237  if(rand_uniform(0,1) < dgp->mu[2]) {
238  orig = dgp->t;
239  dgp->t = irand_uniform(1,xcsf->MAX_T);
240  if(orig != dgp->t) {
241  mod = true;
242  }
243  }
244  return mod;
245 }

257 _Bool graph_crossover(const XCSF *xcsf, GRAPH *dgp1, GRAPH *dgp2)
258 {
259  if(rand_uniform(0,1) > xcsf->P_CROSSOVER) {
260  return false;
261  }
262  // number of update cycles
263  if(rand_uniform(0,1) < 0.5) {
264  int tmp = dgp1->t;
265  dgp1->t = dgp2->t;
266  dgp2->t = tmp;
267  }
268  for(int i = 0; i < dgp1->n; i++) {
269  // functions
270  if(rand_uniform(0,1) < 0.5) {
271  int tmp = dgp1->function[i];
272  dgp1->function[i] = dgp2->function[i];
273  dgp2->function[i] = tmp;
274  }
275  // initial states
276  if(rand_uniform(0,1) < 0.5) {
277  double tmp = dgp1->initial_state[i];
278  dgp1->initial_state[i] = dgp2->initial_state[i];
279  dgp2->initial_state[i] = tmp;
280  }
281  // states
282  if(rand_uniform(0,1) < 0.5) {
283  double tmp = dgp1->state[i];
284  dgp1->state[i] = dgp2->state[i];
285  dgp2->state[i] = tmp;
286  }
287  }
288  // connectivity map
289  for(int i = 0; i < dgp1->klen; i++) {
290  if(rand_uniform(0,1) < 0.5) {
291  double tmp = dgp1->connectivity[i];
292  dgp1->connectivity[i] = dgp2->connectivity[i];
293  dgp2->connectivity[i] = tmp;
294  }
295  }
296  return true;
297 }

306 static double node_activate(int function, const double *inputs, int k)
307 {
308  double state = 0;
309  switch(function)
310  {
311  case 0: // Fuzzy NOT
312  state = 1 - inputs[0];
313  break;
314  case 1: // Fuzzy AND (CFMQVS)
315  state = inputs[0];
316  for(int i = 1; i < k; i++) {
317  state *= inputs[i];
318  }
319  break;
320  case 2: // Fuzzy OR (CFMQVS)
321  state = inputs[0];
322  for(int i = 1; i < k; i++) {
323  state += inputs[i];
324  }
325  break;
326  default: // Invalid function
327  printf("Error updating node: Invalid function: %d\n", function);
328  exit(EXIT_FAILURE);
329  }
330  state = constrain(0, 1, state);
331  return state;
332 }
