46 double xcs_multi_step_exp(XCSF *xcsf)
47 {
48  pa_init(xcsf);
49  double error = 0; // prediction error: individual trial
50  double werr = 0; // prediction error: windowed total
51  double tperf = 0; // steps to goal: total over all trials
52  double wperf = 0; // steps to goal: windowed total
53  for(int cnt = 0; cnt < xcsf->MAX_TRIALS; cnt++) {
54  xcs_multi_trial(xcsf, &error, true); // explore
55  double perf = xcs_multi_trial(xcsf, &error, false); // exploit
56  wperf += perf;
57  tperf += perf;
58  werr += error;
59  perf_print(xcsf, &wperf, &werr, cnt);
60  }
61  pa_free(xcsf);
62  return tperf / xcsf->MAX_TRIALS;
63 }

72 static double xcs_multi_trial(XCSF *xcsf, double *error, _Bool explore)
73 {
74  env_reset(xcsf);
75  xcsf->train = explore;
76  _Bool reset = false;
77  double prev_reward = 0;
78  double prev_pred = 0;
79  double *prev_state = malloc(sizeof(double) * xcsf->x_dim);
80  clset_init(&xcsf->prev_aset);
81  clset_init(&xcsf->kset);
82  *error = 0;
83  int steps = 0;
84  for(steps = 0; steps < xcsf->TELETRANSPORTATION && !reset; steps++) {
85  clset_init(&xcsf->mset);
86  clset_init(&xcsf->aset);
87  const double *state = env_get_state(xcsf);
88  int action = xcs_multi_decision(xcsf, state);
89  clset_action(xcsf, action);
90  double reward = env_execute(xcsf, action);
91  reset = env_is_reset(xcsf);
92  // update previous action set and run EA
93  if(xcsf->prev_aset.list != NULL) {
94  double payoff = prev_reward + (xcsf->GAMMA * pa_best_val(xcsf));
95  clset_validate(&xcsf->prev_aset);
96  clset_update(xcsf, &xcsf->prev_aset, prev_state, &payoff, false);
97  if(xcsf->train) {
98  ea(xcsf, &xcsf->prev_aset);
99  }
100  *error += fabs(xcsf->GAMMA * pa_val(xcsf, action)
101  + prev_reward - prev_pred) / env_max_payoff(xcsf);
102  }
103  // in goal state: update current action set and run EA
104  if(reset) {
105  clset_validate(&xcsf->aset);
106  clset_update(xcsf, &xcsf->aset, state, &reward, true);
107  if(xcsf->train) {
108  ea(xcsf, &xcsf->aset);
109  }
110  *error += fabs(reward - pa_val(xcsf, action)) / env_max_payoff(xcsf);
111  }
112  // next step
113  clset_free(&xcsf->mset);
114  clset_free(&xcsf->prev_aset);
115  xcsf->prev_aset = xcsf->aset;
116  prev_reward = reward;
117  prev_pred = pa_val(xcsf, action);
118  memcpy(prev_state, state, sizeof(double) * xcsf->x_dim);
119  }
120  clset_free(&xcsf->prev_aset);
121  clset_kill(xcsf, &xcsf->kset);
122  free(prev_state);
123  *error /= steps;
124  return steps;
125 }
