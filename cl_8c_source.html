123 double cl_del_vote(const XCSF *xcsf, const CL *c, double avg_fit)
124 {
125  if(c->exp > xcsf->THETA_DEL && c->fit / c->num < xcsf->DELTA * avg_fit) {

155 void cl_update(const XCSF *xcsf, CL *c, const double *x, const double *y, int set_num, _Bool cur)
156 {
157  c->exp++;
158  // propagate inputs for the previous state update
159  if(cur == false) {
160  cl_predict(xcsf, c, x);
161  }
162  cl_update_err(xcsf, c, y);
163  cl_update_size(xcsf, c, set_num);
164  cond_update(xcsf, c, x, y);
165  pred_update(xcsf, c, x, y);
166  act_update(xcsf, c, x, y);
167 }

177 static double cl_update_err(const XCSF *xcsf, CL *c, const double *y)
178 {
179  double error = (xcsf->loss_ptr)(xcsf, c->prediction, y);
180  if(c->exp < 1 / xcsf->BETA) {
181  c->err = (c->err * (c->exp - 1) + error) / c->exp;
182  }
183  else {
184  c->err += xcsf->BETA * (error - c->err);
185  }
186  return c->err * c->num;
187 }
188 
196 void cl_update_fit(const XCSF *xcsf, CL *c, double acc_sum, double acc)
197 {
198  c->fit += xcsf->BETA * ((acc * c->num) / acc_sum - c->fit);
199 }

208 static double cl_update_size(const XCSF *xcsf, CL *c, double num_sum)
209 {
210  if(c->exp < 1 / xcsf->BETA) {
211  c->size = (c->size * (c->exp - 1) + num_sum) / c->exp;
212  }
213  else {
214  c->size += xcsf->BETA * (num_sum - c->size);
215  }
216  return c->size * c->num;
217 }
