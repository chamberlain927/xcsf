123 double cl_del_vote(const XCSF *xcsf, CL *c, double avg_fit)
124 {
125  if(c->exp > xcsf->THETA_DEL && c->fit / c->num < xcsf->DELTA * avg_fit) {

174 static double cl_update_err(const XCSF *xcsf, CL *c, const double *y, _Bool cur)
175 {
176  double error = 0;
177  if(cur) {
178  error = (xcsf->loss_ptr)(xcsf, c->prediction, y);
179  }
180  else {
181  error = (xcsf->loss_ptr)(xcsf, c->prev_prediction, y);
182  }
183  if(c->exp < 1 / xcsf->BETA) {
184  c->err = (c->err * (c->exp - 1) + error) / c->exp;
185  }
186  else {
187  c->err += xcsf->BETA * (error - c->err);
188  }
189  return c->err * c->num;
190 }

199 void cl_update_fit(const XCSF *xcsf, CL *c, double acc_sum, double acc)
200 {
201  c->fit += xcsf->BETA * ((acc * c->num) / acc_sum - c->fit);
202 }

211 static double cl_update_size(const XCSF *xcsf, CL *c, double num_sum)
212 {
213  if(c->exp < 1 / xcsf->BETA) {
214  c->size = (c->size * (c->exp - 1) + num_sum) / c->exp;
215  }
216  else {
217  c->size += xcsf->BETA * (num_sum - c->size);
218  }
219  return c->size * c->num;
220 }
