119 double cl_del_vote(const XCSF *xcsf, const CL *c, double avg_fit)
120 {
121  if(c->exp > xcsf->THETA_DEL && c->fit / c->num < xcsf->DELTA * avg_fit) {
122  return c->size * c->num * avg_fit / (c->fit / c->num);
123  }
124  return c->size * c->num;
125 }

151 void cl_update(const XCSF *xcsf, CL *c, const double *x, const double *y, int set_num, _Bool cur)
152 {
153  c->exp++;
154  // propagate inputs for the previous state update
155  if(cur == false) {
156  cl_predict(xcsf, c, x);
157  }
158  cl_update_err(xcsf, c, y);
159  cl_update_size(xcsf, c, set_num);
160  cond_update(xcsf, c, x, y);
161  pred_update(xcsf, c, x, y);
162  act_update(xcsf, c, x, y);

172 static void cl_update_err(const XCSF *xcsf, CL *c, const double *y)
173 {
174  double error = (xcsf->loss_ptr)(xcsf, c->prediction, y);
175  if(c->exp < 1 / xcsf->BETA) {
176  c->err = (c->err * (c->exp - 1) + error) / c->exp;
177  }
178  else {
179  c->err += xcsf->BETA * (error - c->err);
180  }
181 }

190 void cl_update_fit(const XCSF *xcsf, CL *c, double acc_sum, double acc)
191 {
192  c->fit += xcsf->BETA * ((acc * c->num) / acc_sum - c->fit);
193 }

201 static void cl_update_size(const XCSF *xcsf, CL *c, int num_sum)
202 {
203  if(c->exp < 1 / xcsf->BETA) {
204  c->size = (c->size * (c->exp - 1) + num_sum) / c->exp;
205  }
206  else {
207  c->size += xcsf->BETA * (num_sum - c->size);
208  }

361 {
