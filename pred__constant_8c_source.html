48 void pred_constant_update(const XCSF *xcsf, const CL *c, const double *x, const double *y)
49 {
50  (void)x;
51  if(c->exp < 1.0 / xcsf->PRED_ETA) {
52  for(int var = 0; var < xcsf->y_dim; var++) {
53  c->prediction[var] = (c->prediction[var] * (c->exp-1.0) + y[var]) / (double)c->exp;
54  }
55  }
56  else {
57  for(int var = 0; var < xcsf->y_dim; var++) {
58  c->prediction[var] += xcsf->PRED_ETA * (y[var] - c->prediction[var]);
59  }
60  }
61 }
