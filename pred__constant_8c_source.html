50 void pred_constant_update(XCSF *xcsf, CL *c, const double *x, const double *y)
51 {
52  (void)x;
53  if(c->exp < 1.0 / xcsf->PRED_ETA) {
54  for(int var = 0; var < xcsf->num_y_vars; var++) {
55  c->prediction[var] = (c->prediction[var] * (c->exp-1.0) + y[var]) / (double)c->exp;
56  }
57  }
58  else {
59  for(int var = 0; var < xcsf->num_y_vars; var++) {
60  c->prediction[var] += xcsf->PRED_ETA * (y[var] - c->prediction[var]);
61  }
62  }
63 }
