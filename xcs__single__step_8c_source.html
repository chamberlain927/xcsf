45 double xcs_single_step_exp(XCSF *xcsf)
46 {
47  pa_init(xcsf);
48  double error = 0; // prediction error: individual trial
49  double werr = 0; // prediction error: windowed total
50  double tperf = 0; // accuracy: total over all trials
51  double wperf = 0; // accuracy: windowed total
52  for(int cnt = 0; cnt < xcsf->MAX_TRIALS; cnt++) {
53  xcs_single_trial(xcsf, &error, true); // explore
54  double perf = xcs_single_trial(xcsf, &error, false); // exploit
55  wperf += perf;
56  tperf += perf;
57  werr += error;
58  perf_print(xcsf, &wperf, &werr, cnt);
59  }
60  pa_free(xcsf);
61  return tperf / xcsf->MAX_TRIALS;
62 }

71 static double xcs_single_trial(XCSF *xcsf, double *error, _Bool explore)
72 {
73  xcs_single_init(xcsf);
74  xcsf->train = explore;
75  const double *x = env_get_state(xcsf);
76  int action = xcs_single_decision(xcsf, x);
77  double reward = env_execute(xcsf, action);
78  *error = xcs_single_error(xcsf, reward);
79  if(explore) {
80  xcs_single_update(xcsf, x, action, reward);
81  }
82  xcs_single_free(xcsf);
83  return (reward > 0) ? 1 : 0;
84 }
