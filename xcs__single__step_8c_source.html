46 double xcs_single_step_exp(XCSF *xcsf)
47 {
48  gplot_init(xcsf);
49  pa_init(xcsf);
50  double err = 0; // total error over all trials
51  double perr = 0; // windowed accuracy for averaging
52  double pterr = 0; // windowed prediction error for averaging
53  double perf = 0; // individual trial accuracy
54  for(int cnt = 0; cnt < xcsf->MAX_TRIALS; cnt++) {
55  xcs_single_trial(xcsf, &perf, true); // explore
56  double error = xcs_single_trial(xcsf, &perf, false); // exploit
57  perr += perf;
58  err += error;
59  pterr += error;
60  if(cnt % xcsf->PERF_AVG_TRIALS == 0 && cnt > 0) {
61  disp_perf2(xcsf, perr/xcsf->PERF_AVG_TRIALS, pterr/xcsf->PERF_AVG_TRIALS, cnt);
62  perr = 0;
63  pterr = 0;
64  }
65  }
66  gplot_free(xcsf);
67  pa_free(xcsf);
68  return err / xcsf->MAX_TRIALS;
69 }

78 static double xcs_single_trial(XCSF *xcsf, double *perf, _Bool explore)
79 {
80  SET mset; // match set
81  SET aset; // action set
82  SET kset; // kill set
83  clset_init(xcsf, &mset);
84  clset_init(xcsf, &aset);
85  clset_init(xcsf, &kset);
86  xcsf->train = explore;
87  const double *x = env_get_state(xcsf);
88  int action = xcs_single_decision(xcsf, &mset, &kset, x);
89  double reward = env_execute(xcsf, action);
90  if(reward > 0) {
91  *perf = 1;
92  }
93  else {
94  *perf = 0;
95  }
96  if(explore) {
97  xcs_single_update(xcsf, &mset, &aset, &kset, x, action, reward);
98  }
99  clset_kill(xcsf, &kset);
100  clset_free(xcsf, &aset);
101  clset_free(xcsf, &mset);
102  return xcs_single_error(xcsf, reward);
103 }

113 int xcs_single_decision(XCSF *xcsf, SET *mset, SET *kset, const double *x)
114 {
115  clset_match(xcsf, mset, kset, x);
116  xcsf->msetsize += (mset->size - xcsf->msetsize) * xcsf->BETA;
117  pa_build(xcsf, mset, x);
118  if(xcsf->train && rand_uniform(0,1) < xcsf->P_EXPLORE) {
119  return pa_rand_action(xcsf);
120  }
121  return pa_best_action(xcsf);
122 }
