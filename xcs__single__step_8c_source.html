47 double xcs_single_step_exp(XCSF *xcsf)
48 {
49  gplot_init(xcsf);
50  pa_init(xcsf);
51  double perr = 0, err = 0, pterr = 0;
52  for(int cnt = 0; cnt < xcsf->MAX_TRIALS; cnt++) {
53  xcs_single_explore(xcsf);
54  double error = 0;
55  perr += xcs_single_exploit(xcsf, &error);
56  err += error;
57  pterr += error;
58  if(cnt % xcsf->PERF_AVG_TRIALS == 0 && cnt > 0) {
59  disp_perf2(xcsf, perr/xcsf->PERF_AVG_TRIALS, pterr/xcsf->PERF_AVG_TRIALS, cnt);
60  perr = 0; pterr = 0;
61  }
62  }
63  gplot_free(xcsf);
64  pa_free(xcsf);
65  return err / xcsf->MAX_TRIALS;
66 }

72 void xcs_single_explore(XCSF *xcsf)
73 {
74  xcsf->train = true;
75  double *x = env_get_state(xcsf);
76  SET mset, aset, kset;
77  set_init(xcsf, &mset);
78  set_init(xcsf, &aset);
79  set_init(xcsf, &kset);
80  set_match(xcsf, &mset, &kset, x);
81  pa_build(xcsf, &mset, x);
82  int action = 0;
83  if(rand_uniform(0,1) < xcsf->P_EXPLORE) {
84  action = pa_rand_action(xcsf);
85  }
86  else {
87  action = pa_best_action(xcsf);
88  }
89  double reward = env_execute(xcsf, action);
90  set_action(xcsf, &mset, &aset, action);
91  set_update(xcsf, &aset, &kset, x, &reward, true);
92  ea(xcsf, &aset, &kset);
93  xcsf->time += 1;
94  xcsf->msetsize += (mset.size - xcsf->msetsize) * xcsf->BETA;
95  set_kill(xcsf, &kset);
96  set_free(xcsf, &aset);
97  set_free(xcsf, &mset);
98 }

106 double xcs_single_exploit(XCSF *xcsf, double *error)
107 {
108  xcsf->train = false;
109  double *x = env_get_state(xcsf);
110  SET mset, aset, kset;
111  set_init(xcsf, &mset);
112  set_init(xcsf, &aset);
113  set_init(xcsf, &kset);
114  set_match(xcsf, &mset, &kset, x);
115  pa_build(xcsf, &mset, x);
116  int action = pa_best_action(xcsf);
117  double reward = env_execute(xcsf, action);
118  set_action(xcsf, &mset, &aset, action);
119  xcsf->msetsize += (mset.size - xcsf->msetsize) * xcsf->BETA;
120  set_kill(xcsf, &kset);
121  set_free(xcsf, &aset);
122  set_free(xcsf, &mset);
123  *error = fabs(reward - pa_best_val(xcsf));
124  if(reward > 0) {
125  return 1;
126  }
127  return 0;
128 }
