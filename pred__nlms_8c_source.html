43 void pred_nlms_init(const XCSF *xcsf, CL *c)
44 {
45  PRED_NLMS *pred = malloc(sizeof(PRED_NLMS));
46  c->pred = pred;
47  if(xcsf->PRED_TYPE == PRED_TYPE_NLMS_QUADRATIC) {
48  // offset(1) + n linear + n quadratic + n*(n-1)/2 mixed terms
49  pred->weights_length = 1 + 2 * xcsf->x_dim +
50  xcsf->x_dim * (xcsf->x_dim - 1) / 2;
51  }
52  else {
53  pred->weights_length = xcsf->x_dim + 1;
54  }
55  pred->weights = malloc(sizeof(double*) * xcsf->y_dim);
56  for(int var = 0; var < xcsf->y_dim; var++) {
57  pred->weights[var] = malloc(sizeof(double) * pred->weights_length);
58  }
59  for(int var = 0; var < xcsf->y_dim; var++) {
60  pred->weights[var][0] = xcsf->PRED_X0;
61  for(int i = 1; i < pred->weights_length; i++) {
62  pred->weights[var][i] = 0;
63  }
64  }
65 }

88 void pred_nlms_update(const XCSF *xcsf, const CL *c, const double *x, const double *y)
89 {
90  const PRED_NLMS *pred = c->pred;
91  double norm = xcsf->PRED_X0 * xcsf->PRED_X0;
92  for(int i = 0; i < xcsf->x_dim; i++) {
93  norm += x[i] * x[i];
94  }
95  // prediction must have been computed for the current state
96  for(int var = 0; var < xcsf->y_dim; var++) {
97  double error = y[var] - c->prediction[var];
98  double correction = (xcsf->PRED_ETA * error) / norm;
99  // update first coefficient
100  pred->weights[var][0] += xcsf->PRED_X0 * correction;
101  int index = 1;
102  // update linear coefficients
103  for(int i = 0; i < xcsf->x_dim; i++) {
104  pred->weights[var][index++] += correction * x[i];
105  }
106  if(xcsf->PRED_TYPE == PRED_TYPE_NLMS_QUADRATIC) {
107  // update quadratic coefficients
108  for(int i = 0; i < xcsf->x_dim; i++) {
109  for(int j = i; j < xcsf->x_dim; j++) {
110  pred->weights[var][index++] += correction * x[i] * x[j];
111  }
112  }
113  }
114  }
115 }

117 const double *pred_nlms_compute(const XCSF *xcsf, const CL *c, const double *x)
118 {
119  const PRED_NLMS *pred = c->pred;
120  for(int var = 0; var < xcsf->y_dim; var++) {
121  // first coefficient is offset
122  double pre = xcsf->PRED_X0 * pred->weights[var][0];
123  int index = 1;
124  // multiply linear coefficients with the prediction input
125  for(int i = 0; i < xcsf->x_dim; i++) {
126  pre += pred->weights[var][index++] * x[i];
127  }
128  if(xcsf->PRED_TYPE == PRED_TYPE_NLMS_QUADRATIC) {
129  // multiply quadratic coefficients with prediction input
130  for(int i = 0; i < xcsf->x_dim; i++) {
131  for(int j = i; j < xcsf->x_dim; j++) {
132  pre += pred->weights[var][index++] * x[i] * x[j];
133  }
134  }
135  }
136  c->prediction[var] = pre;
137  }
138  return c->prediction;
139 }
