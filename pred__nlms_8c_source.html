51 void pred_nlms_init(const XCSF *xcsf, CL *c)
52 {
53  PRED_NLMS *pred = malloc(sizeof(PRED_NLMS));
54  c->pred = pred;
55  // set the length of weights per predicted variable
56  if(xcsf->PRED_TYPE == PRED_TYPE_NLMS_QUADRATIC) {
57  // offset(1) + n linear + n quadratic + n*(n-1)/2 mixed terms
58  pred->n = 1 + 2 * xcsf->x_dim + xcsf->x_dim * (xcsf->x_dim - 1) / 2;
59  }
60  else {
61  pred->n = xcsf->x_dim + 1;
62  }
63  // initialise weights
64  pred->n_weights = pred->n * xcsf->y_dim;
65  pred->weights = calloc(pred->n_weights, sizeof(double));
66  blas_fill(xcsf->y_dim, xcsf->PRED_X0, pred->weights, pred->n);
67  // initialise learning rate
68  if(xcsf->PRED_EVOLVE_ETA) {
69  sam_init(xcsf, pred->mu, N_MU);
70  pred->eta = rand_uniform(ETA_MIN, ETA_MAX);
71  }
72  else {
73  memset(pred->mu, 0, sizeof(double) * N_MU);
74  pred->eta = xcsf->PRED_ETA;
75  }
76 }

96 void pred_nlms_update(const XCSF *xcsf, const CL *c, const double *x, const double *y)
97 {
98  const PRED_NLMS *pred = c->pred;
99  int n = pred->n;
100  double norm = xcsf->PRED_X0 * xcsf->PRED_X0;
101  for(int i = 0; i < xcsf->x_dim; i++) {
102  norm += x[i] * x[i];
103  }
104  // update weights using the error
105  for(int var = 0; var < xcsf->y_dim; var++) {
106  double error = y[var] - c->prediction[var];
107  double correction = (pred->eta * error) / norm;
108  // update first coefficient
109  pred->weights[var*n] += xcsf->PRED_X0 * correction;
110  int idx = 1;
111  // update linear coefficients
112  for(int i = 0; i < xcsf->x_dim; i++, idx++) {
113  pred->weights[var*n+idx] += correction * x[i];
114  }
115  if(xcsf->PRED_TYPE == PRED_TYPE_NLMS_QUADRATIC) {
116  // update quadratic coefficients
117  for(int i = 0; i < xcsf->x_dim; i++) {
118  for(int j = i; j < xcsf->x_dim; j++, idx++) {
119  pred->weights[var*n+idx] += correction * x[i] * x[j];
120  }
121  }
122  }
123  }
124 }

126 void pred_nlms_compute(const XCSF *xcsf, const CL *c, const double *x)
127 {
128  const PRED_NLMS *pred = c->pred;
129  int n = pred->n;
130  for(int var = 0; var < xcsf->y_dim; var++) {
131  // first coefficient is offset
132  double pre = xcsf->PRED_X0 * pred->weights[var*n];
133  int idx = 1;
134  // multiply linear coefficients with the prediction input
135  for(int i = 0; i < xcsf->x_dim; i++, idx++) {
136  pre += pred->weights[var*n+idx] * x[i];
137  }
138  if(xcsf->PRED_TYPE == PRED_TYPE_NLMS_QUADRATIC) {
139  // multiply quadratic coefficients with prediction input
140  for(int i = 0; i < xcsf->x_dim; i++) {
141  for(int j = i; j < xcsf->x_dim; j++, idx++) {
142  pre += pred->weights[var*n+idx] * x[i] * x[j];
143  }
144  }
145  }
146  c->prediction[var] = pre;
147  }
148 }

169 _Bool pred_nlms_mutate(const XCSF *xcsf, const CL *c)
170 {
171  if(xcsf->PRED_EVOLVE_ETA) {
172  PRED_NLMS *pred = c->pred;
173  sam_adapt(xcsf, pred->mu, N_MU);
174  double orig = pred->eta;
175  pred->eta += rand_normal(0, pred->mu[0]);
176  pred->eta = constrain(ETA_MIN, ETA_MAX, pred->eta);
177  if(orig != pred->eta) {
178  return true;
179  }
180  }
181  return false;
182 }
