48 void ea(XCSF *xcsf, SET *set, SET *kset)
49 {
50  // check if the evolutionary algorithm should be run
51  if(set->size == 0 || xcsf->time - set_mean_time(xcsf, set) < xcsf->THETA_EA) {
52  return;
53  }
54  set_times(xcsf, set);
55  // select parents
56  CL *c1p; CL *c2p;
57  if(xcsf->EA_SELECT_TYPE == EA_SELECT_ROULETTE) {
58  double fit_sum = set_total_fit(xcsf, set);
59  c1p = ea_select_rw(xcsf, set, fit_sum);
60  c2p = ea_select_rw(xcsf, set, fit_sum);
61  }
62  else {
63  c1p = ea_select_tournament(xcsf, set);
64  c2p = ea_select_tournament(xcsf, set);
65  }
66  // create offspring
67  for(int i = 0; i < xcsf->LAMBDA/2; i++) {
68  // create copies of parents
69  CL *c1 = malloc(sizeof(CL));
70  CL *c2 = malloc(sizeof(CL));
71  cl_init(xcsf, c1, c1p->size, c1p->time);
72  cl_init(xcsf, c2, c2p->size, c2p->time);
73  cl_copy(xcsf, c1, c1p);
74  cl_copy(xcsf, c2, c2p);
75  // adapt mutation rates
76  sam_adapt(xcsf, c1->mu);
77  sam_adapt(xcsf, c2->mu);
78  // apply evolutionary operators to offspring
79  _Bool cmod = cl_crossover(xcsf, c1, c2);
80  _Bool m1mod = cl_mutate(xcsf, c1);
81  _Bool m2mod = cl_mutate(xcsf, c2);
82  // reduce offspring err, fit
83  if(cmod) {
84  c1->err = xcsf->ERR_REDUC * ((c1p->err + c2p->err)/2.0);
85  c2->err = c1->err;
86  c1->fit = c1p->fit / c1p->num;
87  c2->fit = c2p->fit / c2p->num;
88  c1->fit = xcsf->FIT_REDUC * ((c1->fit + c2->fit)/2.0);
89  c2->fit = c1->fit;
90  }
91  else {
92  c1->err = xcsf->ERR_REDUC * c1p->err;
93  c2->err = xcsf->ERR_REDUC * c2p->err;
94  c1->fit = xcsf->FIT_REDUC * (c1p->fit / c1p->num);
95  c2->fit = xcsf->FIT_REDUC * (c2p->fit / c2p->num);
96  }
97  // add offspring to population
98  if(xcsf->EA_SUBSUMPTION) {
99  // c1 no crossover or mutation changes
100  if(!cmod && !m1mod) {
101  c1p->num++;
102  xcsf->pset.num++;
103  cl_free(xcsf, c1);
104  }
105  else {
106  ea_subsume(xcsf, c1, c1p, c2p, set);
107  }
108  // c2 no crossover or mutation changes
109  if(!cmod && !m2mod) {
110  c2p->num++;
111  xcsf->pset.num++;
112  cl_free(xcsf, c2);
113  }
114  else {
115  ea_subsume(xcsf, c2, c1p, c2p, set);
116  }
117  }
118  else {
119  set_add(xcsf, &xcsf->pset, c1);
120  set_add(xcsf, &xcsf->pset, c2);
121  }
122  }
123  pop_enforce_limit(xcsf, kset);
124 }

134 void ea_subsume(XCSF *xcsf, CL *c, CL *c1p, CL *c2p, SET *set)
135 {
136  // check if either parent subsumes the offspring
137  if(cl_subsumer(xcsf, c1p) && cl_general(xcsf, c1p, c)) {
138  c1p->num++;
139  xcsf->pset.num++;
140  cl_free(xcsf, c);
141  }
142  else if(cl_subsumer(xcsf, c2p) && cl_general(xcsf, c2p, c)) {
143  c2p->num++;
144  xcsf->pset.num++;
145  cl_free(xcsf, c);
146  }
147  // attempt to find a random subsumer from the set
148  else {
149  CLIST *candidates[set->size];
150  int choices = 0;
151  for(CLIST *iter = set->list; iter != NULL; iter = iter->next) {
152  if(cl_subsumer(xcsf, iter->cl) && cl_general(xcsf, iter->cl, c)) {
153  candidates[choices] = iter;
154  choices++;
155  }
156  }
157  // found
158  if(choices > 0) {
159  candidates[irand_uniform(0,choices)]->cl->num++;
160  xcsf->pset.num++;
161  cl_free(xcsf, c);
162  }
163  // if no subsumers are found the offspring is added to the population
164  else {
165  set_add(xcsf, &xcsf->pset, c);
166  }
167  }
168 }

177 CL *ea_select_rw(XCSF *xcsf, SET *set, double fit_sum)
178 {
179  (void)xcsf;
180  double p = rand_uniform(0, fit_sum);
181  CLIST *iter = set->list;
182  double sum = iter->cl->fit;
183  while(p > sum) {
184  iter = iter->next;
185  sum += iter->cl->fit;
186  }
187  return iter->cl;
188 }

196 CL *ea_select_tournament(XCSF *xcsf, SET *set)
197 {
198  CL *winner = NULL;
199  while(winner == NULL) {
200  for(CLIST *iter = set->list; iter != NULL; iter = iter->next) {
201  if(rand_uniform(0,1) < xcsf->EA_SELECT_SIZE) {
202  if(winner == NULL || iter->cl->fit > winner->fit) {
203  winner = iter->cl;
204  }
205  }
206  }
207  }
208  return winner;
209 }
