51 void ea(XCSF *xcsf, const SET *set, SET *kset)
52 {
53  // increase EA time
54  xcsf->time += 1;
55  // check if the EA should be run
56  if(set->size == 0 || xcsf->time - clset_mean_time(xcsf, set) < xcsf->THETA_EA) {
57  return;
58  }
59  clset_set_times(xcsf, set);
60  // select parents
61  CL *c1p; CL *c2p;
62  ea_select_parents(xcsf, set, &c1p, &c2p);
63  // create offspring
64  for(int i = 0; i < xcsf->LAMBDA/2; i++) {
65  // create copies of parents
66  CL *c1 = malloc(sizeof(CL));
67  CL *c2 = malloc(sizeof(CL));
68  cl_init(xcsf, c1, c1p->size, c1p->time);
69  cl_init(xcsf, c2, c2p->size, c2p->time);
70  cl_copy(xcsf, c1, c1p);
71  cl_copy(xcsf, c2, c2p);
72  // adapt mutation rates
73  sam_adapt(xcsf, c1->mu);
74  sam_adapt(xcsf, c2->mu);
75  // apply evolutionary operators to offspring
76  _Bool cmod = cl_crossover(xcsf, c1, c2);
77  _Bool m1mod = cl_mutate(xcsf, c1);
78  _Bool m2mod = cl_mutate(xcsf, c2);
79  // initialise parameters
80  ea_init_offspring(xcsf, c1p, c2p, c1, c2, cmod);
81  // add to population
82  ea_add(xcsf, set, c1p, c2p, c1, cmod, m1mod);
83  ea_add(xcsf, set, c2p, c1p, c2, cmod, m2mod);
84  }
85  clset_pop_enforce_limit(xcsf, kset);
86 }

97 static void ea_init_offspring(const XCSF *xcsf, const CL *c1p, const CL *c2p, CL *c1, CL *c2, _Bool cmod)
98 {
99  if(cmod) {
100  c1->err = xcsf->ERR_REDUC * ((c1p->err + c2p->err)/2.0);
101  c2->err = c1->err;
102  c1->fit = c1p->fit / c1p->num;
103  c2->fit = c2p->fit / c2p->num;
104  c1->fit = xcsf->FIT_REDUC * ((c1->fit + c2->fit)/2.0);
105  c2->fit = c1->fit;
106  }
107  else {
108  c1->err = xcsf->ERR_REDUC * c1p->err;
109  c2->err = xcsf->ERR_REDUC * c2p->err;
110  c1->fit = xcsf->FIT_REDUC * (c1p->fit / c1p->num);
111  c2->fit = xcsf->FIT_REDUC * (c2p->fit / c2p->num);
112  }

148 static void ea_subsume(XCSF *xcsf, CL *c, CL *c1p, CL *c2p, const SET *set)
149 {
150  // check if either parent subsumes the offspring
151  if(cl_subsumer(xcsf, c1p) && cl_general(xcsf, c1p, c)) {
152  c1p->num++;
153  xcsf->pset.num++;
154  cl_free(xcsf, c);
155  }
156  else if(cl_subsumer(xcsf, c2p) && cl_general(xcsf, c2p, c)) {
157  c2p->num++;
158  xcsf->pset.num++;
159  cl_free(xcsf, c);
160  }
161  // attempt to find a random subsumer from the set
162  else {
163  CLIST *candidates[set->size];
164  int choices = 0;
165  for(CLIST *iter = set->list; iter != NULL; iter = iter->next) {
166  if(cl_subsumer(xcsf, iter->cl) && cl_general(xcsf, iter->cl, c)) {
167  candidates[choices] = iter;
168  choices++;
169  }
170  }
171  // found
172  if(choices > 0) {
173  candidates[irand_uniform(0,choices)]->cl->num++;
174  xcsf->pset.num++;
175  cl_free(xcsf, c);
176  }
177  // if no subsumers are found the offspring is added to the population
178  else {
179  clset_add(xcsf, &xcsf->pset, c);
180  }
181  }
182 }

211 static CL *ea_select_rw(const XCSF *xcsf, const SET *set, double fit_sum)
212 {
213  (void)xcsf;
214  double p = rand_uniform(0, fit_sum);
215  const CLIST *iter = set->list;
216  double sum = iter->cl->fit;
217  while(p > sum) {
218  iter = iter->next;
219  sum += iter->cl->fit;
220  }
221  return iter->cl;
222 }

230 static CL *ea_select_tournament(const XCSF *xcsf, const SET *set)
231 {
232  CL *winner = NULL;
233  while(winner == NULL) {
234  for(const CLIST *iter = set->list; iter != NULL; iter = iter->next) {
235  if((rand_uniform(0,1) < xcsf->EA_SELECT_SIZE) &&
236  (winner == NULL || iter->cl->fit > winner->fit)) {
237  winner = iter->cl;
238  }
239  }
240  }
241  return winner;
242 }
