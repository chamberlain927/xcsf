52 void pred_rls_init(const XCSF *xcsf, CL *c)
53 {
54  PRED_RLS *pred = malloc(sizeof(PRED_RLS));
55  c->pred = pred;
56  // set length of weights
57  if(xcsf->PRED_TYPE == PRED_TYPE_RLS_QUADRATIC) {
58  // offset(1) + n linear + n quadratic + n*(n-1)/2 mixed terms
59  pred->weights_length = 1 + 2 * xcsf->num_x_vars +
60  xcsf->num_x_vars * (xcsf->num_x_vars - 1) / 2;
61  }
62  else {
63  pred->weights_length = xcsf->num_x_vars + 1;
64  }
65  // initialise weights
66  pred->weights = malloc(sizeof(double*) * xcsf->num_y_vars);
67  for(int var = 0; var < xcsf->num_y_vars; var++) {
68  pred->weights[var] = malloc(sizeof(double)*pred->weights_length);
69  }
70  for(int var = 0; var < xcsf->num_y_vars; var++) {
71  pred->weights[var][0] = xcsf->PRED_X0;
72  for(int i = 1; i < pred->weights_length; i++) {
73  pred->weights[var][i] = 0;
74  }
75  }
76  // initialise gain matrix
77  int len_sqrd = pred->weights_length * pred->weights_length;
78  pred->matrix = malloc(sizeof(double) * len_sqrd);
79  init_matrix(xcsf, pred->matrix, pred->weights_length);
80  // initialise temporary storage for weight updating
81  pred->tmp_input = malloc(sizeof(double) * pred->weights_length);
82  pred->tmp_vec = malloc(sizeof(double) * pred->weights_length);
83  pred->tmp_matrix1 = malloc(sizeof(double) * len_sqrd);
84  pred->tmp_matrix2 = malloc(sizeof(double) * len_sqrd);
85 }

87 static void init_matrix(const XCSF *xcsf, double *matrix, int n)
88 {
89  for(int row = 0; row < n; row++) {
90  for(int col = 0; col < n; col++) {
91  if(row != col) {
92  matrix[row*n+col] = 0;
93  }
94  else {
95  matrix[row*n+col] = xcsf->PRED_RLS_SCALE_FACTOR;
96  }
97  }

127 void pred_rls_update(const XCSF *xcsf, const CL *c, const double *x, const double *y)
128 {
129  const PRED_RLS *pred = c->pred;
130  int n = pred->weights_length;
131  pred->tmp_input[0] = xcsf->PRED_X0;
132  int index = 1;
133  // linear terms
134  for(int i = 0; i < xcsf->num_x_vars; i++) {
135  pred->tmp_input[index++] = x[i];
136  }
137  // quadratic terms
138  if(xcsf->PRED_TYPE == PRED_TYPE_RLS_QUADRATIC) {
139  for(int i = 0; i < xcsf->num_x_vars; i++) {
140  for(int j = i; j < xcsf->num_x_vars; j++) {
141  pred->tmp_input[index++] = x[i] * x[j];
142  }
143  }
144  }
145  // determine gain vector = matrix * tmp_input
146  matrix_vector_multiply(pred->matrix, pred->tmp_input, pred->tmp_vec, n);
147  // divide gain vector by lambda + tmp_vec
148  double divisor = xcsf->PRED_RLS_LAMBDA;
149  for(int i = 0; i < n; i++) {
150  divisor += pred->tmp_input[i] * pred->tmp_vec[i];
151  }
152  for(int i = 0; i < n; i++) {
153  pred->tmp_vec[i] /= divisor;
154  }
155  // update weights using the error
156  // prediction must have been computed for the current state
157  for(int var = 0; var < xcsf->num_y_vars; var++) {
158  double error = y[var] - c->prediction[var];
159  for(int i = 0; i < n; i++) {
160  pred->weights[var][i] += error * pred->tmp_vec[i];
161  }
162  }
163  // update gain matrix
164  for(int i = 0; i < n; i++) {
165  for(int j = 0; j < n; j++) {
166  double tmp = pred->tmp_vec[i] * pred->tmp_input[j];
167  if(i == j) {
168  pred->tmp_matrix1[i*n+j] = 1 - tmp;
169  }
170  else {
171  pred->tmp_matrix1[i*n+j] = -tmp;
172  }
173  }
174  }
175  matrix_matrix_multiply(pred->tmp_matrix1, pred->matrix, pred->tmp_matrix2, n);
176  // divide gain matrix entries by lambda
177  for(int row = 0; row < n; row++) {
178  for(int col = 0; col < n; col++) {
179  pred->matrix[row*n+col] = pred->tmp_matrix2[row*n+col] / xcsf->PRED_RLS_LAMBDA;
180  }
181  }
182 }

184 const double *pred_rls_compute(const XCSF *xcsf, const CL *c, const double *x)
185 {
186  const PRED_RLS *pred = c->pred;
187  for(int var = 0; var < xcsf->num_y_vars; var++) {
188  // first coefficient is offset
189  double pre = xcsf->PRED_X0 * pred->weights[var][0];
190  int index = 1;
191  // multiply linear coefficients with the prediction input
192  for(int i = 0; i < xcsf->num_x_vars; i++) {
193  pre += pred->weights[var][index++] * x[i];
194  }
195  if(xcsf->PRED_TYPE == PRED_TYPE_RLS_QUADRATIC) {
196  // multiply quadratic coefficients with prediction input
197  for(int i = 0; i < xcsf->num_x_vars; i++) {
198  for(int j = i; j < xcsf->num_x_vars; j++) {
199  pre += pred->weights[var][index++] * x[i] * x[j];
200  }
201  }
202  }
203  c->prediction[var] = pre;
204  }
205  return c->prediction;
206 }

220 static void matrix_matrix_multiply(const double *srca, const double *srcb, double *dest, int n)
221 {
222  for(int i = 0; i < n; i++) {
223  for(int j = 0; j < n; j++) {
224  dest[i*n+j] = srca[i*n] * srcb[j];
225  for(int k = 1; k < n; k++) {
226  dest[i*n+j] += srca[i*n+k] * srcb[k*n+j];
227  }
228  }
229  }
230 }
