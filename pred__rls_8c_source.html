51 void pred_rls_init(const XCSF *xcsf, CL *c)
52 {
53  PRED_RLS *pred = malloc(sizeof(PRED_RLS));
54  c->pred = pred;
55  // set the length of weights per predicted variable
56  if(xcsf->PRED_TYPE == PRED_TYPE_RLS_QUADRATIC) {
57  // offset(1) + n linear + n quadratic + n*(n-1)/2 mixed terms
58  pred->n = 1 + 2 * xcsf->x_dim + xcsf->x_dim * (xcsf->x_dim - 1) / 2;
59  }
60  else {
61  pred->n = xcsf->x_dim + 1;
62  }
63  // initialise weights
64  pred->n_weights = pred->n * xcsf->y_dim;
65  pred->weights = calloc(pred->n_weights, sizeof(double));
66  blas_fill(xcsf->y_dim, xcsf->PRED_X0, pred->weights, pred->n);
67  // initialise gain matrix
68  int n_sqrd = pred->n * pred->n;
69  pred->matrix = malloc(n_sqrd * sizeof(double));
70  init_matrix(xcsf, pred->matrix, pred->n);
71  // initialise temporary storage for weight updating
72  pred->tmp_input = malloc(pred->n * sizeof(double));
73  pred->tmp_vec = calloc(pred->n, sizeof(double));
74  pred->tmp_matrix1 = calloc(n_sqrd, sizeof(double));
75  pred->tmp_matrix2 = calloc(n_sqrd, sizeof(double));
76 }

78 static void init_matrix(const XCSF *xcsf, double *matrix, int n)
79 {
80  for(int i = 0; i < n; i++) {
81  for(int j = 0; j < n; j++) {
82  if(i != j) {
83  matrix[i*n+j] = 0;
84  }
85  else {
86  matrix[i*n+j] = xcsf->PRED_RLS_SCALE_FACTOR;
87  }
88  }
89  }
90 }

113 void pred_rls_update(const XCSF *xcsf, const CL *c, const double *x, const double *y)
114 {
115  (void)x;
116  const PRED_RLS *pred = c->pred;
117  int n = pred->n;
118  // gain vector = matrix * tmp_input
119  blas_gemm(0, 0, n, 1, n, 1, pred->matrix, n, pred->tmp_input, 1, 0, pred->tmp_vec, 1);
120  // divide gain vector by lambda + tmp_vec
121  double divisor = blas_dot(n, pred->tmp_input, 1, pred->tmp_vec, 1);
122  divisor = 1 / (divisor + xcsf->PRED_RLS_LAMBDA);
123  blas_scal(n, divisor, pred->tmp_vec, 1);
124  // update weights using the error
125  for(int var = 0; var < xcsf->y_dim; var++) {
126  double error = y[var] - c->prediction[var];
127  blas_axpy(n, error, pred->tmp_vec, 1, &pred->weights[var*n], 1);
128  }
129  // update gain matrix
130  for(int i = 0; i < n; i++) {
131  for(int j = 0; j < n; j++) {
132  double tmp = pred->tmp_vec[i] * pred->tmp_input[j];
133  if(i == j) {
134  pred->tmp_matrix1[i*n+j] = 1 - tmp;
135  }
136  else {
137  pred->tmp_matrix1[i*n+j] = -tmp;
138  }
139  }
140  }
141  // tmp_matrix2 = tmp_matrix1 * pred_matrix
142  blas_gemm(0, 0, n, n, n, 1, pred->tmp_matrix1, n, pred->matrix, n, 0, pred->tmp_matrix2, n);
143  // divide gain matrix entries by lambda
144  for(int i = 0; i < n; i++) {
145  for(int j = 0; j < n; j++) {
146  pred->matrix[i*n+j] = pred->tmp_matrix2[i*n+j] / xcsf->PRED_RLS_LAMBDA;
147  }
148  }
149 }

151 void pred_rls_compute(const XCSF *xcsf, const CL *c, const double *x)
152 {
153  const PRED_RLS *pred = c->pred;
154  int n = pred->n;
155  pred_transform_input(xcsf, x, pred->tmp_input);
156  for(int var = 0; var < xcsf->y_dim; var++) {
157  c->prediction[var] = blas_dot(n, &pred->weights[var*n], 1, pred->tmp_input, 1);
158  }
159 }
