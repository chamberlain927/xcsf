142 static void sam_log_normal_adapt(const XCSF *xcsf, double *mu)
143 {
144  for(int i = 0; i < xcsf->SAM_NUM; i++) {
145  mu[i] *= exp(rand_normal(0,1));
146  if(mu[i] < xcsf->SAM_MIN) {
147  mu[i] = xcsf->SAM_MIN;
148  }
149  else if(mu[i] > 1) {
150  mu[i] = 1;
151  }
152  }
153 }
