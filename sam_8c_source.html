130 static void sam_log_normal_adapt(XCSF *xcsf, double *mu)
131 {
132  for(int i = 0; i < xcsf->SAM_NUM; i++) {
133  mu[i] *= exp(rand_normal(0,1));
134  if(mu[i] < xcsf->SAM_MIN) {
135  mu[i] = xcsf->SAM_MIN;
136  }
137  else if(mu[i] > 1) {
138  mu[i] = 1;
139  }
140  }
141 }
