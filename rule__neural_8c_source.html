55 void rule_neural_cond_init(const XCSF *xcsf, CL *c)
56 {
57  RULE_NEURAL *new = malloc(sizeof(RULE_NEURAL));
58  neural_init(xcsf, &new->net);
59  // hidden layers
60  uint32_t lopt = rule_neural_lopt(xcsf);
61  LAYER *l;
62  int i = 0;
63  int n_inputs = xcsf->num_x_vars;
64  while(i < MAX_LAYERS && xcsf->COND_NUM_NEURONS[i] > 0) {
65  int hinit = xcsf->COND_NUM_NEURONS[i];
66  int hmax = xcsf->COND_MAX_NEURONS[i];
67  if(hmax < hinit || !xcsf->COND_EVOLVE_NEURONS) {
68  hmax = hinit;
69  }
70  int f = xcsf->COND_HIDDEN_ACTIVATION;
71  l = neural_layer_connected_init(xcsf, n_inputs, hinit, hmax, f, lopt);
72  neural_layer_insert(xcsf, &new->net, l, i);
73  n_inputs = hinit;
74  i++;
75  }
76  // output layer
77  int f = xcsf->COND_OUTPUT_ACTIVATION;
78  lopt &= ~LAYER_EVOLVE_NEURONS; // never evolve the number of output neurons
79  int n = fmax(1, ceil(log2(xcsf->num_actions))); // number of action neurons
80  new->num_outputs = n;
81  l = neural_layer_connected_init(xcsf, n_inputs, n+1, n+1, f, lopt);
82  neural_layer_insert(xcsf, &new->net, l, i);
83  c->cond = new;
84 }

86 static uint32_t rule_neural_lopt(const XCSF *xcsf)
87 {
88  uint32_t lopt = 0;
89  if(xcsf->COND_EVOLVE_WEIGHTS) {
90  lopt |= LAYER_EVOLVE_WEIGHTS;
91  }
92  if(xcsf->COND_EVOLVE_NEURONS) {
93  lopt |= LAYER_EVOLVE_NEURONS;
94  }
95  if(xcsf->COND_EVOLVE_FUNCTIONS) {
96  lopt |= LAYER_EVOLVE_FUNCTIONS;
97  }
98  return lopt;
99 }

219 {

221  rule_neural_cond_rand(xcsf, c);

226 int rule_neural_act_compute(const XCSF *xcsf, const CL *c, const double *x)
227 {
228  (void)x; // network already updated
229  const RULE_NEURAL *cond = c->cond;
230  int action = 0;
231  for(int i = 0; i < cond->num_outputs; i++) {
232  if(neural_output(xcsf, &cond->net, i+1) > 0.5) {
233  action += pow(2,i);
234  }
235  }
236  action = iconstrain(0, xcsf->num_actions-1, action);
237  return action;
238 }
