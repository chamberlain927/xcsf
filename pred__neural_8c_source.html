52 void pred_neural_init(const XCSF *xcsf, CL *c)
53 {
54  PRED_NEURAL *new = malloc(sizeof(PRED_NEURAL));
55  neural_init(xcsf, &new->net);
56 
57  uint32_t lopt = pred_neural_lopt(xcsf);
58  int hmax = fmax(xcsf->PRED_MAX_HIDDEN_NEURONS, 1);
59  int hinit = xcsf->PRED_NUM_HIDDEN_NEURONS;
60  if(hinit < 1) {
61  hinit = irand_uniform(1, hmax);
62  }
63  if(hmax < hinit) {
64  hmax = hinit;
65  }
66  if(!xcsf->PRED_EVOLVE_NEURONS) {
67  hmax = hinit;
68  }
69 
70  // hidden layer
71  int f = xcsf->PRED_HIDDEN_ACTIVATION;
72  LAYER *l = neural_layer_connected_init(xcsf, xcsf->num_x_vars, hinit, hmax, f, lopt);
73  neural_layer_insert(xcsf, &new->net, l, 0);
74 
75  // output layer
76  f = xcsf->PRED_OUTPUT_ACTIVATION;
77  lopt &= ~LAYER_EVOLVE_NEURONS; // never evolve the number of output neurons
78  l = neural_layer_connected_init(xcsf, hinit, xcsf->num_y_vars, xcsf->num_y_vars, f, lopt);
79  neural_layer_insert(xcsf, &new->net, l, 1);
80 
81  c->pred = new;
82 }

84 static uint32_t pred_neural_lopt(const XCSF *xcsf)
85 {
86  uint32_t lopt = 0;
87  if(xcsf->PRED_SGD_WEIGHTS) {

90  if(xcsf->PRED_EVOLVE_WEIGHTS) {

92  }
93  if(xcsf->PRED_EVOLVE_ETA) {
94  lopt |= LAYER_EVOLVE_ETA;
95  }
96  if(xcsf->PRED_EVOLVE_NEURONS) {
97  lopt |= LAYER_EVOLVE_NEURONS;
98  }
99  if(xcsf->PRED_EVOLVE_FUNCTIONS) {
100  lopt |= LAYER_EVOLVE_FUNCTIONS;
101  }
102  return lopt;
103 }
