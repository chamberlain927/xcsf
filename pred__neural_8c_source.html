50 void pred_neural_init(const XCSF *xcsf, CL *c)
51 {
52  PRED_NEURAL *new = malloc(sizeof(PRED_NEURAL));
53  neural_init(xcsf, &new->net);
54  // hidden layers
55  uint32_t lopt = pred_neural_lopt(xcsf);
56  LAYER *l;
57  int i = 0;
58  int n_inputs = xcsf->x_dim;
59  while(i < MAX_LAYERS && xcsf->PRED_NUM_NEURONS[i] > 0) {
60  int hinit = xcsf->PRED_NUM_NEURONS[i];
61  int hmax = xcsf->PRED_MAX_NEURONS[i];
62  if(hmax < hinit || !xcsf->PRED_EVOLVE_NEURONS) {
63  hmax = hinit;
64  }
65  int f = xcsf->PRED_HIDDEN_ACTIVATION;
66  l = neural_layer_connected_init(xcsf, n_inputs, hinit, hmax, f, lopt);
67  neural_layer_insert(xcsf, &new->net, l, i);
68  n_inputs = hinit;
69  i++;
70  }
71  // output layer
72  int f = xcsf->PRED_OUTPUT_ACTIVATION;
73  lopt &= ~LAYER_EVOLVE_NEURONS; // never evolve the number of output neurons
74  l = neural_layer_connected_init(xcsf, n_inputs, xcsf->y_dim, xcsf->y_dim, f, lopt);
75  neural_layer_insert(xcsf, &new->net, l, i);
76  c->pred = new;
77 }

79 static uint32_t pred_neural_lopt(const XCSF *xcsf)
80 {
81  uint32_t lopt = 0;
82  if(xcsf->PRED_EVOLVE_ETA) {
83  lopt |= LAYER_EVOLVE_ETA;
84  }
85  if(xcsf->PRED_SGD_WEIGHTS) {
86  lopt |= LAYER_SGD_WEIGHTS;
87  }
88  if(xcsf->PRED_EVOLVE_WEIGHTS) {
89  lopt |= LAYER_EVOLVE_WEIGHTS;
90  }
91  if(xcsf->PRED_EVOLVE_NEURONS) {
92  lopt |= LAYER_EVOLVE_NEURONS;
93  }
94  if(xcsf->PRED_EVOLVE_FUNCTIONS) {
95  lopt |= LAYER_EVOLVE_FUNCTIONS;
96  }
97  return lopt;
98 }

208 void pred_neural_expand(const XCSF *xcsf, const CL *c)
209 {
210  PRED_NEURAL *pred = c->pred;
211  NET *net = &pred->net;
212  // insert a new layer
213  int size = net->head->layer->n_inputs;
214  int max = net->head->next->layer->max_outputs;
215  int pos = net->n_layers - 1;
216  int f = xcsf->PRED_HIDDEN_ACTIVATION;
217  uint32_t lopt = pred_neural_lopt(xcsf);
218  // initially same size as last hidden layer
219  LAYER *l = neural_layer_connected_init(xcsf, size, size, max, f, lopt);
220  neural_layer_insert(xcsf, net, l, pos);
221  // resize layers as necessary
222  const LAYER *prev = NULL;
223  for(const LLIST *iter = net->tail; iter != NULL; iter = iter->prev) {
224  if(prev != NULL && iter->layer->n_inputs != prev->n_outputs) {
225  layer_resize(xcsf, iter->layer, prev);
226  }
227  prev = iter->layer;
228  }
