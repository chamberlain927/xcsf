107 static int tree_grow(XCSF *xcsf, int *buffer, int p, int max, int depth)
108 {
109  int prim = irand_uniform(0,2);
110  if(p >= max) {
111  return (-1);
112  }
113  if(p == 0) {
114  prim = 1;
115  }
116  // add constant or external input
117  if(prim == 0 || depth == 0) {
118  prim = irand_uniform(GP_NUM_FUNC, GP_NUM_FUNC + xcsf->GP_NUM_CONS + xcsf->num_x_vars);
119  buffer[p] = prim;
120  return (p+1);
121  }
122  // add new function
123  else {
124  prim = irand_uniform(0,GP_NUM_FUNC);
125  switch(prim) {
126  case ADD:
127  case SUB:
128  case MUL:
129  case DIV:
130  buffer[p] = prim;
131  int one_child = tree_grow(xcsf, buffer, p+1, max, depth-1);
132  if(one_child < 0) {
133  return (-1);
134  }
135  return (tree_grow(xcsf, buffer, one_child, max, depth-1));
136  default:
137  printf("tree_grow() invalid function: %d\n", prim);
138  exit(EXIT_FAILURE);
139  }
140  }
141  printf("tree_grow() shouldn't be here\n");
142  exit(EXIT_FAILURE);
143 }

152 double tree_eval(XCSF *xcsf, GP_TREE *gp, double *x)
153 {
154  int node = gp->tree[(gp->p)++];
155  // external input
156  if(node >= GP_NUM_FUNC + xcsf->GP_NUM_CONS) {
157  return(x[node - GP_NUM_FUNC - xcsf->GP_NUM_CONS]);
158  }
159  // constant
160  else if(node >= GP_NUM_FUNC) {
161  return(xcsf->gp_cons[node-GP_NUM_FUNC]);
162  }
163  // function
164  switch(node) {
165  case ADD: return(tree_eval(xcsf,gp,x) + tree_eval(xcsf,gp,x));
166  case SUB: return(tree_eval(xcsf,gp,x) - tree_eval(xcsf,gp,x));
167  case MUL: return(tree_eval(xcsf,gp,x) * tree_eval(xcsf,gp,x));
168  case DIV: {
169  double num = tree_eval(xcsf,gp,x);
170  double den = tree_eval(xcsf,gp,x);
171  if(den == 0) {
172  return(num);
173  }
174  else {
175  return(num/den);
176  }
177  }
178  default:
179  printf("eval() invalid function: %d\n", node);
180  exit(EXIT_FAILURE);
181  }
182  printf("eval() shouldn't be here\n");
183  return 0;
184 }

247 void tree_crossover(XCSF *xcsf, GP_TREE *p1, GP_TREE *p2)
248 {
249  int len1 = p1->len;
250  int len2 = p2->len;
251  int start1 = irand_uniform(0,len1);
252  int end1 = tree_traverse(p1->tree, start1);
253  int start2 = irand_uniform(0,len2);
254  int end2 = tree_traverse(p2->tree, start2);
255  int nlen1 = start1+(end2-start2)+(len1-end1);
256  int *new1 = malloc(sizeof(int)*nlen1);
257  memcpy(&new1[0], &p1->tree[0], sizeof(int)*start1);
258  memcpy(&new1[start1], &p2->tree[start2], sizeof(int)*(end2-start2));
259  memcpy(&new1[start1+(end2-start2)], &p1->tree[end1], sizeof(int)*(len1-end1));
260  int nlen2 = start2+(end1-start1)+(len2-end2);
261  int *new2 = malloc(sizeof(int)*nlen2);
262  memcpy(&new2[0], &p2->tree[0], sizeof(int)*start2);
263  memcpy(&new2[start2], &p1->tree[start1], sizeof(int)*(end1-start1));
264  memcpy(&new2[start2+(end1-start1)], &p2->tree[end2], sizeof(int)*(len2-end2));
265  tree_free(xcsf, p1);
266  tree_free(xcsf, p2);
267  p1->tree = new1;
268  p2->tree = new2;
269  p1->len = tree_traverse(p1->tree, 0);
270  p2->len = tree_traverse(p2->tree, 0);
271 }

279 void tree_mutation(XCSF *xcsf, GP_TREE *offspring, double rate)
280 {
281  for(int i = 0; i < offspring->len; i++) {
282  if(rand_uniform(0,1) < rate) {
283  // terminals randomly replaced with other terminals
284  if(offspring->tree[i] >= GP_NUM_FUNC) {
285  offspring->tree[i] = irand_uniform(GP_NUM_FUNC,
286  GP_NUM_FUNC + xcsf->GP_NUM_CONS + xcsf->num_x_vars);
287  }
288  // functions randomly replaced with other functions
289  else {
290  offspring->tree[i] = irand_uniform(0, GP_NUM_FUNC);
291  }
292  }
293  }
294 }

303 {
