45 void cond_rectangle_init(const XCSF *xcsf, CL *c)
46 {

107 static double cond_rectangle_dist(const XCSF *xcsf, const CL *c, const double *x)
108 {
109  const COND_RECTANGLE *cond = c->cond;
110  double dist = 0;
111  for(int i = 0; i < xcsf->num_x_vars; i++) {
112  double d = fabs((x[i] - cond->center[i]) / cond->spread[i]);
113  if(d > dist) {
114  dist = d; // max distance
115  }
116  }
117  return dist;
118 }

120 _Bool cond_rectangle_crossover(const XCSF *xcsf, const CL *c1, const CL *c2)
121 {
122  const COND_RECTANGLE *cond1 = c1->cond;
123  const COND_RECTANGLE *cond2 = c2->cond;
124  _Bool changed = false;
125  if(rand_uniform(0,1) < xcsf->P_CROSSOVER) {
126  for(int i = 0; i < xcsf->num_x_vars; i++) {
127  if(rand_uniform(0,1) < 0.5) {
128  double tmp = cond1->center[i];
129  cond1->center[i] = cond2->center[i];
130  cond2->center[i] = tmp;
131  changed = true;
132  }
133  if(rand_uniform(0,1) < 0.5) {
134  double tmp = cond1->spread[i];
135  cond1->spread[i] = cond2->spread[i];
136  cond2->spread[i] = tmp;
137  changed = true;
138  }
139  }
140  }
141  return changed;
142 }

144 _Bool cond_rectangle_mutate(const XCSF *xcsf, const CL *c)
145 {
146  const COND_RECTANGLE *cond = c->cond;
147  _Bool changed = false;
148  for(int i = 0; i < xcsf->num_x_vars; i++) {
149  // centers
150  double orig = cond->center[i];
151  cond->center[i] += rand_normal(0, xcsf->S_MUTATION);
152  cond->center[i] = constrain(xcsf->COND_MIN, xcsf->COND_MAX, cond->center[i]);
153  if(orig != cond->center[i]) {
154  changed = true;
155  }
156  // spreads
157  orig = cond->spread[i];
158  cond->spread[i] += rand_normal(0, xcsf->S_MUTATION);
159  if(orig != cond->spread[i]) {
160  changed = true;
161  }
162  }
163  return changed;
164 }

166 _Bool cond_rectangle_general(const XCSF *xcsf, const CL *c1, const CL *c2)
167 {
168  // returns whether cond1 is more general than cond2
169  const COND_RECTANGLE *cond1 = c1->cond;
170  const COND_RECTANGLE *cond2 = c2->cond;
171  for(int i = 0; i < xcsf->num_x_vars; i++) {
172  double l1 = cond1->center[i] - cond1->spread[i];
173  double l2 = cond2->center[i] - cond2->spread[i];
174  double u1 = cond1->center[i] + cond1->spread[i];
175  double u2 = cond2->center[i] + cond2->spread[i];
176  if(l1 > l2 || u1 < u2) {
177  return false;
178  }
179  }
180  return true;
181 }
