68 #define NN 312
69 #define MM 156
70 #define MATRIX_A 0xB5026F5AA96619E9ULL
71 #define UM 0xFFFFFFFF80000000ULL /* Most significant 33 bits */
72 #define LM 0x7FFFFFFFULL /* Least significant 31 bits */

75 /* The array for the state vector */
76 static unsigned long long mt[NN];
77 /* mti==NN+1 means mt[NN] is not initialized */
78 static int mti=NN+1;

80 /* initializes mt[NN] with a seed */
81 void init_genrand64(unsigned long long seed)
82 {
83  mt[0] = seed;
84  for (mti=1; mti<NN; mti++)
85  mt[mti] = (6364136223846793005ULL * (mt[mti-1] ^ (mt[mti-1] >> 62)) + mti);
86 }

88 /* initialize by an array with array-length */
89 /* init_key is the array for initializing keys */
90 /* key_length is its length */
91 void init_by_array64(unsigned long long init_key[],
92  unsigned long long key_length)
93 {
94  unsigned long long i, j, k;
95  init_genrand64(19650218ULL);
96  i=1; j=0;
97  k = (NN>key_length ? NN : key_length);
98  for (; k; k--) {
99  mt[i] = (mt[i] ^ ((mt[i-1] ^ (mt[i-1] >> 62)) * 3935559000370003845ULL))
100  + init_key[j] + j; /* non linear */
101  i++; j++;
102  if (i>=NN) { mt[0] = mt[NN-1]; i=1; }
103  if (j>=key_length) j=0;
104  }
105  for (k=NN-1; k; k--) {
106  mt[i] = (mt[i] ^ ((mt[i-1] ^ (mt[i-1] >> 62)) * 2862933555777941757ULL))
107  - i; /* non linear */
108  i++;
109  if (i>=NN) { mt[0] = mt[NN-1]; i=1; }
110  }
111 
112  mt[0] = 1ULL << 63; /* MSB is 1; assuring non-zero initial array */
113 }

115 /* generates a random number on [0, 2^64-1]-interval */
116 unsigned long long genrand64_int64(void)
117 {
118  unsigned long long x;
119 
120  if (mti >= NN) { /* generate NN words at one time */
121 
122  /* if init_genrand64() has not been called, */
123  /* a default initial seed is used */
124  if (mti == NN+1) {
125  init_genrand64(5489ULL);
126  }
127 
128  int i;
129  static unsigned long long mag01[2]={0ULL, MATRIX_A};
130  for (i=0;i<NN-MM;i++) {
131  x = (mt[i]&UM)|(mt[i+1]&LM);
132  mt[i] = mt[i+MM] ^ (x>>1) ^ mag01[(int)(x&1ULL)];
133  }
134  for (;i<NN-1;i++) {
135  x = (mt[i]&UM)|(mt[i+1]&LM);
136  mt[i] = mt[i+(MM-NN)] ^ (x>>1) ^ mag01[(int)(x&1ULL)];
137  }
138  x = (mt[NN-1]&UM)|(mt[0]&LM);
139  mt[NN-1] = mt[MM-1] ^ (x>>1) ^ mag01[(int)(x&1ULL)];
140 
141  mti = 0;
142  }
143 
144  x = mt[mti++];
145 
146  x ^= (x >> 29) & 0x5555555555555555ULL;
147  x ^= (x << 17) & 0x71D67FFFEDA60000ULL;
148  x ^= (x << 37) & 0xFFF7EEE000000000ULL;
149  x ^= (x >> 43);
150 
151  return x;
152 }
