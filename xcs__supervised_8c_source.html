50 double xcs_supervised_fit(XCSF *xcsf, const INPUT *train_data,
51  const INPUT *test_data, _Bool shuffle)
52 {
53  double err = 0; // training error: total over all trials
54  double werr = 0; // training error: windowed total
55  double wterr = 0; // testing error: windowed total
56  double *pred = malloc(sizeof(double) * xcsf->y_dim);
57  for(int cnt = 0; cnt < xcsf->MAX_TRIALS; cnt++) {
58  // training sample
59  int row = xcs_supervised_sample(train_data, cnt, shuffle);
60  const double *x = &train_data->x[row * train_data->x_dim];
61  const double *y = &train_data->y[row * train_data->y_dim];
62  xcsf->train = true;
63  xcs_supervised_trial(xcsf, pred, x, y);
64  double error = (xcsf->loss_ptr)(xcsf, pred, y);
65  werr += error;
66  err += error;
67  // test sample
68  if(test_data != NULL) {
69  row = xcs_supervised_sample(test_data, cnt, shuffle);
70  x = &test_data->x[row * test_data->x_dim];
71  y = &test_data->y[row * test_data->y_dim];
72  xcsf->train = false;
73  xcs_supervised_trial(xcsf, pred, x, y);
74  wterr += (xcsf->loss_ptr)(xcsf, pred, y);
75  }
76  perf_print(xcsf, &werr, &wterr, cnt);
77  }
78  free(pred);
79  return err / xcsf->MAX_TRIALS;
80 }

103 double xcs_supervised_score(XCSF *xcsf, const INPUT *test_data)
104 {
105  xcsf->train = false;
106  double err = 0;
107  double *pred = malloc(sizeof(double) * xcsf->y_dim);
108  for(int row = 0; row < test_data->n_samples; row++) {
109  const double *x = &test_data->x[row * test_data->x_dim];
110  const double *y = &test_data->y[row * test_data->y_dim];
111  xcs_supervised_trial(xcsf, pred, x, y);
112  err += (xcsf->loss_ptr)(xcsf, pred, y);
113  }
114  free(pred);
115  return err / test_data->n_samples;

140 static void xcs_supervised_trial(XCSF *xcsf, double *pred, const double *x, const double *y)
141 {
142  clset_init(&xcsf->mset);
143  clset_init(&xcsf->kset);
144  clset_match(xcsf, x);
145  clset_pred(xcsf, &xcsf->mset, x, pred);
146  if(xcsf->train) {
147  clset_update(xcsf, &xcsf->mset, x, y, true);
148  ea(xcsf, &xcsf->mset);
149  }
150  clset_kill(xcsf, &xcsf->kset);
151  clset_free(&xcsf->mset);
