74 void neural_layer_softmax_forward(XCSF *xcsf, LAYER *l, double *input)
75 {
76  (void)xcsf;
77  double largest = input[0];
78  for(int i = 1; i < l->num_inputs; i++) {
79  if(input[i] > largest) {
80  largest = input[i];
81  }
82  }
83  double sum = 0;
84  for(int i = 0; i < l->num_inputs; i++) {
85  double e = exp(input[i]/l->temp - largest/l->temp);
86  sum += e;
87  l->output[i] = e;
88  }
89  for(int i = 0; i < l->num_inputs; i++) {
90  l->output[i] /= sum;
91  }
92 }
