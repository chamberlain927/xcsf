88 void env_maze_init(XCSF *xcsf, char *fname)
89 {
90  // open maze file
91  FILE *fp = fopen(fname, "rt");
92  if(fp == 0) {
93  printf("could not open %s. %s.\n", fname, strerror(errno));
94  exit(EXIT_FAILURE);
95  }
96  // read maze
97  ENV_MAZE *env = malloc(sizeof(ENV_MAZE));
98  int x = 0;
99  int y = 0;
100  int c;
101  while((c = fgetc(fp)) != EOF) {
102  if(c == '\n') {
103  y++;
104  env->xsize = x;
105  x = 0;
106  }
107  else {
108  env->maze[y][x] = (char)c;
109  x++;
110  }
111  // check maximum maze size not exceeded
112  if(x > MAX_SIZE || y > MAX_SIZE) {
113  printf("Maze too big to be read. Max size = [%d,%d]\n", MAX_SIZE, MAX_SIZE);
114  exit(EXIT_FAILURE);
115  }
116  }
117  // check if EOF came from an end-of-file or an error
118  if (ferror(fp)) {
119  printf("EOF read error: could not open %s. %s.\n", fname, strerror(errno));
120  exit(EXIT_FAILURE);
121  }
122  env->ysize = y;
123  env->state = malloc(sizeof(double) * 8);
124  xcsf->num_actions = 8;
125  xcsf->num_x_vars = 8;
126  xcsf->num_y_vars = 1;
127  xcsf->env = env;
128  fclose(fp);
129 }

146 void env_maze_reset(XCSF *xcsf)
147 {
148  ENV_MAZE *env = xcsf->env;
149  env->reset = false;
150  do {
151  env->xpos = irand_uniform(0, env->xsize);
152  env->ypos = irand_uniform(0, env->ysize);
153  } while(env->maze[env->ypos][env->xpos] != '*');
154 
155  if(MAZE_DEBUG && !xcsf->train) {

157  env_maze_print(xcsf);

177 double *env_maze_get_state(XCSF *xcsf)
178 {
179  ENV_MAZE *env = xcsf->env;
180  int spos = 0;
181  for(int y = -1; y < 2; y++) {
182  for(int x = -1; x < 2; x++) {
183  // ignore current pos
184  if(x == 0 && y == 0) {
185  continue;
186  }
187  // toroidal maze
188  int xsense = ((env->xpos + x) % env->xsize + env->xsize) % env->xsize;
189  int ysense = ((env->ypos + y) % env->ysize + env->ysize) % env->ysize;
190  char s = env->maze[ysense][xsense];
191  // convert sensor to real number
192  env->state[spos] = env_maze_sensor(xcsf, s);
193  spos++;
194  }
195  }
196  return env->state;
197 }

205 static double env_maze_sensor(XCSF *xcsf, char s)
206 {
207  (void)xcsf;
208  double ret = 0;
209  switch(s) {
210  case '*': ret = 0.1; break;
211  case 'O': ret = 0.3; break;
212  case 'G': ret = 0.5; break;
213  case 'F': ret = 0.7; break;
214  case 'Q': ret = 0.9; break;
215  default:
216  printf("unsupported maze state: %c\n", s);
217  exit(EXIT_FAILURE);
218  }
219  return ret;
220 }

228 double env_maze_execute(XCSF *xcsf, int action)
229 {
230  if(action < 0 || action > 7) {
231  printf("invalid maze action\n");
232  exit(EXIT_FAILURE);
233  }
234  ENV_MAZE *env = xcsf->env;
235  // toroidal maze
236  int newx = ((env->xpos + x_moves[action]) % env->xsize + env->xsize) % env->xsize;
237  int newy = ((env->ypos + y_moves[action]) % env->ysize + env->ysize) % env->ysize;
238  // make the move and recieve reward
239  double reward = 0;
240  switch(env->maze[newy][newx]) {
241  case 'O':
242  case 'Q':
243  break;
244  case '*':
245  env->ypos = newy;
246  env->xpos = newx;
247  break;
248  case 'F':
249  case 'G':
250  env->ypos = newy;
251  env->xpos = newx;
252  env->reset = true;
253  reward = MAX_PAYOFF;
254  break;
255  default:
256  printf("invalid maze type\n");
257  exit(EXIT_FAILURE);
258  }
259  if(MAZE_DEBUG && !xcsf->train) {
260  env_maze_print(xcsf);
261  }
262  return reward;
263 }
