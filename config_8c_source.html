457 static void config_get_ints(char *str, int *val)
458 {
459  int num = 0;

474 static void config_trim(char *s) {
475  const char *d = s;
476  do {
477  while(*d == ' ' || *d == '\t' || *d == '\n' || *d == '\r') {
478  d++;
479  }
480  } while((*s++ = *d++));
481 }

488 static void config_newnvpair(XCSF *xcsf, const char *param) {
489  // get length of name
490  size_t namelen = 0;
491  _Bool err = true;
492  for(namelen = 0; namelen < strnlen(param, MAXLEN); namelen++) {
493  if(param[namelen] == '=') {
494  err = false;
495  break;
496  }
497  }
498  if(err) {
499  return; // no '=' found
500  }
501  // get name
502  char *name = malloc(namelen+1);
503  for(size_t i = 0; i < namelen; i++) {
504  name[i] = param[i];
505  }
506  name[namelen] = '\0';
507  // get value
508  size_t valuelen = strnlen(param,MAXLEN)-namelen; // length of value
509  char *value = malloc(valuelen+1);
510  for(size_t i = 0; i < valuelen; i++) {
511  value[i] = param[namelen+1+i];
512  }
513  value[valuelen] = '\0';
514  // add
515  param_add(xcsf, name, value);
516  // clean up
517  free(name);
518  free(value);
519 }

526 static void config_process(XCSF *xcsf, const char *configline) {
527  // ignore empty lines
528  if(strnlen(configline, MAXLEN) == 0) {
529  return;
530  }
531  // lines starting with # are comments
532  if(configline[0] == '#') {
533  return;
534  }
535  // remove anything after #
536  char *ptr = strchr(configline, '#');
537  if(ptr != NULL) {
538  *ptr = '\0';
539  }
540  config_newnvpair(xcsf, configline);
541 }

548 static void config_read(XCSF *xcsf, const char *filename) {
549  FILE *f = fopen(filename, "rt");
550  if(f == NULL) {
551  printf("Warning: could not open %s.\n", filename);
552  return;
553  }
554  char buff[MAXLEN];
555  while(!feof(f)) {
556  if(fgets(buff, MAXLEN-2, f) == NULL) {
557  break;
558  }
559  config_trim(buff);
560  config_process(xcsf, buff);
561  }
562  fclose(f);
563 }
