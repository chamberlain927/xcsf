285 static void config_trim(char *s) {
286  const char *d = s;
287  do {

289  d++;

291  } while((*s++ = *d++));

298 static void config_newnvpair(const char *config) {
299  // first pair
300  if(head == NULL) {
301  head = malloc(sizeof(PARAM_LIST));
302  head->next = NULL;
303  }
304  // other pairs
305  else {
306  PARAM_LIST *new = malloc(sizeof(PARAM_LIST));
307  new->next = head;
308  head = new;
309  }
310  // get length of name
311  size_t namelen = 0; // length of name
312  _Bool err = true;
313  for(namelen = 0; namelen < strnlen(config, MAXLEN); namelen++) {
314  if(config[namelen] == '=') {
315  err = false;
316  break;
317  }
318  }
319  // no = found
320  if(err) {
321  printf("error reading config: no '=' found\n");
322  exit(EXIT_FAILURE);
323  }
324  // get name
325  char *name = malloc(namelen+1);
326  for(size_t i = 0; i < namelen; i++) {
327  name[i] = config[i];
328  }
329  name[namelen] = '\0';
330  // get value
331  size_t valuelen = strnlen(config,MAXLEN)-namelen; // length of value
332  char *value = malloc(valuelen+1);
333  for(size_t i = 0; i < valuelen; i++) {
334  value[i] = config[namelen+1+i];
335  }
336  value[valuelen] = '\0';
337  // add pair
338  head->name = name;
339  head->value = value;
340 }

347 static char *config_getvalue(const char *name) {
348  char *result = NULL;
349  for(const PARAM_LIST *iter = head; iter != NULL; iter = iter->next) {
350  if(strcmp(name, iter->name) == 0) {
351  result = iter->value;
352  break;
353  }
354  }
355  return result;
356 }

362 static void config_process(const char *configline) {
363  // ignore empty lines
364  if(strnlen(configline, MAXLEN) == 0) {
365  return;
366  }
367  // lines starting with # are comments
368  if(configline[0] == '#') {
369  return;
370  }
371  // remove anything after #
372  char *ptr = strchr(configline, '#');
373  if(ptr != NULL) {
374  *ptr = '\0';
375  }
376  config_newnvpair(configline);
377 }

383 static void config_read(const char *filename) {
384  FILE *f = fopen(filename, "rt");
385  if(f == NULL) {
386  printf("ERROR: cannot open %s\n", filename);
387  return;
388  }
389  char buff[MAXLEN];
390  head = NULL;
391  while(!feof(f)) {
392  if(fgets(buff, MAXLEN-2, f) == NULL) {
393  break;
394  }
395  config_trim(buff);
396  config_process(buff);
397  }
398  fclose(f);
399 }

404 static void config_tidyup()
405 {
406  PARAM_LIST *iter = head;
407  while(iter != NULL) {
408  free(head->value);
409  free(head->name);
410  head = iter->next;
411  free(iter);
412  iter = head;
413  }
414  head = NULL;
415 }
