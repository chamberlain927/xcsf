294 static int config_get_ints(const char *name, int *value)
295 {
296  int num = 0;
297  char *end;
298  char *saveptr;
299  char *val = config_getvalue(name);
300  const char *ptok = strtok_r(val, ARRAY_DELIM, &saveptr);
301  while(ptok != NULL) {
302  value[num] = strtoimax(ptok, &end, BASE);
303  ptok = strtok_r(NULL, ARRAY_DELIM, &saveptr);
304  num++;
305  }
306  return num;
307 }

326 static void config_newnvpair(const char *config) {
327  // first pair
328  if(head == NULL) {
329  head = malloc(sizeof(PARAM_LIST));
330  head->next = NULL;
331  }
332  // other pairs
333  else {
334  PARAM_LIST *new = malloc(sizeof(PARAM_LIST));
335  new->next = head;
336  head = new;
337  }
338  // get length of name
339  size_t namelen = 0; // length of name
340  _Bool err = true;
341  for(namelen = 0; namelen < strnlen(config, MAXLEN); namelen++) {
342  if(config[namelen] == '=') {
343  err = false;
344  break;
345  }
346  }
347  // no = found
348  if(err) {
349  printf("error reading config: no '=' found\n");
350  exit(EXIT_FAILURE);
351  }
352  // get name
353  char *name = malloc(namelen+1);
354  for(size_t i = 0; i < namelen; i++) {
355  name[i] = config[i];
356  }
357  name[namelen] = '\0';
358  // get value
359  size_t valuelen = strnlen(config,MAXLEN)-namelen; // length of value
360  char *value = malloc(valuelen+1);
361  for(size_t i = 0; i < valuelen; i++) {
362  value[i] = config[namelen+1+i];
363  }
364  value[valuelen] = '\0';
365  // add pair
366  head->name = name;
367  head->value = value;
368 }

375 static char *config_getvalue(const char *name) {
376  char *result = NULL;
377  for(const PARAM_LIST *iter = head; iter != NULL; iter = iter->next) {
378  if(strcmp(name, iter->name) == 0) {
379  result = iter->value;
380  break;
381  }
382  }
383  return result;
384 }

390 static void config_process(const char *configline) {
391  // ignore empty lines
392  if(strnlen(configline, MAXLEN) == 0) {
393  return;
394  }
395  // lines starting with # are comments
396  if(configline[0] == '#') {
397  return;
398  }
399  // remove anything after #
400  char *ptr = strchr(configline, '#');
401  if(ptr != NULL) {
402  *ptr = '\0';
403  }
404  config_newnvpair(configline);
405 }

411 static void config_read(const char *filename) {
412  FILE *f = fopen(filename, "rt");
413  if(f == NULL) {
414  printf("ERROR: cannot open %s\n", filename);
415  return;
416  }
417  char buff[MAXLEN];
418  head = NULL;
419  while(!feof(f)) {
420  if(fgets(buff, MAXLEN-2, f) == NULL) {
421  break;
422  }
423  config_trim(buff);
424  config_process(buff);
425  }
426  fclose(f);
427 }

439  free(iter);
440  iter = head;
441  }
442  head = NULL;
443 }
