195 void trim(char *s)
196 {
197  size_t i = 0;
198  while(s[i]==' ' || s[i]=='\t' || s[i] =='\n' || s[i]=='\r') {
199  i++;
200  }
201  if(i > 0) {
202  size_t j = 0;
203  while(j < strnlen(s, MAXLEN)) {
204  s[j] = s[j+i];
205  j++;
206  }
207  s[j] = '\0';
208  }
209  i = strnlen(s, MAXLEN)-1;
210  while(s[i]==' ' || s[i]=='\t' || s[i] =='\n' || s[i]=='\r') {
211  i--;
212  }
213  if(i < (strnlen(s, MAXLEN)-1)) {
214  s[i+1] = '\0';
215  }
216 }

222 void newnvpair(const char *config) {
223  // first pair
224  if(head == NULL) {
225  head = malloc(sizeof(PARAM_LIST));
226  head->next = NULL;
227  }
228  // other pairs
229  else {
230  PARAM_LIST *new = malloc(sizeof(PARAM_LIST));
231  new->next = head;
232  head = new;
233  }
234  // get length of name
235  size_t namelen = 0; // length of name
236  _Bool err = true;
237  for(namelen = 0; namelen < strnlen(config, MAXLEN); namelen++) {
238  if(config[namelen] == '=') {
239  err = false;
240  break;
241  }
242  }
243  // no = found
244  if(err) {
245  printf("error reading config: no '=' found\n");
246  exit(EXIT_FAILURE);
247  }
248  // get name
249  char *name = malloc(namelen+1);
250  snprintf(name, namelen+1, "%s", config);
251  // get value
252  size_t valuelen = strnlen(config,MAXLEN)-namelen; // length of value
253  char *value = malloc(valuelen);
254  snprintf(value, valuelen, "%s", config+namelen+1);
255  // add pair
256  head->name = name;
257  head->value = value;
258 }

280 void process(char *configline) {
281  // ignore empty lines
282  if(strnlen(configline, MAXLEN) == 0) {
283  return;
284  }
285  // lines starting with # are comments
286  if(configline[0] == '#') {
287  return;
288  }
289  // remove anything after #
290  char *ptr = strchr(configline, '#');
291  if(ptr != NULL) {
292  *ptr = '\0';
293  }
294  newnvpair(configline);
295 }

301 void init_config(const char *filename) {
302  FILE *f = fopen(filename, "rt");
303  if(f == NULL) {
304  printf("ERROR: cannot open %s\n", filename);
305  return;
306  }
307  char buff[MAXLEN];
308  head = NULL;
309  while(!feof(f)) {
310  if(fgets(buff, MAXLEN-2, f) == NULL) {
311  break;
312  }
313  trim(buff);
314  process(buff);
315  }
316  fclose(f);
317 }

322 void tidyup()

326  free(head->value);
