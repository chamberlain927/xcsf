68 double xcsf_fit2(XCSF *xcsf, const INPUT *train_data, const INPUT *test_data, _Bool shuffle)
69 {
70  gplot_init(xcsf);
71  double perr = 0;
72  double err = 0;
73  double pterr = 0;
74  double *pred = malloc(sizeof(double) * xcsf->num_y_vars);
75  for(int cnt = 0; cnt < xcsf->MAX_TRIALS; cnt++) {
76  // select training sample
77  int row = 0;
78  if(shuffle) {
79  row = irand_uniform(0, train_data->rows);
80  }
81  else {
82  row = (cnt % train_data->rows + train_data->rows) % train_data->rows;
83  }
84  const double *x = &train_data->x[row * train_data->x_cols];
85  const double *y = &train_data->y[row * train_data->y_cols];
86  // execute a learning trial
87  xcsf->train = true;
88  double error = xcsf_learn_trial(xcsf, pred, x, y);
89  perr += error;
90  err += error;
91  // if some test data has been supplied
92  if(test_data != NULL) {
93  // select test sample
94  if(shuffle) {
95  row = irand_uniform(0, test_data->rows);
96  }
97  else {
98  row = (cnt % test_data->rows + test_data->rows) % test_data->rows;
99  }
100  x = &test_data->x[row * test_data->x_cols];
101  y = &test_data->y[row * test_data->y_cols];
102  // execute a test trial
103  xcsf->train = false;
104  pterr += xcsf_test_trial(xcsf, pred, x, y);
105  // display performance as necessary (training and testing)
106  if(cnt % xcsf->PERF_AVG_TRIALS == 0 && cnt > 0) {
107  disp_perf2(xcsf, perr/xcsf->PERF_AVG_TRIALS, pterr/xcsf->PERF_AVG_TRIALS, cnt);
108  perr = 0; pterr = 0;
109  }
110  }
111  // display performance as necessary (training only)
112  else if(cnt % xcsf->PERF_AVG_TRIALS == 0 && cnt > 0) {
113  disp_perf1(xcsf, perr/xcsf->PERF_AVG_TRIALS, cnt);
114  perr = 0;
115  }
116  }
117  // clean up
118  free(pred);
119  gplot_free(xcsf);
120  return err/xcsf->MAX_TRIALS;
121 }

131 static double xcsf_learn_trial(XCSF *xcsf, double *pred, const double *x, const double *y)
132 {
133  SET mset; // match set
134  SET kset; // kill set
135  clset_init(xcsf, &mset);
136  clset_init(xcsf, &kset);
137  clset_match(xcsf, &mset, &kset, x);
138  clset_pred(xcsf, &mset, x, pred);
139  clset_update(xcsf, &mset, &kset, x, y, true);
140  ea(xcsf, &mset, &kset);
141  xcsf->time += 1;
142  xcsf->msetsize += (mset.size - xcsf->msetsize) * xcsf->BETA;
143  clset_kill(xcsf, &kset); // kills deleted classifiers
144  clset_free(xcsf, &mset); // frees the match set list
145  return (xcsf->loss_ptr)(xcsf, pred, y);
146 }

156 static double xcsf_test_trial(XCSF *xcsf, double *pred, const double *x, const double *y)
157 {
158  SET mset; // match set
159  SET kset; // kill set
160  clset_init(xcsf, &mset);
161  clset_init(xcsf, &kset);

163  clset_pred(xcsf, &mset, x, pred);
164  xcsf->msetsize += (mset.size - xcsf->msetsize) * xcsf->BETA;
165  clset_kill(xcsf, &kset); // kills deleted classifiers
166  clset_free(xcsf, &mset); // frees the match set list
167  return (xcsf->loss_ptr)(xcsf, pred, y);

177 void xcsf_predict(XCSF *xcsf, const double *x, double *pred, int rows)
178 {
179  xcsf->train = false;
180  for(int row = 0; row < rows; row++) {
181  SET mset; // match set
182  SET kset; // kill set
183  clset_init(xcsf, &mset);
184  clset_init(xcsf, &kset);
185  clset_match(xcsf, &mset, &kset, &x[row * xcsf->num_x_vars]);
186  clset_pred(xcsf, &mset, &x[row * xcsf->num_x_vars], &pred[row * xcsf->num_y_vars]);
187  clset_kill(xcsf, &kset); // kills deleted classifiers
188  clset_free(xcsf, &mset); // frees the match set list
189  }
190 }

198 double xcsf_score(XCSF *xcsf, const INPUT *test_data)
199 {
200  xcsf->train = false;
201  double err = 0;
202  double *pred = malloc(sizeof(double) * xcsf->num_y_vars);
203  for(int row = 0; row < test_data->rows; row++) {
204  const double *x = &test_data->x[row * test_data->x_cols];
205  const double *y = &test_data->y[row * test_data->y_cols];
206  err += xcsf_test_trial(xcsf, pred, x, y);
207  }
208  free(pred);
209  return err/(double)test_data->rows;
210 }

271 size_t xcsf_load(XCSF *xcsf, const char *fname)
272 {
273  if(xcsf->pset.size > 0) {
274  clset_kill(xcsf, &xcsf->pset);
275  clset_init(xcsf, &xcsf->pset);
276  }
277  FILE *fp = fopen(fname, "rb");
278  if(fp == 0) {
279  printf("Error opening load file: %s. %s.\n", fname, strerror(errno));
280  exit(EXIT_FAILURE);
281  }
282  size_t s = 0;
283  double version = 0;
284  s += fread(&version, sizeof(double), 1, fp);
285  if(version != VERSION) {
286  printf("Error loading file: %s. Version mismatch. ", fname);
287  printf("This version: %f.\nLoaded version: %f", VERSION, version);
288  exit(EXIT_FAILURE);
289  }
290  s += xcsf_load_params(xcsf, fp);
291  s += clset_pop_load(xcsf, fp);
292  fclose(fp);
293  return s;
294 }
