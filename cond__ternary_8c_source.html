63 static void cond_ternary_rand(const XCSF *xcsf, const CL *c)
64 {
65  (void)xcsf;
66  const COND_TERNARY *cond = c->cond;
67  for(int i = 0; i < cond->len; i++) {
68  if(rand_uniform(0,1) < P_DONTCARE) {
69  cond->string[i] = DONT_CARE;
70  }
71  else {
72  if(rand_uniform(0,1) < 0.5) {
73  cond->string[i] = '0';
74  }
75  else {
76  cond->string[i] = '1';
77  }
78  }
79  }
80 }

102 void cond_ternary_cover(const XCSF *xcsf, const CL *c, const double *x)
103 {
104  const COND_TERNARY *cond = c->cond;
105  char state[xcsf->COND_BITS];
106  for(int i = 0; i < xcsf->x_dim; i++) {
107  float_to_binary(x[i], state, xcsf->COND_BITS);
108  for(int b = 0; b < xcsf->COND_BITS; b++) {
109  if(rand_uniform(0,1) < P_DONTCARE) {
110  cond->string[i*xcsf->COND_BITS+b] = DONT_CARE;
111  }
112  else {
113  cond->string[i*xcsf->COND_BITS+b] = state[b];
114  }
115  }
116  }
117 }

140 _Bool cond_ternary_crossover(const XCSF *xcsf, const CL *c1, const CL *c2)
141 {
142  const COND_TERNARY *cond1 = c1->cond;
143  const COND_TERNARY *cond2 = c2->cond;
144  _Bool changed = false;
145  if(rand_uniform(0,1) < xcsf->P_CROSSOVER) {
146  for(int i = 0; i < cond1->len; i++) {
147  if(rand_uniform(0,1) < 0.5) {
148  double tmp = cond1->string[i];
149  cond1->string[i] = cond2->string[i];
150  cond2->string[i] = tmp;
151  changed = true;
152  }
153  }
154  }
155  return changed;
156 }

158 _Bool cond_ternary_mutate(const XCSF *xcsf, const CL *c)
159 {
160  COND_TERNARY *cond = c->cond;
161  sam_adapt(xcsf, cond->mu, N_MU);
162  _Bool changed = false;
163  for(int i = 0; i < cond->len; i++) {
164  if(rand_uniform(0,1) < cond->mu[0]) {
165  if(cond->string[i] == DONT_CARE) {
166  cond->string[i] = irand_uniform(0,2) + '0';
167  }
168  else {
169  cond->string[i] = DONT_CARE;
170  }
171  changed = true;
172  }
173  }
174  return changed;
