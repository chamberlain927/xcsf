56 static void cond_ternary_rand(XCSF *xcsf, CL *c)
57 {
58  COND_TERNARY *cond = c->cond;
59  for(int i = 0; i < xcsf->num_x_vars * xcsf->COND_BITS; i++) {
60  if(rand_uniform(0,1) < P_DONTCARE) {
61  cond->string[i] = DONT_CARE;
62  }
63  else {
64  if(rand_uniform(0,1) < 0.5) {
65  cond->string[i] = '0';
66  }
67  else {
68  cond->string[i] = '1';
69  }
70  }
71  }
72 }

91 void cond_ternary_cover(XCSF *xcsf, CL *c, double *x)
92 {
93  COND_TERNARY *cond = c->cond;
94  char state[xcsf->COND_BITS];
95  for(int i = 0; i < xcsf->num_x_vars; i++) {
96  float_to_binary(x[i], state, xcsf->COND_BITS);
97  for(int b = 0; b < xcsf->COND_BITS; b++) {
98  if(rand_uniform(0,1) < P_DONTCARE) {
99  cond->string[i*xcsf->COND_BITS+b] = DONT_CARE;
100  }
101  else {
102  cond->string[i*xcsf->COND_BITS+b] = state[b];
103  }
104  }
105  }
106 }

113 _Bool cond_ternary_match(XCSF *xcsf, CL *c, double *x)
114 {
115  COND_TERNARY *cond = c->cond;
116  char state[xcsf->COND_BITS];
117  for(int i = 0; i < xcsf->num_x_vars; i++) {
118  float_to_binary(x[i], state, xcsf->COND_BITS);
119  for(int b = 0; b < xcsf->COND_BITS; b++) {
120  char s = cond->string[i*xcsf->COND_BITS+b];
121  if(s != DONT_CARE && s != state[b]) {
122  c->m = false;
123  return false;
124  }
125  }
126  }
127  c->m = true;
128  return true;

131 _Bool cond_ternary_crossover(XCSF *xcsf, CL *c1, CL *c2)
132 {
133  COND_TERNARY *cond1 = c1->cond;
134  COND_TERNARY *cond2 = c2->cond;
135  _Bool changed = false;
136  if(rand_uniform(0,1) < xcsf->P_CROSSOVER) {
137  for(int i = 0; i < xcsf->num_x_vars * xcsf->COND_BITS; i++) {
138  if(rand_uniform(0,1) < 0.5) {
139  double tmp = cond1->string[i];
140  cond1->string[i] = cond2->string[i];
141  cond2->string[i] = tmp;
142  changed = true;
143  }
144  }
145  }
146  return changed;
147 }

172 _Bool cond_ternary_general(XCSF *xcsf, CL *c1, CL *c2)
173 {
174  // returns whether cond1 is more general than cond2
175  COND_TERNARY *cond1 = c1->cond;
176  COND_TERNARY *cond2 = c2->cond;
177  _Bool general = false;
178  for(int i = 0; i < xcsf->num_x_vars * xcsf->COND_BITS; i++) {
179  if(cond1->string[i] != DONT_CARE && cond1->string[i] != cond2->string[i]) {
180  return false;
181  }
182  else if(cond1->string[i] != cond2->string[i]) {
183  general = true;
184  }
185  }
186  return general;
187 }
