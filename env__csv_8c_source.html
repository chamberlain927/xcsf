108 static void env_csv_read(const char *fname, double **data, int *n_samples, int *n_dim)
109 {
110  FILE *fin = fopen(fname, "rt");
111  if(fin == 0) {
112  printf("Error opening file: %s. %s.\n", fname, strerror(errno));
113  exit(EXIT_FAILURE);
114  }
115  *n_samples = env_csv_samples(fin);
116  *n_dim = env_csv_dim(fin);
117  if(*n_samples > 0 && *n_dim > 0) {
118  env_csv_read_data(fin, data, *n_samples, *n_dim);
119  fclose(fin);
120  }
121  else {
122  printf("Error reading file: %s. No samples found\n", fname);
123  fclose(fin);
124  exit(EXIT_FAILURE);
125  }
126  printf("Loaded: %s: %d samples, %d dimensions\n", fname, *n_samples, *n_dim);
127 }

149 static int env_csv_dim(FILE *fin)
150 {
151  rewind(fin);
152  int n_dim = 0;
153  char line[MAX_COLS];
154  char *saveptr;
155  if(fgets(line, MAX_COLS, fin) != NULL) {
156  const char *ptok = strtok_r(line, DELIM, &saveptr);
157  while(ptok != NULL) {
158  if(strnlen(ptok,MAX_COLS) > 0) {
159  n_dim++;
160  }
161  ptok = strtok_r(NULL, DELIM, &saveptr);
162  }
163  }
164  return n_dim;
165 }

174 static void env_csv_read_data(FILE *fin, double **data, int n_samples, int n_dim)
175 {
176  rewind(fin);
177  *data = malloc(sizeof(double) * n_dim * n_samples);
178  char line[MAX_COLS];
179  char *saveptr;
180  int i = 0;
181  while(fgets(line, MAX_COLS, fin) != NULL && i < n_samples) {
182  (*data)[i * n_dim] = atof(strtok_r(line, DELIM, &saveptr));
183  for(int j = 1; j < n_dim; j++) {
184  (*data)[i * n_dim + j] = atof(strtok_r(NULL, DELIM, &saveptr));
185  }
186  i++;
187  }
188 }
