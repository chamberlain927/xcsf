106 static void env_csv_read(char *fname, double **data, int *num_rows, int *num_cols)
107 {
108  FILE *fin = fopen(fname, "rt");
109  if(fin == 0) {
110  printf("Error opening file: %s. %s.\n", fname, strerror(errno));
111  exit(EXIT_FAILURE);
112  }
113  // ascertain the file length and number of variables per line
114  *num_rows = 0;
115  *num_cols = 0;
116  char line[MAX_COLS];
117  char *saveptr;
118  while(fgets(line, MAX_COLS, fin) != NULL) {
119  if(*num_rows > MAX_ROWS) {
120  printf("data file %s is too big; maximum: %d\n", fname, MAX_ROWS);
121  exit(EXIT_FAILURE);
122  }
123  // use the first line to count the number of variables on a line
124  if(*num_rows == 0) {
125  char *ptok = strtok_r(line, DELIM, &saveptr);
126  while(ptok != NULL) {
127  if(strnlen(ptok,MAX_COLS) > 0) {
128  (*num_cols)++;
129  }
130  ptok = strtok_r(NULL, DELIM, &saveptr);
131  }
132  }
133  (*num_rows)++; // count the number of lines
134  }
135  // read data file to memory
136  rewind(fin);
137  *data = malloc(sizeof(double) * (*num_cols) * (*num_rows));
138  for(int i = 0; fgets(line,MAX_COLS,fin) != NULL; i++) {
139  (*data)[i * (*num_cols)] = atof(strtok_r(line, DELIM, &saveptr));
140  for(int j = 1; j < *num_cols; j++) {
141  (*data)[i * (*num_cols)+j] = atof(strtok_r(NULL, DELIM, &saveptr));
142  }
143  }
144  fclose(fin);
145  printf("Loaded: %s: %d rows, %d cols\n", fname, *num_rows, *num_cols);
146 }
