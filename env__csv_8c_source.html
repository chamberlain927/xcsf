71 void env_csv_read(char *fname, double **data, int *num_rows, int *num_cols)
72 {
73  FILE *fin = fopen(fname, "rt");
74  if(fin == 0) {
75  printf("Error opening file: %s. %s.\n", fname, strerror(errno));
76  exit(EXIT_FAILURE);
77  }
78  // ascertain the file length and number of variables per line
79  *num_rows = 0;
80  *num_cols = 0;
81  char line[MAX_COLS];
82  char *saveptr;
83  while(fgets(line, MAX_COLS, fin) != NULL) {
84  if(*num_rows > MAX_ROWS) {
85  printf("data file %s is too big; maximum: %d\n", fname, MAX_ROWS);
86  exit(EXIT_FAILURE);
87  }
88  // use the first line to count the number of variables on a line
89  if(*num_rows == 0) {
90  char *ptok = strtok_r(line, DELIM, &saveptr);
91  while(ptok != NULL) {
92  if(strnlen(ptok,MAX_COLS) > 0) {
93  (*num_cols)++;
94  }
95  ptok = strtok_r(NULL, DELIM, &saveptr);
96  }
97  }
98  (*num_rows)++; // count the number of lines
99  }
100  // read data file to memory
101  rewind(fin);
102  *data = malloc(sizeof(double) * (*num_cols) * (*num_rows));
103  for(int i = 0; fgets(line,MAX_COLS,fin) != NULL; i++) {
104  (*data)[i * (*num_cols)] = atof(strtok_r(line, DELIM, &saveptr));
105  for(int j = 1; j < *num_cols; j++) {
106  (*data)[i * (*num_cols)+j] = atof(strtok_r(NULL, DELIM, &saveptr));
107  }
108  }
109  fclose(fin);
110  printf("Loaded: %s: %d rows, %d cols\n", fname, *num_rows, *num_cols);
111 }
