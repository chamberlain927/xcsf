180 void rule_dgp_act_cover(const XCSF *xcsf, CL *c, const double *x, int action)
181 {
182  do {
183  rule_dgp_cond_rand(xcsf, c);
184  } while(!rule_dgp_cond_match(xcsf, c, x)

188 int rule_dgp_act_compute(const XCSF *xcsf, CL *c, const double *x)
189 {
190  (void)x; // graph already updated
191  RULE_DGP *cond = c->cond;
192  c->action = 0;
193  for(int i = 0; i < cond->num_outputs; i++) {
194  if(graph_output(xcsf, &cond->dgp, i+1) > 0.5) {
195  c->action += pow(2,i);
196  }
197  }
198  c->action = iconstrain(0, xcsf->num_actions-1, c->action);
199  return c->action;
200 }
