178 void rule_dgp_act_cover(const XCSF *xcsf, const CL *c, const double *x, int action)
179 {
180  do {
181  rule_dgp_cond_rand(xcsf, c);
182  } while(!rule_dgp_cond_match(xcsf, c, x)
183  && rule_dgp_act_compute(xcsf, c, x) != action);
184 }

186 int rule_dgp_act_compute(const XCSF *xcsf, const CL *c, const double *x)
187 {
188  (void)x; // graph already updated
189  const RULE_DGP *cond = c->cond;
190  int action = 0;
191  for(int i = 0; i < cond->n_outputs; i++) {
192  if(graph_output(xcsf, &cond->dgp, i+1) > 0.5) {
193  action += pow(2,i);
194  }
195  }
196  action = iconstrain(0, xcsf->n_actions-1, action);
197  return action;
198 }
