56 void env_mux_init(XCSF *xcsf, int bits)
57 {
58  ENV_MUX *env = malloc(sizeof(ENV_MUX));
59  env->pos_bits = 1;
60  while(env->pos_bits + pow(2, env->pos_bits) <= bits) {
61  (env->pos_bits)++;
62  }
63  (env->pos_bits)--;
64  int n = env->pos_bits + pow(2, env->pos_bits);
65  env->state = malloc(sizeof(double) * n);
66  xcsf->num_actions = 2;
67  xcsf->num_x_vars = n;
68  xcsf->num_y_vars = 1;
69  xcsf->env = env;
70 }

103 double env_mux_execute(XCSF *xcsf, int action)
104 {
105  ENV_MUX *env = xcsf->env;
106  int pos = env->pos_bits;
107  for(int i = 0; i < env->pos_bits; i++) {
108  if(env->state[i] > 0.5) {
109  pos += pow(2, (double)(env->pos_bits - 1 - i));
110  }
111  }
112  int answer = (env->state[pos] > 0.5) ? 1 : 0;
113  if(action == answer) {
114  return MAX_PAYOFF;
115  }
116  else {
117  return 0;
118  }
119 }
