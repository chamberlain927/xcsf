57 void env_mux_init(XCSF *xcsf, int bits)
58 {
59  ENV_MUX *env = malloc(sizeof(ENV_MUX));
60  env->pos_bits = 1;
61  while(env->pos_bits + pow(2, env->pos_bits) <= bits) {
62  (env->pos_bits)++;
63  }
64  (env->pos_bits)--;
65  int n = env->pos_bits + pow(2, env->pos_bits);
66  env->state = malloc(sizeof(double) * n);
67  xcsf->n_actions = 2;

69  xcsf->y_dim = 1;

104 double env_mux_execute(const XCSF *xcsf, int action)
105 {
106  const ENV_MUX *env = xcsf->env;
107  int pos = env->pos_bits;
108  for(int i = 0; i < env->pos_bits; i++) {
109  if(env->state[i] > 0.5) {
110  pos += pow(2, (double)(env->pos_bits - 1 - i));
111  }
112  }
113  int answer = (env->state[pos] > 0.5) ? 1 : 0;
114  if(action == answer) {
115  return MAX_PAYOFF;
116  }
117  else {
118  return 0;
119  }
120 }
