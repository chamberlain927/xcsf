92 void pop_del(XCSF *xcsf, SET *kset)
93 {
94  double avg_fit = set_total_fit(xcsf, &xcsf->pset) / xcsf->pset.num;
95  double total = 0;
96  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
97  total += cl_del_vote(xcsf, iter->cl, avg_fit);
98  }
99  CLIST *del = NULL;
100  CLIST *delprev = NULL;
101  int delsize = 0;
102  for(int i = 0; i < 2; i++) {
103  double p = rand_uniform(0,total);
104  double sum = 0;
105  CLIST *prev = NULL;
106  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
107  sum += cl_del_vote(xcsf, iter->cl, avg_fit);
108  if(sum > p) {
109  int size = cl_cond_size(xcsf, iter->cl) + cl_pred_size(xcsf, iter->cl);
110  if(del == NULL || size > delsize) {
111  del = iter;
112  delprev = prev;
113  delsize = size;
114  }
115  break;
116  }
117  prev = iter;
118  }
119  }
120  // decrement numerosity
121  (del->cl->num)--;
122  (xcsf->pset.num)--;
123  // macro classifier must be deleted
124  if(del->cl->num == 0) {
125  set_add(xcsf, kset, del->cl);
126  (xcsf->pset.size)--;
127  if(delprev == NULL) {
128  xcsf->pset.list = del->next;
129  }
130  else {
131  delprev->next = del->next;
132  }
133  free(del);
134  }
135 }

160 void set_match(XCSF *xcsf, SET *mset, SET *kset, double *x)
161 {
162  _Bool *act_covered = calloc(xcsf->num_actions, sizeof(_Bool));
163 #ifdef PARALLEL_MATCH
164  // prepare for parallel processing of matching conditions
165  CLIST *blist[xcsf->pset.size];
166  int j = 0;
167  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
168  blist[j] = iter;
169  j++;
170  }
171  // update current matching conditions setting m flags in parallel
172 #pragma omp parallel for
173  for(int i = 0; i < xcsf->pset.size; i++) {
174  cl_match(xcsf, blist[i]->cl, x);
175  }
176  // build match set list in series
177  for(int i = 0; i < xcsf->pset.size; i++) {
178  if(cl_m(xcsf, blist[i]->cl)) {
179  set_add(xcsf, mset, blist[i]->cl);
180  int action = cl_action(xcsf, blist[i]->cl, x);
181  act_covered[action] = true;
182  }
183  }
184 #else
185  // update matching conditions and build match set list in series
186  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
187  if(cl_match(xcsf, iter->cl, x)) {
188  set_add(xcsf, mset, iter->cl);
189  int action = cl_action(xcsf, iter->cl, x);
190  act_covered[action] = true;
191  }
192  }
193 #endif
194  // perform covering if all actions are not represented
195  set_cover(xcsf, mset, kset, x, act_covered);
196  free(act_covered);
197 }

207 void set_cover(XCSF *xcsf, SET *mset, SET *kset, double *x, _Bool *act_covered)
208 {
209  int attempts = 0;
210  _Bool again;
211  do {
212  again = false;
213  for(int i = 0; i < xcsf->num_actions; i++) {
214  if(!act_covered[i]) {
215  // new classifier with matching condition and action
216  CL *new = malloc(sizeof(CL));
217  cl_init(xcsf, new, (mset->num)+1, xcsf->time);
218  cl_cover(xcsf, new, x, i);
219  set_add(xcsf, &xcsf->pset, new);
220  set_add(xcsf, mset, new);
221  act_covered[i] = true;
222  }
223  }
224  // enforce population size
225  int prev_psize = xcsf->pset.size;
226  pop_enforce_limit(xcsf, kset);
227  // if a macro classifier was deleted, remove any deleted rules from the match set
228  if(prev_psize > xcsf->pset.size) {
229  int prev_msize = mset->size;
230  set_validate(xcsf, mset);
231  // if the deleted classifier was in the match set,
232  // check if an action is now not covered
233  if(prev_msize > mset->size) {
234  for(int i = 0; i < xcsf->num_actions; i++) {
235  if(!set_action_covered(xcsf, mset, i)) {
236  act_covered[i] = false;
237  again = true;
238  }
239  }
240  }
241  }
242  attempts++;
243  if(attempts > MAX_COVER) {
244  printf("Error: maximum covering attempts (%d) exceeded\n", MAX_COVER);
245  exit(EXIT_FAILURE);
246  }
247 
248  } while(again);
249 }

258 void set_pred(XCSF *xcsf, SET *set, double *x, double *p)
259 {
260  double *presum = calloc(xcsf->num_y_vars, sizeof(double));
261  double fitsum = 0;
262 #ifdef PARALLEL_PRED
263  CLIST *blist[set->size];
264  int j = 0;
265  for(CLIST *iter = set->list; iter != NULL; iter = iter->next) {
266  blist[j] = iter;
267  j++;
268  }
269 #pragma omp parallel for reduction(+:presum[:xcsf->num_y_vars],fitsum)
270  for(int i = 0; i < set->size; i++) {
271  double *predictions = cl_predict(xcsf, blist[i]->cl, x);
272  for(int var = 0; var < xcsf->num_y_vars; var++) {
273  presum[var] += predictions[var] * blist[i]->cl->fit;
274  }
275  fitsum += blist[i]->cl->fit;
276  }
277 #pragma omp parallel for
278  for(int var = 0; var < xcsf->num_y_vars; var++) {
279  p[var] = presum[var]/fitsum;
280  }
281 #else
282  for(CLIST *iter = set->list; iter != NULL; iter = iter->next) {
283  double *predictions = cl_predict(xcsf, iter->cl, x);
284  for(int var = 0; var < xcsf->num_y_vars; var++) {
285  presum[var] += predictions[var] * iter->cl->fit;
286  }
287  fitsum += iter->cl->fit;
288  }
289  for(int var = 0; var < xcsf->num_y_vars; var++) {
290  p[var] = presum[var]/fitsum;
291  }
292 #endif
293  free(presum);
294 }

417 void set_subsumption(XCSF *xcsf, SET *set, SET *kset)
418 {
419  CL *s = NULL;
420  CLIST *iter;
421  // find the most general subsumer in the set
422  for(iter = set->list; iter != NULL; iter = iter->next) {
423  CL *c = iter->cl;
424  if(cl_subsumer(xcsf, c) && (s == NULL || cl_general(xcsf, c, s))) {
425  s = c;
426  }
427  }
428  // subsume the more specific classifiers in the set
429  if(s != NULL) {
430  iter = set->list;
431  while(iter != NULL) {
432  CL *c = iter->cl;
433  iter = iter->next;
434  if(s != c && cl_general(xcsf, s, c)) {
435  s->num += c->num;
436  c->num = 0;
437  set_add(xcsf, kset, c);
438  set_validate(xcsf, set);
439  set_validate(xcsf, &xcsf->pset);
440  }
441  }
442  }
443 }

450 void set_validate(XCSF *xcsf, SET *set)
451 {
452  (void)xcsf;
453  set->size = 0;
454  set->num = 0;
455  CLIST *prev = NULL;
456  CLIST *iter = set->list;
457  while(iter != NULL) {
458  if(iter->cl == NULL || iter->cl->num == 0) {
459  if(prev == NULL) {
460  set->list = iter->next;
461  free(iter);
462  iter = set->list;
463  }
464  else {
465  prev->next = iter->next;
466  free(iter);
467  iter = prev->next;
468  }
469  }
470  else {
471  set->size++;
472  set->num += iter->cl->num;
473  prev = iter;
474  iter = iter->next;
475  }
476  }

588 double set_avg_mut(XCSF *xcsf, SET *set, int m)
589 {
590  // return the fixed value if not adapted

592  switch(m) {

594  case 1: return xcsf->P_MUTATION;

602  int cnt = 0;

604  sum += cl_mutation_rate(xcsf, iter->cl, m);
605  cnt++;
606  }
607  return sum / cnt;
