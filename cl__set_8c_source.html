91 void pop_del(XCSF *xcsf, SET *kset)
92 {
93  double avg_fit = set_total_fit(xcsf, &xcsf->pset) / xcsf->pset.num;
94  double total = 0;
95  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
96  total += cl_del_vote(xcsf, iter->cl, avg_fit);
97  }
98  CLIST *del = NULL;
99  CLIST *delprev = NULL;
100  int delsize = 0;
101  for(int i = 0; i < 2; i++) {
102  double p = rand_uniform(0,total);
103  double sum = 0;
104  CLIST *prev = NULL;
105  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
106  sum += cl_del_vote(xcsf, iter->cl, avg_fit);
107  if(sum > p) {
108  int size = cl_cond_size(xcsf, iter->cl) + cl_pred_size(xcsf, iter->cl);
109  if(del == NULL || size > delsize) {
110  del = iter;
111  delprev = prev;
112  delsize = size;
113  }
114  break;
115  }
116  prev = iter;
117  }
118  }
119  // decrement numerosity
120  (del->cl->num)--;
121  (xcsf->pset.num)--;
122  // macro classifier must be deleted
123  if(del->cl->num == 0) {
124  set_add(xcsf, kset, del->cl);
125  (xcsf->pset.size)--;
126  if(delprev == NULL) {
127  xcsf->pset.list = del->next;
128  }
129  else {
130  delprev->next = del->next;
131  }
132  free(del);
133  }
134 }

159 void set_match(XCSF *xcsf, SET *mset, SET *kset, double *x)
160 {
161  _Bool act_covered[xcsf->num_actions];
162  for(int i = 0; i < xcsf->num_actions; i++) {
163  act_covered[i] = false;
164  }
165 
166  // add classifiers that match the input state to the match set
167 #ifdef PARALLEL_MATCH
168  CLIST *blist[xcsf->pset.size];
169  int j = 0;
170  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
171  blist[j] = iter;
172  j++;
173  }
174  // update current matching conditions setting m flags
175 #pragma omp parallel for
176  for(int i = 0; i < xcsf->pset.size; i++) {
177  cl_match(xcsf, blist[i]->cl, x);
178  }
179  // build m list
180  for(int i = 0; i < xcsf->pset.size; i++) {
181  if(cl_m(xcsf, blist[i]->cl)) {
182  set_add(xcsf, mset, blist[i]->cl);
183  int action = cl_action(xcsf, blist[i]->cl, x);
184  act_covered[action] = true;
185  }
186  }
187 #else
188  for(CLIST *iter = xcsf->pset.list; iter != NULL; iter = iter->next) {
189  if(cl_match(xcsf, iter->cl, x)) {
190  set_add(xcsf, mset, iter->cl);
191  int action = cl_action(xcsf, iter->cl, x);
192  act_covered[action] = true;
193  }
194  }
195 #endif
196 
197  // perform covering if all actions are not represented
198  int attempts = 0;
199  _Bool again;
200  do {
201  again = false;
202  for(int i = 0; i < xcsf->num_actions; i++) {
203  if(!act_covered[i]) {
204  // new classifier with matching condition and action
205  CL *new = malloc(sizeof(CL));
206  cl_init(xcsf, new, (mset->num)+1, xcsf->time);
207  cl_cover(xcsf, new, x, i);
208  set_add(xcsf, &xcsf->pset, new);
209  set_add(xcsf, mset, new);
210  act_covered[i] = true;
211  }
212  }
213  // enforce population size
214  int prev_psize = xcsf->pset.size;
215  pop_enforce_limit(xcsf, kset);
216  // if a macro classifier was deleted, remove any deleted rules from the match set
217  if(prev_psize > xcsf->pset.size) {
218  int prev_msize = mset->size;
219  set_validate(xcsf, mset);
220  // if the deleted classifier was in the match set,
221  // check if an action is now not covered
222  if(prev_msize > mset->size) {
223  for(int i = 0; i < xcsf->num_actions; i++) {
224  if(!set_action_covered(xcsf, mset, i)) {
225  act_covered[i] = false;
226  again = true;
227  }
228  }
229  }
230  }
231  attempts++;
232  if(attempts > MAX_COVER) {
233  printf("Error: maximum covering attempts (%d) exceeded\n", MAX_COVER);
234  exit(EXIT_FAILURE);
235  }
236 
237  } while(again);
238 }

247 void set_pred(XCSF *xcsf, SET *set, double *x, double *p)
248 {
249  double *presum = calloc(xcsf->num_y_vars, sizeof(double));
250  double fitsum = 0;
251 #ifdef PARALLEL_PRED
252  CLIST *blist[set->size];
253  int j = 0;
254  for(CLIST *iter = set->list; iter != NULL; iter = iter->next) {
255  blist[j] = iter;
256  j++;
257  }
258 #pragma omp parallel for reduction(+:presum[:xcsf->num_y_vars],fitsum)
259  for(int i = 0; i < set->size; i++) {
260  double *predictions = cl_predict(xcsf, blist[i]->cl, x);
261  for(int var = 0; var < xcsf->num_y_vars; var++) {
262  presum[var] += predictions[var] * blist[i]->cl->fit;
263  }
264  fitsum += blist[i]->cl->fit;
265  }
266 #pragma omp parallel for
267  for(int var = 0; var < xcsf->num_y_vars; var++) {
268  p[var] = presum[var]/fitsum;
269  }
270 #else
271  for(CLIST *iter = set->list; iter != NULL; iter = iter->next) {
272  double *predictions = cl_predict(xcsf, iter->cl, x);
273  for(int var = 0; var < xcsf->num_y_vars; var++) {
274  presum[var] += predictions[var] * iter->cl->fit;
275  }
276  fitsum += iter->cl->fit;
277  }
278  for(int var = 0; var < xcsf->num_y_vars; var++) {
279  p[var] = presum[var]/fitsum;
280  }
281 #endif
282  free(presum);
283 }

406 void set_subsumption(XCSF *xcsf, SET *set, SET *kset)
407 {
408  CL *s = NULL;
409  CLIST *iter;
410  // find the most general subsumer in the set
411  for(iter = set->list; iter != NULL; iter = iter->next) {
412  CL *c = iter->cl;
413  if(cl_subsumer(xcsf, c)) {
414  if(s == NULL || cl_general(xcsf, c, s)) {
415  s = c;
416  }
417  }
418  }
419  // subsume the more specific classifiers in the set
420  if(s != NULL) {
421  iter = set->list;
422  while(iter != NULL) {
423  CL *c = iter->cl;
424  iter = iter->next;
425  if(s != c && cl_general(xcsf, s, c)) {
426  s->num += c->num;
427  c->num = 0;
428  set_add(xcsf, kset, c);
429  set_validate(xcsf, set);
430  set_validate(xcsf, &xcsf->pset);
431  }
432  }
433  }
434 }

442 {
443  (void)xcsf;
444  set->size = 0;
445  set->num = 0;
446  CLIST *prev = NULL;
447  CLIST *iter = set->list;
448  while(iter != NULL) {
449  if(iter->cl == NULL || iter->cl->num == 0) {
450  if(prev == NULL) {
451  set->list = iter->next;
452  free(iter);
453  iter = set->list;
454  }
455  else {
456  prev->next = iter->next;
457  free(iter);
458  iter = prev->next;
459  }
460  }

462  set->size++;
463  set->num += iter->cl->num;
464  prev = iter;

466  }
