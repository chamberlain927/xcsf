49 void pa_build(const XCSF *xcsf, const SET *set, const double *x)
50 {
51  for(int i = 0; i < xcsf->num_actions; i++) {
52  xcsf->pa[i] = 0;
53  xcsf->nr[i] = 0;
54  }
55  for(const CLIST *iter = set->list; iter != NULL; iter = iter->next) {
56  const double *predictions = cl_predict(xcsf, iter->cl, x);
57  xcsf->pa[iter->cl->action] += predictions[0] * iter->cl->fit;
58  xcsf->nr[iter->cl->action] += iter->cl->fit;
59  }
60  for(int i = 0; i < xcsf->num_actions; i++) {
61  if(xcsf->nr[i] != 0) {
62  xcsf->pa[i] /= xcsf->nr[i];
63  }
64  else {
65  xcsf->pa[i] = 0;
66  }
67  }
68 }
