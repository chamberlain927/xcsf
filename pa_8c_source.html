50 void pa_build(XCSF *xcsf, SET *set, double *x)
51 {
52  for(int i = 0; i < xcsf->num_actions; i++) {
53  xcsf->pa[i] = 0;
54  xcsf->nr[i] = 0;
55  }
56  for(CLIST *iter = set->list; iter != NULL; iter = iter->next) {
57  double *predictions = cl_predict(xcsf, iter->cl, x);
58  xcsf->pa[iter->cl->action] += predictions[0] * iter->cl->fit;
59  xcsf->nr[iter->cl->action] += iter->cl->fit;
60  }
61  for(int i = 0; i < xcsf->num_actions; i++) {
62  if(xcsf->nr[i] != 0) {
63  xcsf->pa[i] /= xcsf->nr[i];
64  }
65  else {
66  xcsf->pa[i] = 0;
67  }
68  }
69 }
