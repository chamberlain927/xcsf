48 void pa_build(const XCSF *xcsf, const double *x)
49 {
50  for(int i = 0; i < xcsf->n_actions; i++) {
51  xcsf->pa[i] = 0;
52  xcsf->nr[i] = 0;
53  }
54  for(const CLIST *iter = xcsf->mset.list; iter != NULL; iter = iter->next) {
55  const double *predictions = cl_predict(xcsf, iter->cl, x);
56  xcsf->pa[iter->cl->action] += predictions[0] * iter->cl->fit;
57  xcsf->nr[iter->cl->action] += iter->cl->fit;
58  }
59  for(int i = 0; i < xcsf->n_actions; i++) {
60  if(xcsf->nr[i] != 0) {
61  xcsf->pa[i] /= xcsf->nr[i];
62  }
63  else {
64  xcsf->pa[i] = 0;
65  }
66  }
67 }
