113 _Bool cond_dgp_general(XCSF *xcsf, CL *c1, CL *c2)
114 {
115  if(c1->exp < xcsf->THETA_SUB || c2->exp < xcsf->THETA_SUB) {
116  return false;
117  }
118  for(int i = 0; i < xcsf->THETA_SUB; i++) {
119  int i1 = (c1->exp + i) % xcsf->THETA_SUB;
120  int i2 = (c2->exp + i) % xcsf->THETA_SUB;
121  if(c1->mhist[i1] == false && c2->mhist[i2] == true) {
122  return false;
123  }
124  }
125  return true;
126 }
