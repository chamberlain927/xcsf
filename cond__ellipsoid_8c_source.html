45 void cond_ellipsoid_init(XCSF *xcsf, CL *c)
46 {
47  COND_ELLIPSOID *new = malloc(sizeof(COND_ELLIPSOID));
48  new->center = malloc(sizeof(double)*xcsf->num_x_vars);
49  new->spread = malloc(sizeof(double)*xcsf->num_x_vars);
50  for(int i = 0; i < xcsf->num_x_vars; i++) {
51  new->center[i] = rand_uniform(xcsf->MIN_CON, xcsf->MAX_CON);

54  c->cond = new;

106 double cond_ellipsoid_dist(XCSF *xcsf, CL *c, double *x)
107 {
108  COND_ELLIPSOID *cond = c->cond;
109  double dist = 0;
110  for(int i = 0; i < xcsf->num_x_vars; i++) {
111  double d = (x[i] - cond->center[i]) / cond->spread[i];
112  dist += d*d; // squared distance
113  }
114  return dist;
115 }

117 _Bool cond_ellipsoid_crossover(XCSF *xcsf, CL *c1, CL *c2)
118 {
119  COND_ELLIPSOID *cond1 = c1->cond;
120  COND_ELLIPSOID *cond2 = c2->cond;
121  _Bool changed = false;
122  if(rand_uniform(0,1) < xcsf->P_CROSSOVER) {
123  for(int i = 0; i < xcsf->num_x_vars; i++) {
124  if(rand_uniform(0,1) < 0.5) {
125  double tmp = cond1->center[i];
126  cond1->center[i] = cond2->center[i];
127  cond2->center[i] = tmp;
128  changed = true;
129  }
130  if(rand_uniform(0,1) < 0.5) {
131  double tmp = cond1->spread[i];
132  cond1->spread[i] = cond2->spread[i];
133  cond2->spread[i] = tmp;
134  changed = true;
135  }
136  }
137  }
138  return changed;
139 }

141 _Bool cond_ellipsoid_mutate(XCSF *xcsf, CL *c)
142 {
143  COND_ELLIPSOID *cond = c->cond;
144  _Bool changed = false;
145  for(int i = 0; i < xcsf->num_x_vars; i++) {
146  double orig = cond->center[i];
147  cond->center[i] += rand_normal(0, xcsf->S_MUTATION);
148  if(cond->center[i] < xcsf->MIN_CON) {
149  cond->center[i] = xcsf->MIN_CON;
150  }
151  else if(cond->center[i] > xcsf->MAX_CON) {
152  cond->center[i] = xcsf->MAX_CON;
153  }
154  if(orig != cond->center[i]) {
155  changed = true;
156  }
157  }
158  for(int i = 0; i < xcsf->num_x_vars; i++) {
159  double orig = cond->spread[i];
160  cond->spread[i] += rand_normal(0, xcsf->S_MUTATION);
161  if(orig != cond->spread[i]) {
162  changed = true;
163  }
164  }
165  return changed;
166 }

168 _Bool cond_ellipsoid_general(XCSF *xcsf, CL *c1, CL *c2)
169 {
170  // returns whether cond1 is more general than cond2
171  COND_ELLIPSOID *cond1 = c1->cond;
172  COND_ELLIPSOID *cond2 = c2->cond;
173  for(int i = 0; i < xcsf->num_x_vars; i++) {
174  double l1 = cond1->center[i] - cond1->spread[i];
175  double l2 = cond2->center[i] - cond2->spread[i];
176  double u1 = cond1->center[i] + cond1->spread[i];
177  double u2 = cond2->center[i] + cond2->spread[i];
178  if(l1 > l2 || u1 < u2) {
179  return false;
180  }
181  }
182  return true;
183 }
