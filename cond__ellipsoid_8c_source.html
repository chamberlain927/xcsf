45 void cond_ellipsoid_init(XCSF *xcsf, CL *c)
46 {
47  COND_ELLIPSOID *new = malloc(sizeof(COND_ELLIPSOID));
48  new->center = malloc(sizeof(double) * xcsf->num_x_vars);

108 static double cond_ellipsoid_dist(XCSF *xcsf, CL *c, const double *x)
109 {
110  COND_ELLIPSOID *cond = c->cond;
111  double dist = 0;
112  for(int i = 0; i < xcsf->num_x_vars; i++) {
113  double d = (x[i] - cond->center[i]) / cond->spread[i];
114  dist += d*d; // squared distance
115  }
116  return dist;

119 _Bool cond_ellipsoid_crossover(XCSF *xcsf, CL *c1, CL *c2)
120 {
121  COND_ELLIPSOID *cond1 = c1->cond;
122  COND_ELLIPSOID *cond2 = c2->cond;
123  _Bool changed = false;
124  if(rand_uniform(0,1) < xcsf->P_CROSSOVER) {
125  for(int i = 0; i < xcsf->num_x_vars; i++) {
126  if(rand_uniform(0,1) < 0.5) {
127  double tmp = cond1->center[i];
128  cond1->center[i] = cond2->center[i];
129  cond2->center[i] = tmp;
130  changed = true;
131  }
132  if(rand_uniform(0,1) < 0.5) {
133  double tmp = cond1->spread[i];
134  cond1->spread[i] = cond2->spread[i];
135  cond2->spread[i] = tmp;
136  changed = true;
137  }
138  }
139  }
140  return changed;
141 }

143 _Bool cond_ellipsoid_mutate(XCSF *xcsf, CL *c)
144 {
145  COND_ELLIPSOID *cond = c->cond;
146  _Bool changed = false;
147  for(int i = 0; i < xcsf->num_x_vars; i++) {
148  // centers
149  double orig = cond->center[i];
150  cond->center[i] += rand_normal(0, xcsf->S_MUTATION);
151  cond->center[i] = constrain(xcsf->COND_MIN, xcsf->COND_MAX, cond->center[i]);
152  if(orig != cond->center[i]) {
153  changed = true;
154  }
155  // spreads
156  orig = cond->spread[i];
157  cond->spread[i] += rand_normal(0, xcsf->S_MUTATION);
158  if(orig != cond->spread[i]) {
159  changed = true;
160  }
161  }
162  return changed;
163 }

165 _Bool cond_ellipsoid_general(XCSF *xcsf, CL *c1, CL *c2)
166 {
167  // returns whether cond1 is more general than cond2
168  COND_ELLIPSOID *cond1 = c1->cond;
169  COND_ELLIPSOID *cond2 = c2->cond;
170  for(int i = 0; i < xcsf->num_x_vars; i++) {
171  double l1 = cond1->center[i] - cond1->spread[i];
172  double l2 = cond2->center[i] - cond2->spread[i];
173  double u1 = cond1->center[i] + cond1->spread[i];
174  double u2 = cond2->center[i] + cond2->spread[i];
175  if(l1 > l2 || u1 < u2) {
176  return false;
177  }
178  }
179  return true;
180 }
